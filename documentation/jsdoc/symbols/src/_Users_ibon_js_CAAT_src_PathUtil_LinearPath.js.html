 10     aliases:["CAAT.LinearPath"],
 11     extendsClass:"CAAT.PathUtil.PathSegment",
 12     extendsWith:function () {
 13 
 14         return  {
 15             __init:function () {
 16                 this.__super();
 17 
 18                 this.points = [];
 19                 this.points.push(new CAAT.Math.Point());
 20                 this.points.push(new CAAT.Math.Point());
 21 
 22                 this.newPosition = new CAAT.Math.Point(0, 0, 0);
 23                 return this;
 24             },
 25 
 26             points:null,
 27             newPosition:null, // spare holder for getPosition coordinate return.
 28 
 29             applyAsPath:function (director) {
 30                 // Fixed: Thanks https://github.com/roed
 31                 director.ctx.lineTo(this.points[1].x, this.points[1].y);
 32             },
 33             setPoint:function (point, index) {
 34                 if (index === 0) {
 35                     this.points[0] = point;
 36                 } else if (index === 1) {
 37                     this.points[1] = point;
 38                 }
 39             },
 40             /**
 41              * Update this segments length and bounding box info.
 42              */
 43             updatePath:function (point) {
 44                 var x = this.points[1].x - this.points[0].x;
 45                 var y = this.points[1].y - this.points[0].y;
 46                 this.length = Math.sqrt(x * x + y * y);
 47 
 48                 this.bbox.setEmpty();
 49                 this.bbox.union(this.points[0].x, this.points[0].y);
 50                 this.bbox.union(this.points[1].x, this.points[1].y);
 51 
 52                 return this;
 53             },
 54             setPoints:function (points) {
 55                 this.points[0] = points[0];
 56                 this.points[1] = points[1];
 57                 this.updatePath();
 58                 return this;
 59             },
 60             /**
 61              * Set this path segment's starting position.
 62              * @param x {number}
 63              * @param y {number}
 64              */
 65             setInitialPosition:function (x, y) {
 66                 this.points[0].x = x;
 67                 this.points[0].y = y;
 68                 this.newPosition.set(x, y);
 69                 return this;
 70             },
 71             /**
 72              * Set this path segment's ending position.
 73              * @param finalX {number}
 74              * @param finalY {number}
 75              */
 76             setFinalPosition:function (finalX, finalY) {
 77                 this.points[1].x = finalX;
 78                 this.points[1].y = finalY;
 79                 return this;
 80             },
 81             /**
 82              * @inheritDoc
 83              */
 84             endCurvePosition:function () {
 85                 return this.points[1];
 86             },
 87             /**
 88              * @inheritsDoc
 89              */
 90             startCurvePosition:function () {
 91                 return this.points[0];
 92             },
 93             /**
 94              * @inheritsDoc
 95              */
 96             getPosition:function (time) {
 97 
 98                 if (time > 1 || time < 0) {
 99                     time %= 1;
100                 }
101                 if (time < 0) {
102                     time = 1 + time;
103                 }
104 
105                 this.newPosition.set(
106                     (this.points[0].x + (this.points[1].x - this.points[0].x) * time),
107                     (this.points[0].y + (this.points[1].y - this.points[0].y) * time));
108 
109                 return this.newPosition;
110             },
111             getPositionFromLength:function (len) {
112                 return this.getPosition(len / this.length);
113             },
114             /**
115              * Returns initial path segment point's x coordinate.
116              * @return {number}
117              */
118             initialPositionX:function () {

120             },
121             /**
122              * Returns final path segment point's x coordinate.
123              * @return {number}
124              */
125             finalPositionX:function () {
126                 return this.points[1].x;
127             },
128             /**
129              * Draws this path segment on screen. Optionally it can draw handles for every control point, in
130              * this case, start and ending path segment points.
131              * @param director {CAAT.Director}
132              * @param bDrawHandles {boolean}
133              */
134             paint:function (director, bDrawHandles) {
135 
136                 var ctx = director.ctx;
137 
138                 ctx.save();
139 
140                 ctx.strokeStyle = this.color;
141                 ctx.beginPath();
142                 ctx.moveTo(this.points[0].x, this.points[0].y);
143                 ctx.lineTo(this.points[1].x, this.points[1].y);
144                 ctx.stroke();
145 
146                 if (bDrawHandles) {
147                     ctx.globalAlpha = 0.5;
148                     ctx.fillStyle = '#7f7f00';
149                     ctx.beginPath();
150                     this.drawHandle(ctx, this.points[0].x, this.points[0].y);
151                     this.drawHandle(ctx, this.points[1].x, this.points[1].y);
152 
153                 }
154 
155                 ctx.restore();
156             },
157             /**
158              * Get the number of control points. For this type of path segment, start and
159              * ending path segment points. Defaults to 2.
160              * @return {number}
161              */
162             numControlPoints:function () {
163                 return 2;
164             },
165             /**
166              * @inheritsDoc
167              */
168             getControlPoint:function (index) {
169                 if (0 === index) {
170                     return this.points[0];
171                 } else if (1 === index) {
172                     return this.points[1];
173                 }
174             },
175             /**
176              * @inheritsDoc
177              */
178             getContour:function (iSize) {
179                 var contour = [];
180

182                 contour.push(this.getPosition(1).clone());

185             }
