 50 		/**
 51 		 * Removes a circle from the simulations
 52 		 * @param aCircle	Circle to remove
 53 		 */
 54 		removeCircle: function(aCircle)
 55 		{
 56 			var index = 0,
 57 				found = false,
 58 				len = this.allCircles.length;
 59 
 60 			if(len === 0) {
 61 				throw "Error: (PackedCircleManager) attempting to remove circle, and allCircles.length === 0!!";
 62 			}
 63 
 64 			while (len--) {
 65 				if(this.allCircles[len] === aCircle) {
 66 					found = true;
 67 					index = len;
 68 					break;
 69 				}
 70 			}
 71 
 72 			if(!found) {
 73 				throw "Could not locate circle in allCircles array!";
 74 			}
 75 
 76 			// Remove
 77 			this.allCircles[index].dealloc();
 78 			this.allCircles[index] = null;
 79 
 80 			return this;
 81 		},

 83 		/**
 84 		 * Forces all circles to move to where their delegate position is
 85 		 * Assumes all targets have a 'position' property!
 86 		 */
 87 		forceCirclesToMatchDelegatePositions: function()
 88 		{
 89 			var len = this.allCircles.length;
 90 
 91 			// push toward target position
 92 			for(var n = 0; n < len; n++)
 93 			{
 94 				var aCircle = this.allCircles[n];
 95 				if(!aCircle || !aCircle.delegate) {
 96 					continue;
 97 				}
 98 
 99 				aCircle.position.set(aCircle.delegate.x + aCircle.offset.x,
100 						aCircle.delegate.y + aCircle.offset.y);
101 			}
102 		},

104 		pushAllCirclesTowardTarget: function(aTarget)
105 		{
106 			var v = new CAAT.Math.Point(0,0,0),
107 				circleList = this.allCircles,
108 				len = circleList.length;
109 
110 			// push toward target position
111 			for(var n = 0; n < this.numberOfTargetingPasses; n++)
112 			{
113 				for(var i = 0; i < len; i++)
114 				{
115 					var c = circleList[i];
116 
117 					if(c.isFixed) continue;
118 
119 					v.x = c.position.x - (c.targetPosition.x+c.offset.x);
120 					v.y = c.position.y - (c.targetPosition.y+c.offset.y);
121 					v.multiply(c.targetChaseSpeed);
122 
123 					c.position.x -= v.x;
124 					c.position.y -= v.y;
125 				}
126 			}
127 		},

129 		/**
130 		 * Packs the circles towards the center of the bounds.
131 		 * Each circle will have it's own 'targetPosition' later on
132 		 */
133 		handleCollisions: function()
134 		{
135 			this.removeExpiredElements();
136 
137 			var v = new CAAT.Math.Point(0,0, 0),
138 				circleList = this.allCircles,
139 				len = circleList.length;
140 
141 			// Collide circles
142 			for(var n = 0; n < this.numberOfCollisionPasses; n++)
143 			{
144 				for(var i = 0; i < len; i++)
145 				{
146 					var ci = circleList[i];
147 
148 
149 					for (var j = i + 1; j< len; j++)
150 					{
151 						var cj = circleList[j];
152 
153 						if( !this.circlesCanCollide(ci, cj) ) continue;   // It's us!
154 
155 						var dx = cj.position.x - ci.position.x,
156 							dy = cj.position.y - ci.position.y;
157 
158 						// The distance between the two circles radii, but we're also gonna pad it a tiny bit
159 						var r = (ci.radius + cj.radius) * 1.08,
160 							d = ci.position.getDistanceSquared(cj.position);
161 
162 						/**
163 						 * Collision detected!
164 						 */
165 						if (d < (r * r) - 0.02 )
166 						{
167 							v.x = dx;
168 							v.y = dy;
169 							v.normalize();
170 
171 							var inverseForce = (r - Math.sqrt(d)) * 0.5;
172 							v.multiply(inverseForce);
173 
174 							// Move cj opposite of the collision as long as its not fixed
175 							if(!cj.isFixed)
176 							{
177 								if(ci.isFixed)
178 									v.multiply(2.2);	// Double inverse force to make up for the fact that the other object is fixed
179 
180 								// ADD the velocity
181 								cj.position.translatePoint(v);
182 							}
183 
184 							// Move ci opposite of the collision as long as its not fixed
185 							if(!ci.isFixed)
186 							{
187 								if(cj.isFixed)
188 									v.multiply(2.2);	// Double inverse force to make up for the fact that the other object is fixed
189 
190 								 // SUBTRACT the velocity
191 								ci.position.subtract(v);
192 							}
193 
194 							// Emit the collision event from each circle, with itself as the first parameter
195 //							if(this.dispatchCollisionEvents && n == this.numberOfCollisionPasses-1)
196 //							{
197 //								this.eventEmitter.emit('collision', cj, ci, v);
198 //							}
199 						}
200 					}
201 				}
202 			}
203 		},

205 		handleBoundaryForCircle: function(aCircle, boundsRule)
206 		{
207 //			if(aCircle.boundsRule === true) return; // Ignore if being dragged
208 
209 			var xpos = aCircle.position.x;
210 			var ypos = aCircle.position.y;
211 
212 			var radius = aCircle.radius;
213 			var diameter = radius*2;
214 
215 			// Toggle these on and off,
216 			// Wrap and bounce, are opposite behaviors so pick one or the other for each axis, or bad things will happen.
217 			var wrapXMask = 1 << 0;
218 			var wrapYMask = 1 << 2;
219 			var constrainXMask = 1 << 3;
220 			var constrainYMask = 1 << 4;
221 			var emitEvent = 1 << 5;
222 
223 			// TODO: Promote to member variable
224 			// Convert to bitmask - Uncomment the one you want, or concact your own :)
225 	//		boundsRule = wrapY; // Wrap only Y axis
226 	//		boundsRule = wrapX; // Wrap only X axis
227 	//		boundsRule = wrapXMask | wrapYMask; // Wrap both X and Y axis
228 			boundsRule = wrapYMask | constrainXMask;  // Wrap Y axis, but constrain horizontally
229 
230 			// Wrap X
231 			if(boundsRule & wrapXMask && xpos-diameter > this.bounds.right) {
232 				aCircle.position.x = this.bounds.left + radius;
233 			} else if(boundsRule & wrapXMask && xpos+diameter < this.bounds.left) {
234 				aCircle.position.x = this.bounds.right - radius;
235 			}
236 			// Wrap Y
237 			if(boundsRule & wrapYMask && ypos-diameter > this.bounds.bottom) {
238 				aCircle.position.y = this.bounds.top - radius;
239 			} else if(boundsRule & wrapYMask && ypos+diameter < this.bounds.top) {
240 				aCircle.position.y = this.bounds.bottom + radius;
241 			}
242 
243 			// Constrain X
244 			if(boundsRule & constrainXMask && xpos+radius >= this.bounds.right) {
245 				aCircle.position.x = aCircle.position.x = this.bounds.right-radius;
246 			} else if(boundsRule & constrainXMask && xpos-radius < this.bounds.left) {
247 				aCircle.position.x = this.bounds.left + radius;
248 			}
249 
250 			// Constrain Y
251 			if(boundsRule & constrainYMask && ypos+radius > this.bounds.bottom) {
252 				aCircle.position.y = this.bounds.bottom - radius;
253 			} else if(boundsRule & constrainYMask && ypos-radius < this.bounds.top) {
254 				aCircle.position.y = this.bounds.top + radius;
255 			}
256 		},

258 		/**
259 		 * Given an x,y position finds circle underneath and sets it to the currently grabbed circle
260 		 * @param {Number} xpos		An x position
261 		 * @param {Number} ypos		A y position
262 		 * @param {Number} buffer	A radiusSquared around the point in question where something is considered to match
263 		 */
264 		getCircleAt: function(xpos, ypos, buffer)
265 		{
266 			var circleList = this.allCircles;
267 			var len = circleList.length;
268 			var grabVector = new CAAT.Math.Point(xpos, ypos, 0);
269 
270 			// These are set every time a better match i found
271 			var closestCircle = null;
272 			var closestDistance = Number.MAX_VALUE;
273 
274 			// Loop thru and find the closest match
275 			for(var i = 0; i < len; i++)
276 			{
277 				var aCircle = circleList[i];
278 				if(!aCircle) continue;
279 				var distanceSquared = aCircle.position.getDistanceSquared(grabVector);
280 
281 				if(distanceSquared < closestDistance && distanceSquared < aCircle.radiusSquared + buffer)
282 				{
283 					closestDistance = distanceSquared;
284 					closestCircle = aCircle;
285 				}
286 			}
287 
288 			return closestCircle;
289 		},

331 		sortOnDistanceToTarget: function(circleA, circleB)
332 		{
333 			var valueA = circleA.getDistanceSquaredFromPosition(circleA.targetPosition);
334 			var valueB = circleB.getDistanceSquaredFromPosition(circleA.targetPosition);
335 			var comparisonResult = 0;
336 
337 			if(valueA > valueB) comparisonResult = -1;
338 			else if(valueA < valueB) comparisonResult = 1;
339 
340 			return comparisonResult;
341 		},
