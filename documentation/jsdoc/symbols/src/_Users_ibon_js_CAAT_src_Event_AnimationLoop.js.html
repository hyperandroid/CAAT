1 CAAT.Module({
  2     defines:"CAAT.Event.AnimationLoop",
  3     onCreate : function() {
  4         CAAT.ENDRAF=false; // if RAF, this value signals end of RAF.
  5         CAAT.INTERVAL_ID=null; // if setInterval, this value holds CAAT.setInterval return value.
  6         CAAT.renderEnabled=false; // Boolean flag to determine if CAAT.loop has already been called.
  7         CAAT.FPS=60; // expected FPS when using setInterval animation.
  8         CAAT.NO_RAF=0; // Use RAF shim instead of setInterval.
  9         CAAT.FPS_REFRESH=500; // debug panel update time.
 10         CAAT.RAF=0; // requestAnimationFrame time reference.
 11         CAAT.REQUEST_ANIMATION_FRAME_TIME=0; // time between two consecutive RAF. usually bigger than FRAME_TIME
 12         CAAT.SET_INTERVAL=0; // time between two consecutive setInterval calls.
 13         CAAT.FRAME_TIME=0; // time to process one frame.
 14         CAAT.currentDirector=null; // Current animated director.
 15         CAAT.director=[];
 16 
 17         /**
 18          * Register and keep track of every CAAT.Director instance in the document.
 19          */
 20         CAAT.RegisterDirector=function (director) {
 21             if (!CAAT.currentDirector) {
 22                 CAAT.currentDirector = director;
 23             }
 24             CAAT.director.push(director);
 25         };
 26 
 27         /**
 28          * Return current scene.
 29          * @return {CAAT.Foundation.Scene}
 30          */
 31         CAAT.getCurrentScene=function () {
 32             return CAAT.currentDirector.getCurrentScene();
 33         };
 34 
 35         /**
 36          * Return current director's current scene's time.
 37          * The way to go should be keep local scene references, but anyway, this function is always handy.
 38          * @return {number} current scene's virtual time.
 39          */
 40         CAAT.getCurrentSceneTime=function () {
 41             return CAAT.currentDirector.getCurrentScene().time;
 42         };
 43 
 44         /**
 45          * Stop animation loop.
 46          */
 47         CAAT.endLoop=function () {
 48             if (CAAT.NO_RAF) {
 49                 if (CAAT.INTERVAL_ID !== null) {
 50                     clearInterval(CAAT.INTERVAL_ID);
 51                 }
 52             } else {
 53                 CAAT.ENDRAF = true;
 54             }
 55 
 56             CAAT.renderEnabled = false;
 57         };
 58 
 59         /**
 60          * Main animation loop entry point.
 61          * Must called only once, or only after endLoop.
 62          *
 63          * @param fps {number} desired fps. fps parameter will only be used if CAAT.NO_RAF is specified, that is
 64          * switch from RequestAnimationFrame to setInterval for animation loop.
 65          */
 66         CAAT.loop=function (fps) {
 67             if (CAAT.renderEnabled) {
 68                 return;
 69             }
 70 
 71             for (var i = 0, l = CAAT.director.length; i < l; i++) {
 72                 CAAT.director[i].timeline = new Date().getTime();
 73             }
 74 
 75             CAAT.FPS = fps || 60;
 76             CAAT.renderEnabled = true;
 77             if (CAAT.NO_RAF) {
 78                 CAAT.INTERVAL_ID = setInterval(
 79                     function () {
 80                         var t = new Date().getTime();
 81 
 82                         for (var i = 0, l = CAAT.director.length; i < l; i++) {
 83                             var dr = CAAT.director[i];
 84                             if (dr.renderMode === CAAT.Foundation.Director.RENDER_MODE_CONTINUOUS || dr.needsRepaint) {
 85                                 dr.renderFrame();
 86                             }
 87                         }
 88 
 89                         CAAT.FRAME_TIME = t - CAAT.SET_INTERVAL;
 90 
 91                         if (CAAT.RAF) {
 92                             CAAT.REQUEST_ANIMATION_FRAME_TIME = new Date().getTime() - CAAT.RAF;
 93                         }
 94                         CAAT.RAF = new Date().getTime();
 95 
 96                         CAAT.SET_INTERVAL = t;
 97 
 98                     },
 99                     1000 / CAAT.FPS
100                 );
101             } else {
102                 CAAT.renderFrameRAF();
103             }
104         };
105         
106         CAAT.renderFrameRAF= function () {
107             var c= CAAT;
108 
109             if (c.ENDRAF) {
110                 c.ENDRAF = false;
111                 return;
112             }
113 
114             var t = new Date().getTime();
115             for (var i = 0, l = c.director.length; i < l; i++) {
116                 c.director[i].renderFrame();
117             }
118             t = new Date().getTime() - t;
119             c.FRAME_TIME = t;
120 
121             if (c.RAF) {
122                 c.REQUEST_ANIMATION_FRAME_TIME = new Date().getTime() - c.RAF;
123             }
124             c.RAF = new Date().getTime();
125 
126             window.requestAnimFrame(c.renderFrameRAF, 0);
127         };
128         
129         /**
130          * Polyfill for requestAnimationFrame.
131          */
132         window.requestAnimFrame = (function () {
133             return  window.requestAnimationFrame ||
134                 window.webkitRequestAnimationFrame ||
135                 window.mozRequestAnimationFrame ||
136                 window.oRequestAnimationFrame ||
137                 window.msRequestAnimationFrame ||
138                 function raf(/* function */ callback, /* DOMElement */ element) {
139                     window.setTimeout(callback, 1000 / CAAT.FPS);
140                 };
141         })();        
142     },
143 
144     extendsWith:function () {
145         return {
146         };
147     }
148 });
