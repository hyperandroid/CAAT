  6 CAAT.Module({
  7     defines:"CAAT.Math.Curve",
  8     depends:["CAAT.Math.Point"],
  9     extendsWith:function () {
 10 
 11         return {
 12             coordlist:null,
 13             k:0.05,
 14             length:-1,
 15             interpolator:false,
 16             HANDLE_SIZE:20,
 17             drawHandles:true,
 18 
 19             /**
 20              * Paint the curve control points.
 21              * @param director {CAAT.Director}
 22              */
 23             paint:function (director) {
 24                 if (false === this.drawHandles) {
 25                     return;
 26                 }
 27 
 28                 var cl = this.coordlist;
 29                 var ctx = director.ctx;
 30 
 31                 // control points
 32                 ctx.save();
 33                 ctx.beginPath();
 34 
 35                 ctx.strokeStyle = '#a0a0a0';
 36                 ctx.moveTo(cl[0].x, cl[0].y);
 37                 ctx.lineTo(cl[1].x, cl[1].y);
 38                 ctx.stroke();
 39                 if (this.cubic) {
 40                     ctx.moveTo(cl[2].x, cl[2].y);
 41                     ctx.lineTo(cl[3].x, cl[3].y);
 42                     ctx.stroke();
 43                 }
 44 
 45 
 46                 ctx.globalAlpha = 0.5;
 47                 for (var i = 0; i < this.coordlist.length; i++) {
 48                     ctx.fillStyle = '#7f7f00';
 49                     var w = this.HANDLE_SIZE / 2;
 50                     ctx.beginPath();
 51                     ctx.arc(cl[i].x, cl[i].y, w, 0, 2 * Math.PI, false);
 52                     ctx.fill();
 53                 }
 54 
 55                 ctx.restore();
 56             },
 57             /**
 58              * Signal the curve has been modified and recalculate curve length.
 59              */
 60             update:function () {
 61                 this.calcLength();
 62             },
 63             /**
 64              * This method must be overriden by subclasses. It is called whenever the curve must be solved for some time=t.
 65              * The t parameter must be in the range 0..1
 66              * @param point {CAAT.Point} to store curve solution for t.
 67              * @param t {number}
 68              * @return {CAAT.Point} the point parameter.
 69              */
 70             solve:function (point, t) {
 71             },
 72             /**
 73              * Get an array of points defining the curve contour.
 74              * @param numSamples {number} number of segments to get.
 75              */
 76             getContour:function (numSamples) {
 77                 var contour = [], i;
 78 
 79                 for (i = 0; i <= numSamples; i++) {
 80                     var point = new CAAT.Math.Point();
 81                     this.solve(point, i / numSamples);
 82                     contour.push(point);
 83                 }
 84 
 85                 return contour;
 86             },
 87             /**
 88              * Calculates a curve bounding box.
 89              *
 90              * @param rectangle {CAAT.Rectangle} a rectangle to hold the bounding box.
 91              * @return {CAAT.Rectangle} the rectangle parameter.
 92              */
 93             getBoundingBox:function (rectangle) {
 94                 if (!rectangle) {
 95                     rectangle = new CAAT.Math.Rectangle();
 96                 }
 97 
 98                 // thanks yodesoft.com for spotting the first point is out of the BB
 99                 rectangle.setEmpty();
100                 rectangle.union(this.coordlist[0].x, this.coordlist[0].y);
101 
102                 var pt = new CAAT.Math.Point();
103                 for (var t = this.k; t <= 1 + this.k; t += this.k) {
104                     this.solve(pt, t);
105                     rectangle.union(pt.x, pt.y);
106                 }
107 
108                 return rectangle;
109             },
110             /**
111              * Calculate the curve length by incrementally solving the curve every substep=CAAT.Curve.k. This value defaults
112              * to .05 so at least 20 iterations will be performed.
113              *
114              * @return {number} the approximate curve length.
115              */
116             calcLength:function () {
117                 var x1, y1;
118                 x1 = this.coordlist[0].x;
119                 y1 = this.coordlist[0].y;
120                 var llength = 0;
121                 var pt = new CAAT.Math.Point();
122                 for (var t = this.k; t <= 1 + this.k; t += this.k) {
123                     this.solve(pt, t);
124                     llength += Math.sqrt((pt.x - x1) * (pt.x - x1) + (pt.y - y1) * (pt.y - y1));
125                     x1 = pt.x;
126                     y1 = pt.y;
127                 }
128 
129                 this.length = llength;
130                 return llength;
131             },
132             /**
133              * Return the cached curve length.
134              * @return {number} the cached curve length.
135              */
136             getLength:function () {
137                 return this.length;
138             },
139             /**
140              * Return the first curve control point.
141              * @return {CAAT.Point}
142              */
143             endCurvePosition:function () {
144                 return this.coordlist[ this.coordlist.length - 1 ];
145             },
146             /**
147              * Return the last curve control point.
148              * @return {CAAT.Point}
149              */
150             startCurvePosition:function () {
151                 return this.coordlist[ 0 ];
152             },
153 
154             setPoints:function (points) {
155             },
156 
157             setPoint:function (point, index) {
158                 if (index >= 0 && index < this.coordlist.length) {
159                     this.coordlist[index] = point;
160                 }
161             },
162             /**
163              *
164              * @param director <=CAAT.Director>
165              */
166             applyAsPath:function (director) {
167             }
168         }
169     }
170 
171 });
