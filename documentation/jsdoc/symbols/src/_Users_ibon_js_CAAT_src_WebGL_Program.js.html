  5 CAAT.Module( {
  6     defines : "CAAT.WebGL.Program",
  7     extendsWith : {
  8 
  9         __init : function(gl) {
 10             this.gl= gl;
 11             return this;
 12         },
 13 
 14         shaderProgram:  null,
 15         gl:             null,
 16 
 17         /**
 18          * Set fragment shader's alpha composite value.
 19          * @param alpha {number} float value 0..1.
 20          */
 21         setAlpha : function( alpha ) {
 22 
 23         },
 24         getShader : function (gl,type,str) {
 25             var shader;
 26             if (type === "x-shader/x-fragment") {
 27                 shader = gl.createShader(gl.FRAGMENT_SHADER);
 28             } else if (type === "x-shader/x-vertex") {
 29                 shader = gl.createShader(gl.VERTEX_SHADER);
 30             } else {
 31                 return null;
 32             }
 33 
 34             gl.shaderSource(shader, str);
 35             gl.compileShader(shader);
 36 
 37             if (!gl.getShaderParameter(shader, gl.COMPILE_STATUS)) {
 38                 alert(gl.getShaderInfoLog(shader));
 39                 return null;
 40             }
 41 
 42             return shader;
 43 
 44         },
 45         getDomShader : function(gl, id) {
 46             var shaderScript = document.getElementById(id);
 47             if (!shaderScript) {
 48                 return null;
 49             }
 50 
 51             var str = "";
 52             var k = shaderScript.firstChild;
 53             while (k) {
 54                 if (k.nodeType === 3) {
 55                     str += k.textContent;
 56                 }
 57                 k = k.nextSibling;
 58             }
 59 
 60             var shader;
 61             if (shaderScript.type === "x-shader/x-fragment") {
 62                 shader = gl.createShader(gl.FRAGMENT_SHADER);
 63             } else if (shaderScript.type === "x-shader/x-vertex") {
 64                 shader = gl.createShader(gl.VERTEX_SHADER);
 65             } else {
 66                 return null;
 67             }
 68 
 69             gl.shaderSource(shader, str);
 70             gl.compileShader(shader);
 71 
 72             if (!gl.getShaderParameter(shader, gl.COMPILE_STATUS)) {
 73                 alert(gl.getShaderInfoLog(shader));
 74                 return null;
 75             }
 76 
 77             return shader;
 78         },
 79         initialize : function() {
 80             return this;
 81         },
 82         getFragmentShader : function() {
 83             return null;
 84         },
 85         getVertexShader : function() {
 86             return null;
 87         },
 88         create : function() {
 89             var gl= this.gl;
 90 
 91             this.shaderProgram = gl.createProgram();
 92             gl.attachShader(this.shaderProgram, this.getVertexShader());
 93             gl.attachShader(this.shaderProgram, this.getFragmentShader());
 94             gl.linkProgram(this.shaderProgram);
 95             gl.useProgram(this.shaderProgram);
 96             return this;
 97         },
 98         setMatrixUniform : function( caatMatrix4 ) {
 99             this.gl.uniformMatrix4fv(
100                     this.shaderProgram.pMatrixUniform,
101                     false,
102                     new Float32Array(caatMatrix4.flatten()));
103 
104         },
105         useProgram : function() {
106             this.gl.useProgram(this.shaderProgram);
107             return this;
108         }
109     }
110 });
