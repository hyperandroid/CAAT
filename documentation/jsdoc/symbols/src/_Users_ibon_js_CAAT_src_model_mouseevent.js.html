433 /**
434  * Enable window level input events, keys and redimension.
435  */
436 CAAT.GlobalEnableEvents= function __GlobalEnableEvents() {
437 
438     if ( CAAT.GlobalEventsEnabled ) {
439         return;
440     }
441 
442     this.GlobalEventsEnabled= true;
443 
444     window.addEventListener('keydown',
445         function(evt) {
446             var key = (evt.which) ? evt.which : evt.keyCode;
447 
448             if ( key===CAAT.SHIFT_KEY ) {
449                 CAAT.KEY_MODIFIERS.shift= true;
450             } else if ( key===CAAT.CONTROL_KEY ) {
451                 CAAT.KEY_MODIFIERS.control= true;
452             } else if ( key===CAAT.ALT_KEY ) {
453                 CAAT.KEY_MODIFIERS.alt= true;
454             } else {
455                 for( var i=0; i<CAAT.keyListeners.length; i++ ) {
456                     CAAT.keyListeners[i]( new CAAT.KeyEvent(
457                         key,
458                         'down',
459                         {
460                             alt:        CAAT.KEY_MODIFIERS.alt,
461                             control:    CAAT.KEY_MODIFIERS.control,
462                             shift:      CAAT.KEY_MODIFIERS.shift
463                         },
464                         evt)) ;
465                 }
466             }
467         },
468         false);
469 
470     window.addEventListener('keyup',
471         function(evt) {
472 
473             var key = (evt.which) ? evt.which : evt.keyCode;
474             if ( key===CAAT.SHIFT_KEY ) {
475                 CAAT.KEY_MODIFIERS.shift= false;
476             } else if ( key===CAAT.CONTROL_KEY ) {
477                 CAAT.KEY_MODIFIERS.control= false;
478             } else if ( key===CAAT.ALT_KEY ) {
479                 CAAT.KEY_MODIFIERS.alt= false;
480             } else {
481 
482                 for( var i=0; i<CAAT.keyListeners.length; i++ ) {
483                     CAAT.keyListeners[i]( new CAAT.KeyEvent(
484                         key,
485                         'up',
486                         {
487                             alt:        CAAT.KEY_MODIFIERS.alt,
488                             control:    CAAT.KEY_MODIFIERS.control,
489                             shift:      CAAT.KEY_MODIFIERS.shift
490                         },
491                         evt));
492                 }
493             }
494         },
495         false );
496 
497     window.addEventListener('resize',
498         function(evt) {
499             for( var i=0; i<CAAT.windowResizeListeners.length; i++ ) {
500                 CAAT.windowResizeListeners[i].windowResized(
501                         window.innerWidth,
502                         window.innerHeight);
503             }
504         },
505         false);
506 };

523 /**
524  * Main animation loop entry point.
525  * @param fps {number} desired fps. This parameter makes no sense unless requestAnimationFrame function
526  * is not present in the system.
527  */
528 CAAT.loop= function(fps) {
529     if (CAAT.renderEnabled) {
530         return;
531     }
532 
533 
534     CAAT.FPS= fps || 60;
535     CAAT.renderEnabled= true;
536     if (CAAT.NO_RAF) {
537         setInterval(
538                 function() {
539                     var t= new Date().getTime();
540                     for (var i = 0, l = CAAT.director.length; i < l; i++) {
541                         CAAT.director[i].renderFrame();
542                     }
543                     //t= new Date().getTime()-t;
544                     CAAT.FRAME_TIME= t - CAAT.SET_INTERVAL;
545                     
546                     CAAT.SET_INTERVAL= t;
547 
548                 },
549                 1000 / CAAT.FPS
550         );
551     } else {
552         CAAT.renderFrame();
553     }
554 }

561 CAAT.FPS_REFRESH= 500;  // debug panel update time.
562 CAAT.RAF= 0;            // requestAnimationFrame time reference.
563 CAAT.REQUEST_ANIMATION_FRAME_TIME=   0;
564 /**
565  * Make a frame for each director instance present in the system.
566  */
567 CAAT.renderFrame= function() {
568     var t= new Date().getTime();
569     for( var i=0, l=CAAT.director.length; i<l; i++ ) {
570         var dr= CAAT.director[i];
571         if ( dr.renderMode===CAAT.Director.RENDER_MODE_CONTINUOUS || dr.needsRepaint ) {
572             dr.renderFrame();
573         }
574     }
575     t= new Date().getTime()-t;
576     CAAT.FRAME_TIME= t;
577 
578     if (CAAT.RAF)   {
579         CAAT.REQUEST_ANIMATION_FRAME_TIME= new Date().getTime()-CAAT.RAF;
580     }
581     CAAT.RAF= new Date().getTime();
582 
583     window.requestAnimFrame(CAAT.renderFrame, 0 );
584 }

609  * Enable at window level accelerometer events.

611 (function() {
612 
613     function tilt(data) {
614         CAAT.rotationRate= {
615                 alpha : 0,
616                 beta  : data[0],
617                 gamma : data[1]
618             };
619     }
620 
621     if (window.DeviceOrientationEvent) {
622         window.addEventListener("deviceorientation", function (event) {
623             tilt([event.beta, event.gamma]);
624         }, true);
625     } else if (window.DeviceMotionEvent) {
626         window.addEventListener('devicemotion', function (event) {
627             tilt([event.acceleration.x * 2, event.acceleration.y * 2]);
628         }, true);
629     } else {
630         window.addEventListener("MozOrientation", function (event) {
631             tilt([-event.y * 45, event.x * 45]);
632         }, true);
633     }
634 
635 })();
