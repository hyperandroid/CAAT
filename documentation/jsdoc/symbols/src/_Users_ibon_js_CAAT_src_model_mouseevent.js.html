443 /**
444  * Enable window level input events, keys and redimension.
445  */
446 CAAT.GlobalEnableEvents= function __GlobalEnableEvents() {
447 
448     if ( CAAT.GlobalEventsEnabled ) {
449         return;
450     }
451 
452     this.GlobalEventsEnabled= true;
453 
454     window.addEventListener('keydown',
455         function(evt) {
456             var key = (evt.which) ? evt.which : evt.keyCode;
457 
458             if ( key===CAAT.SHIFT_KEY ) {
459                 CAAT.KEY_MODIFIERS.shift= true;
460             } else if ( key===CAAT.CONTROL_KEY ) {
461                 CAAT.KEY_MODIFIERS.control= true;
462             } else if ( key===CAAT.ALT_KEY ) {
463                 CAAT.KEY_MODIFIERS.alt= true;
464             } else {
465                 for( var i=0; i<CAAT.keyListeners.length; i++ ) {
466                     CAAT.keyListeners[i]( new CAAT.KeyEvent(
467                         key,
468                         'down',
469                         {
470                             alt:        CAAT.KEY_MODIFIERS.alt,
471                             control:    CAAT.KEY_MODIFIERS.control,
472                             shift:      CAAT.KEY_MODIFIERS.shift
473                         },
474                         evt)) ;
475                 }
476             }
477         },
478         false);
479 
480     window.addEventListener('keyup',
481         function(evt) {
482 
483             var key = (evt.which) ? evt.which : evt.keyCode;
484             if ( key===CAAT.SHIFT_KEY ) {
485                 CAAT.KEY_MODIFIERS.shift= false;
486             } else if ( key===CAAT.CONTROL_KEY ) {
487                 CAAT.KEY_MODIFIERS.control= false;
488             } else if ( key===CAAT.ALT_KEY ) {
489                 CAAT.KEY_MODIFIERS.alt= false;
490             } else {
491 
492                 for( var i=0; i<CAAT.keyListeners.length; i++ ) {
493                     CAAT.keyListeners[i]( new CAAT.KeyEvent(
494                         key,
495                         'up',
496                         {
497                             alt:        CAAT.KEY_MODIFIERS.alt,
498                             control:    CAAT.KEY_MODIFIERS.control,
499                             shift:      CAAT.KEY_MODIFIERS.shift
500                         },
501                         evt));
502                 }
503             }
504         },
505         false );
506 
507     window.addEventListener('resize',
508         function(evt) {
509             for( var i=0; i<CAAT.windowResizeListeners.length; i++ ) {
510                 CAAT.windowResizeListeners[i].windowResized(
511                         window.innerWidth,
512                         window.innerHeight);
513             }
514         },
515         false);
516 };

534 CAAT.endLoop= function() {
535     if ( CAAT.NO_RAF ) {
536         if ( CAAT.INTERVAL_ID!==null ) {
537             clearInterval( CAAT.INTERVAL_ID );
538         }
539     } else {
540         CAAT.ENDRAF= true;
541     }
542 
543     CAAT.renderEnabled= false;
544 }

549 /**
550  * Main animation loop entry point.
551  * @param fps {number} desired fps. This parameter makes no sense unless requestAnimationFrame function
552  * is not present in the system.
553  */
554 CAAT.loop= function(fps) {
555     if (CAAT.renderEnabled) {
556         return;
557     }
558 
559     for (var i = 0, l = CAAT.director.length; i < l; i++) {
560         CAAT.director[i].timeline= new Date().getTime();
561     }
562 
563     CAAT.FPS= fps || 60;
564     CAAT.renderEnabled= true;
565     if (CAAT.NO_RAF) {
566         CAAT.INTERVAL_ID= setInterval(
567                 function() {
568                     var t= new Date().getTime();
569 
570                     for( var i=0, l=CAAT.director.length; i<l; i++ ) {
571                         var dr= CAAT.director[i];
572                         if ( dr.renderMode===CAAT.Director.RENDER_MODE_CONTINUOUS || dr.needsRepaint ) {
573                             dr.renderFrame();
574                         }
575                     }
576 
577                     CAAT.FRAME_TIME= t - CAAT.SET_INTERVAL;
578 
579                     if (CAAT.RAF)   {
580                         CAAT.REQUEST_ANIMATION_FRAME_TIME= new Date().getTime()-CAAT.RAF;
581                     }
582                     CAAT.RAF= new Date().getTime();
583 
584                     CAAT.SET_INTERVAL= t;
585 
586                 },
587                 1000 / CAAT.FPS
588         );
589     } else {
590         CAAT.renderFrame();
591     }
592 }

602 /**
603  * Make a frame for each director instance present in the system.
604  */
605 CAAT.renderFrame= function() {
606     if ( CAAT.ENDRAF ) {
607         CAAT.ENDRAF= false;
608         return;
609     }
610 
611     var t= new Date().getTime();
612     for( var i=0, l=CAAT.director.length; i<l; i++ ) {
613         var dr= CAAT.director[i];
614         if ( dr.renderMode===CAAT.Director.RENDER_MODE_CONTINUOUS || dr.needsRepaint ) {
615             dr.renderFrame();
616         }
617     }
618     t= new Date().getTime()-t;
619     CAAT.FRAME_TIME= t;
620 
621     if (CAAT.RAF)   {
622         CAAT.REQUEST_ANIMATION_FRAME_TIME= new Date().getTime()-CAAT.RAF;
623     }
624     CAAT.RAF= new Date().getTime();
625 
626     window.requestAnimFrame(CAAT.renderFrame, 0 );
627 }

653  * Enable at window level accelerometer events.

655 (function() {
656 
657     function tilt(data) {
658         CAAT.rotationRate= {
659                 alpha : 0,
660                 beta  : data[0],
661                 gamma : data[1]
662             };
663     }
664 
665     if (window.DeviceOrientationEvent) {
666         window.addEventListener("deviceorientation", function (event) {
667             tilt([event.beta, event.gamma]);
668         }, true);
669     } else if (window.DeviceMotionEvent) {
670         window.addEventListener('devicemotion', function (event) {
671             tilt([event.acceleration.x * 2, event.acceleration.y * 2]);
672         }, true);
673     } else {
674         window.addEventListener("MozOrientation", function (event) {
675             tilt([-event.y * 45, event.x * 45]);
676         }, true);
677     }
