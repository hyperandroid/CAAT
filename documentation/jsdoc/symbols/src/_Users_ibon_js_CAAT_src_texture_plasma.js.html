 17     /**
 18      * ImageProcessor is a class to encapsulate image processing operations. These image processing
 19      * manipulates individual image pixels and from an array of pixels builds an image which can
 20      * be used as a pattern or image.
 21      * <p>
 22      * This class pre-creates a canvas of the given dimensions and extracts an imageData object to
 23      * hold the pixel manipulation.
 24      *
 25      * @constructor
 26      */
 27     CAAT.ImageProcessor= function() {
 28         return this;
 29     };

 31     CAAT.ImageProcessor.prototype= {
 32         canvas:     null,
 33         ctx:        null,
 34         width:      0,
 35         height:     0,
 36         imageData:  null,
 37         bufferImage:null,
 38 
 39         /**
 40          * Grabs an image pixels.
 41          *
 42          * @param image {HTMLImageElement}
 43          * @return {ImageData} returns an ImageData object with the image representation or null in
 44          * case the pixels can not be grabbed.
 45          *
 46          * @static
 47          */
 48         grabPixels : function(image) {
 49             var canvas= document.createElement('canvas');
 50             if ( canvas!==null ) {
 51                 canvas.width= image.width;
 52                 canvas.height= image.height;
 53                 var ctx= canvas.getContext('2d');
 54                 ctx.drawImage(image,0,0);
 55                 try {
 56                     var imageData= ctx.getImageData(0,0,canvas.width,canvas.height);
 57                     return imageData;
 58                 }
 59                 catch(e) {
 60                     CAAT.log('error pixelgrabbing.', image);
 61                     return null;
 62                 }
 63             }
 64             return null;
 65         },
 66         /**
 67          * Helper method to create an array.
 68          *
 69          * @param size {number} integer number of elements in the array.
 70          * @param initValue {number} initial array values.
 71          *
 72          * @return {[]} an array of 'initialValue' elements.
 73          *
 74          * @static
 75          */
 76         makeArray : function(size, initValue) {
 77             var a= [];
 78 
 79             for(var i=0; i<size; i++ )  {
 80                 a.push( initValue );
 81             }
 82 
 83             return a;
 84         },
 85         /**
 86          * Helper method to create a bidimensional array.
 87          *
 88          * @param size {number} number of array rows.
 89          * @param size2 {number} number of array columns.
 90          * @param initvalue array initial values.
 91          *
 92          * @return {[]} a bidimensional array of 'initvalue' elements.
 93          *
 94          * @static
 95          *
 96          */
 97         makeArray2D : function (size, size2, initvalue)  {
 98             var a= [];
 99 
100             for( var i=0; i<size; i++ ) {
101                 a.push( this.makeArray(size2,initvalue) );
102             }
103 
104             return a;
105         },
106         /**
107          * Initializes and creates an offscreen Canvas object. It also creates an ImageData object and
108          * initializes the internal bufferImage attribute to imageData's data.
109          * @param width {number} canvas width.
110          * @param height {number} canvas height.
111          * @return this
112          */
113         initialize : function(width,height) {
114 
115             this.width=  width;
116             this.height= height;
117 
118             this.canvas= document.createElement('canvas');
119             if ( this.canvas!==null ) {
120                 this.canvas.width= width;
121                 this.canvas.height= height;
122                 this.ctx= this.canvas.getContext('2d');
123                 this.imageData= this.ctx.getImageData(0,0,width,height);
124                 this.bufferImage= this.imageData.data;
125             }
126 
127             return this;
128         },
129         /**
130          * Clear this ImageData object to the given color components.
131          * @param r {number} red color component 0..255.
132          * @param g {number} green color component 0..255.
133          * @param b {number} blue color component 0..255.
134          * @param a {number} alpha color component 0..255.
135          * @return this
136          */
137         clear : function( r,g,b,a ) {
138             if ( null===this.imageData ) {
139                 return this;
140             }
141             var data= this.imageData.data;
142             for( var i=0; i<this.width*this.height; i++ ) {
143                 data[i*4+0]= r;
144                 data[i*4+1]= g;
145                 data[i*4+2]= b;
146                 data[i*4+3]= a;
147             }
148 this.imageData.data= data;
149 
150             return this;
151         },
152         /**
153          * Get this ImageData.
154          * @return {ImageData}
155          */
156         getImageData : function() {
157             return this.ctx.getImageData(0,0,this.width,this.height);
158         },
159         /**
160          * Sets canvas pixels to be the applied effect. After process pixels, this method must be called
161          * to show the result of such processing.
162          * @param director {CAAT.Director}
163          * @param time {number}
164          * @return this
165          */
166         apply : function(director, time) {
167             if ( null!==this.imageData ) {
168 this.imageData.data= this.bufferImage;
169                 this.ctx.putImageData(this.imageData, 0, 0);
170             }
171             return this;
172         },
173         /**
174          * Returns the offscreen canvas.
175          * @return {HTMLCanvasElement}
176          */
177         getCanvas : function() {
178             return this.canvas;
179         },
180         /**
181          * Creates a pattern that will make this ImageProcessor object suitable as a fillStyle value.
182          * This effect can be drawn too as an image by calling: canvas_context.drawImage methods.
183          * @param type {string} the pattern type. if no value is supplied 'repeat' will be used.
184          * @return CanvasPattern.
185          */
186         createPattern : function( type ) {
187             return this.ctx.createPattern(this.canvas,type ? type : 'repeat');
188         },
189         /**
190          * Paint this ImageProcessor object result.
191          * @param director {CAAT.Director}.
192          * @param time {number} scene time.
193          */
194         paint : function( director, time ) {
195             if ( null!==this.canvas ) {
196                 var ctx= director.ctx;
197                 ctx.drawImage( this.getCanvas(), 0, 0 );
198             }
199         }

207      * Creates an additive plasma wave image.
208      *
209      * @constructor
210      * @extendsClass CAAT.ImageProcessor

213     CAAT.IMPlasma= function() {
214         CAAT.IMPlasma.superclass.constructor.call(this);
215         return this;

217 
218     CAAT.IMPlasma.prototype= {
219         wavetable: null,
220         m_colorMap: null,
221         spd1: 1,
222         spd2: 2,
223         spd3: 3,
224         spd4: 4,
225         pos1: 0,
226         pos2: 0,
227         pos3: 0,
228         pos4: 0,
229         tpos1: 0,
230         tpos2: 0,
231         tpos3: 0,
232         tpos4: 0,
233         m_colorMapSize: 256,
234         i1: 0,
235         i2: 0,
236         i3: 0,
237         i4: 0,
238         b1: false,
239         b2: false,
240         b3: false,
241         b4: false,

243         color: [0xffffffff, 0xffff00ff, 0xffffff00, 0xff00ff00, 0xffff0000, 0xff0000ff, 0xff000000],
244 
245         /**
246          * Initialize the plasma image processor.
247          * <p>
248          * This image processor creates a color ramp of 256 elements from the colors of the parameter 'colors'.
249          * Be aware of color definition since the alpha values count to create the ramp.
250          *
251          * @param width {number}
252          * @param height {number}
253          * @param colors {Array.<number>} an array of color values.
254          *
255          * @return this
256          */
257         initialize : function(width,height,colors) {
258             CAAT.IMPlasma.superclass.initialize.call(this,width,height);
259 
260             this.wavetable= [];
261             for (var x=0; x<256; x++)   {
262                 this.wavetable.push( Math.floor(32 * (1 + Math.cos(x*2 * Math.PI / 256))) );
263             }
264 
265             this.pos1=Math.floor(255*Math.random());
266             this.pos2=Math.floor(255*Math.random());
267             this.pos3=Math.floor(255*Math.random());
268             this.pos4=Math.floor(255*Math.random());
269 
270             this.m_colorMap= CAAT.Color.prototype.makeRGBColorRamp(
271                     colors!==null ? colors : this.color,
272                     256,
273                     CAAT.Color.prototype.RampEnumeration.RAMP_CHANNEL_RGBA_ARRAY );
274 
275             this.setB();
276 
277             return this;
278         },
279         /**
280          * Initialize internal plasma structures. Calling repeatedly this method will make the plasma
281          * look different.
282          */
283         setB : function() {
284 
285             this.b1= Math.random()>0.5;
286             this.b2= Math.random()>0.5;
287             this.b3= Math.random()>0.5;
288             this.b4= Math.random()>0.5;
289 
290             this.spd1= Math.floor((Math.random()*3+1)*(Math.random()<0.5?1:-1));
291             this.spd2= Math.floor((Math.random()*3+1)*(Math.random()<0.5?1:-1));
292             this.spd3= Math.floor((Math.random()*3+1)*(Math.random()<0.5?1:-1));
293             this.spd4= Math.floor((Math.random()*3+1)*(Math.random()<0.5?1:-1));
294 
295             this.i1= Math.floor((Math.random()*2.4+1)*(Math.random()<0.5?1:-1));
296             this.i2= Math.floor((Math.random()*2.4+1)*(Math.random()<0.5?1:-1));
297             this.i3= Math.floor((Math.random()*2.4+1)*(Math.random()<0.5?1:-1));
298             this.i4= Math.floor((Math.random()*2.4+1)*(Math.random()<0.5?1:-1));
299         },
300         /**
301          * Apply image processing to create the plasma and call superclass's apply to make the result
302          * visible.
303          * @param director {CAAT.Director}
304          * @param time {number}
305          *
306          * @return this
307          */
308         apply : function(director,time) {
309 
310             var v = 0;
311 	        this.tpos1 = this.pos1;
312 	        this.tpos2 = this.pos2;
313 
314             var bi= this.bufferImage;
315             var cm= this.m_colorMap;
316             var wt= this.wavetable;
317             var z;
318             var cmz;
319 
320 	        for (var x=0; x<this.height; x++) {
321                 this.tpos3 = this.pos3;
322                 this.tpos4 = this.pos4;
323 
324                 for(var y=0; y<this.width; y++) {
325                     // mix at will, or at your own risk.
326                     var o1= this.tpos1+this.tpos2+this.tpos3;
327                     var o2= this.tpos2+this.tpos3-this.tpos1;
328                     var o3= this.tpos3+this.tpos4-this.tpos2;
329                     var o4= this.tpos4+this.tpos1-this.tpos2;
330 
331                     // set different directions. again, change at will.
332                     if ( this.b1 ) o1= -o1;
333                     if ( this.b2 ) o2= -o2;
334                     if ( this.b3 ) o3= -o3;
335                     if ( this.b4 ) o4= -o4;
336 
337                     z = Math.floor( wt[o1&255] + wt[o2&255] + wt[o3&255] + wt[o4&255] );
338                     cmz= cm[z];
339 
340                     bi[ v++ ]= cmz[0];
341                     bi[ v++ ]= cmz[1];
342                     bi[ v++ ]= cmz[2];
343                     bi[ v++ ]= cmz[3];
344 
345                     this.tpos3 += this.i1;
346                     this.tpos3&=255;
347                     this.tpos4 += this.i2;
348                     this.tpos4&=255;
349                 }
350 
351                 this.tpos1 += this.i3;
352                 this.tpos1&=255;
353                 this.tpos2 += this.i4;
354                 this.tpos2&=255;
355             }
356 
357             this.pos1 += this.spd1;
358             this.pos2 -= this.spd2;
359             this.pos3 += this.spd3;
360             this.pos4 -= this.spd4;
361             this.pos1&=255;
362             this.pos3&=255;
363             this.pos2&=255;
364             this.pos4&=255;
365 
366             return CAAT.IMPlasma.superclass.apply.call(this,director,time);
367         }
368     };

376     /**
377      * This class creates a bumpy effect from a given image. The effect can be applied by different lights
378      * each of which can bump the image with a different color. The lights will have an additive color
379      * effect on affected pixels.
380      *
381      * @constructor
382      * @extendsClass CAAT.ImageProcessor
383      */
384     CAAT.IMBump=function() {
385         CAAT.IMBump.superclass.constructor.call(this);
386         return this;
387     };
388 
389     CAAT.IMBump.prototype= {
390 
391         // bump
392         m_avgX:         null,
393         m_avgY:         null,
394         m_tt:           null,
395         phong:          null,
396 
397         m_radius:       75,
398 
399         m_lightcolor:   null,
400         bcolor:         false,
401         lightPosition:  [],
402 
403         /**
404          * Initializes internal bump effect data.
405          *
406          * @param image {HTMLImageElement}
407          * @param radius {number} lights radius.
408          *
409          * @private
410          */
411         prepareBump : function(image, radius) {
412             var i,j;
413 
414             this.m_radius= (radius ? radius : 75);
415 
416             var imageData= this.grabPixels(image);
417 
418             this.m_tt= this.makeArray(this.height,0);
419             for( i=0; i<this.height; i++ ){
420                 this.m_tt[ i ]=this.width*i;
421             }
422 
423             this.m_avgX= this.makeArray2D(this.height,this.width,0);
424             this.m_avgY= this.makeArray2D(this.height,this.width,0);
425 
426             var bump=this.makeArray2D(this.height,this.width,0);
427 
428             if ( null===imageData ) {
429                 return;
430             }
431             
432             var sourceImagePixels= imageData.data;
433 
434             for (i=0;i<this.height;i++) {
435                 for (j=0;j<this.width;j++) {
436                     var pos= (i*this.width+j)*4;
437                     bump[i][j]=
438                         sourceImagePixels[pos  ]+
439                         sourceImagePixels[pos+1]+
440                         sourceImagePixels[pos+2];
441                 }
442             }
443 
444             bump= this.soften( bump );
445 
446             for (var x=1;x<this.width-1;x++)    {
447                 for (var y=1;y<this.height-1;y++)   {
448                     this.m_avgX[y][x]=Math.floor(bump[y][x+1]-bump[y][x-1]);
449                     this.m_avgY[y][x]=Math.floor(bump[y+1][x]-bump[y-1][x]);
450                 }
451             }
452 
453             bump=null;
454         },
455         /**
456          * Soften source images extracted data on prepareBump method.
457          * @param bump bidimensional array of black and white source image version.
458          * @return bidimensional array with softened version of source image's b&w representation.
459          */
460         soften : function( bump ) {
461             var temp;
462             var sbump=this.makeArray2D( this.height,this.width, 0);
463 
464             for (var j=0;j<this.width;j++) {
465                 for (var i=0;i<this.height;i++) {
466                     temp=(bump[i][j]);
467                     temp+=(bump[(i+1)%this.height][j]);
468                     temp+=(bump[(i+this.height-1)%this.height][j]);
469                     temp+=(bump[i][(j+1)%this.width]);
470                     temp+=(bump[i][(j+this.width-1)%this.width]);
471                     temp+=(bump[(i+1)%this.height][(j+1)%this.width]);
472                     temp+=(bump[(i+this.height-1)%this.height][(j+this.width-1)%this.width]);
473                     temp+=(bump[(i+this.height-1)%this.height][(j+1)%this.width]);
474                     temp+=(bump[(i+1)%this.height][(j+this.width-1)%this.width]);
475                     temp/=9;
476                     sbump[i][j]=temp/3;
477                 }
478             }
479 
480             return sbump;
481         },
482         /**
483          * Create a phong image to apply bump effect.
484          * @private
485          */
486         calculatePhong : function( ) {
487             this.phong= this.makeArray2D(this.m_radius,this.m_radius,0);
488 
489             var i,j,z;
490             for( i=0; i<this.m_radius; i++ ) {
491                 for( j=0; j<this.m_radius; j++ ) {
492                     var x= j/this.m_radius;
493                     var y= i/this.m_radius;
494                     z= (1-Math.sqrt(x*x+y*y))*0.8;
495                     if ( z<0 ) {
496                         z=0;
497                     }
498                     this.phong[ i ][ j ]= Math.floor(z*255);
499                 }
500             }
501         },
502         /**
503          * Generates a bump image.
504          * @param dstPixels {ImageData.data} destinarion pixel array to store the calculated image.
505          */
506         drawColored : function(dstPixels)	{
507             var i,j,k;
508             for( i=0; i<this.height; i++ ) {
509                 for( j=0; j<this.width; j++ ){
510 
511                     var rrr=0;
512                     var ggg=0;
513                     var bbb=0;
514 
515                     for( k=0; k<this.m_lightcolor.length; k++ ) {
516 
517                         var lx= this.lightPosition[k].x;
518                         var ly= this.lightPosition[k].y;
519 
520                         var dx=Math.floor(Math.abs(this.m_avgX[i][j]-j+lx));
521                         var dy=Math.floor(Math.abs(this.m_avgY[i][j]-i+ly));
522 
523                         if (dx>=this.m_radius) {
524                             dx=this.m_radius-1;
525                         }
526                         if (dy>=this.m_radius) {
527                             dy=this.m_radius-1;
528                         }
529 
530                         var c= this.phong[ dx ] [ dy ];
531                         var r=0;
532                         var g=0;
533                         var b=0;
534 
535                         if ( c>=0 ) {// oscurecer
536                             r= (this.m_lightcolor[k][0]*c/128);
537                             g= (this.m_lightcolor[k][1]*c/128);
538                             b= (this.m_lightcolor[k][2]*c/128);
539                         }
540                         else {			// blanquear.
541                             c=128+c;
542                             var rr= (this.m_lightcolor[k][0]);
543                             var gg= (this.m_lightcolor[k][1]);
544                             var bb= (this.m_lightcolor[k][2]);
545 
546                             r= Math.floor(rr+ (255-rr)*c/128);
547                             g= Math.floor(gg+ (255-gg)*c/128);
548                             b= Math.floor(bb+ (255-bb)*c/128);
549                         }
550 
551                         rrr+=r;
552                         ggg+=g;
553                         bbb+=b;
554                     }
555 
556                     if ( rrr>255 ) {
557                         rrr=255;
558                     }
559                     if ( ggg>255 ) {
560                         ggg=255;
561                     }
562                     if ( bbb>255 ) {
563                         bbb=255;
564                     }
565 
566                     var pos= (j+this.m_tt[i])*4;
567                     dstPixels[pos  ]= rrr;
568                     dstPixels[pos+1]= ggg;
569                     dstPixels[pos+2]= bbb;
570                     dstPixels[pos+3]= 255;
571                 }
572             }
573         },
574         /**
575          * Sets lights color.
576          * @param colors_rgb_array an array of arrays. Each internal array has three integers defining an RGB color.
577          * ie:
578          *  [
579          *     [ 255,0,0 ],
580          *     [ 0,255,0 ]
581          *  ]
582          * @return this
583          */
584         setLightColors : function( colors_rgb_array ) {
585             this.m_lightcolor= colors_rgb_array;
586             this.lightPosition= [];
587             for( var i=0; i<this.m_lightcolor.length; i++ ) {
588                 var x= this.width*Math.random();
589                 var y= this.height*Math.random();
590                 this.lightPosition.push( new CAAT.Point().set(x,y) );
591             }
592             return this;
593         },
594         /**
595          * Initialize the bump image processor.
596          * @param image {HTMLImageElement} source image to bump.
597          * @param radius {number} light radius.
598          */
599         initialize : function(image,radius) {
600             CAAT.IMBump.superclass.initialize.call(this,image.width,image.height);
601 
602             this.setLightColors(
603                     [
604                         [255,128,0],
605                         [0,0,255]
606                     ]);
607 
608             this.prepareBump(image,radius);
609             this.calculatePhong();
610 
611             return this;
612         },
613         /**
614          * Set a light position.
615          * @param lightIndex {number} light index to position.
616          * @param x {number} light x coordinate.
617          * @param y {number} light y coordinate.
618          * @return this
619          */
620         setLightPosition : function( lightIndex, x, y ) {
621             this.lightPosition[lightIndex].set(x,y);
622             return this;
623         },
624         /**
625          * Applies the bump effect and makes it visible on the canvas surface.
626          * @param director {CAAT.Director}
627          * @param time {number}
628          */
629         apply : function(director,time) {
630             this.drawColored(this.bufferImage);
631             return CAAT.IMBump.superclass.apply.call(this,director,time);
632         }
633     };

696                         break;
697                     case 32:
698                         this.mask=31;
699                         this.shift=5;
700                         break;
701                     case 16:
702                         this.mask=15;

710             }
711 
712             this.setCenter();
713 
714             return this;
715         },
716         /**
717          * Performs the process of tiling rotozoom.
718          * @param director {CAAT.Director}
719          * @param time {number}
720          *
721          * @private
722          */
723         rotoZoom: function(director,time)  {
724 
725             var timer = new Date().getTime();
726 
727             var angle=Math.PI*2 * Math.cos(timer * 0.0001);
728             var distance= 600+ 550*Math.sin(timer*0.0002);
729 
730             var dist= this.distortion;
731 
732             var off=0;
733             var ddx=Math.floor(Math.cos(angle)*distance);
734             var ddy=Math.floor(Math.sin(angle)*distance);
735 
736             var hh=0, ww=0;
737 
738             switch( this.m_alignh )	{
739                 case 0:
740                     hh = 0;
741                     break;
742                 case 1:
743                     hh = (this.height >> 1);
744                     break;
745                 case 2:
746                     hh = this.height - 1;
747                     break;
748             }
749 
750             switch (this.m_alignv) {
751                 case 0:
752                     ww = 0;
753                     break;
754                 case 1:
755                     ww = (this.width >> 1);
756                     break;
757                 case 2:
758                     ww = this.width - 1;
759                     break;
760             }
761 
762             var i = (((this.width >> 1) << 8)  - ddx * ww + ddy * hh)&0xffff;
763             var j = (((this.height >> 1) << 8) - ddy * ww - ddx * hh) & 0xffff;
764 
765             var srcwidth=   this.sourceImageData.width;
766             var srcheight=  this.sourceImageData.height;
767             var srcdata=    this.sourceImageData.data;
768             var bi=         this.bufferImage;
769             var dstoff;
770             var addx;
771             var addy;
772 
773             while (off < this.width * this.height * 4) {
774                 addx = i;
775                 addy = j;
776 
777                 for (var m = 0; m < this.width; m++) {
778                     dstoff = ((addy >> this.shift) & this.mask) * srcwidth + ((addx >> this.shift) & this.mask);
779                     dstoff <<= 2;
780 
781                     bi[ off++ ] = srcdata[ dstoff++ ];
782                     bi[ off++ ] = srcdata[ dstoff++ ];
783                     bi[ off++ ] = srcdata[ dstoff++ ];
784                     bi[ off++ ] = srcdata[ dstoff++ ];
785 
786                     addx += ddx;
787                     addy += ddy;
788 
789                 }
790 
791                 dist += this.distortion;
792                 i -= ddy;
793                 j += ddx - dist;
794             }
795         },
796         /**
797          * Perform and apply the rotozoom effect.
798          * @param director {CAAT.Director}
799          * @param time {number}
800          * @return this
801          */
802         apply : function(director,time) {
803             if ( null!==this.sourceImageData ) {
804                 this.rotoZoom(director,time);
805             }
806             return CAAT.IMRotoZoom.superclass.apply.call(this,director,time);
807         },
808         /**
809          * Change the effect's rotation anchor. Call this method repeatedly to make the effect look
810          * different.
811          */
812         setCenter: function() {
813             var d = Math.random();
814             if (d < 0.33) {
815                 this.m_alignv = 0;
816             } else if (d < 0.66) {
817                 this.m_alignv = 1;
818             } else {
819                 this.m_alignv = 2;
820             }
821 
822             d = Math.random();
823             if (d < 0.33) {
824                 this.m_alignh = 0;
825             } else if (d < 0.66) {
826                 this.m_alignh = 1;
827             } else {
828                 this.m_alignh = 2;
829             }
830         }
831 
832     };
833 
834     extend( CAAT.IMRotoZoom, CAAT.ImageProcessor, null);
