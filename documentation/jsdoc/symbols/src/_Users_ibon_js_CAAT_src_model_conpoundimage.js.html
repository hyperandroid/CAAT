 42     /**
 43      *
 44      * This class is used by CAAT.Actor to draw images. It differs from CAAT.CompoundImage in that it
 45      * manages the subimage change based on time and a list of animation sub-image indexes.
 46      * A common use of this class will be:
 47      * 
 48      * <code>
 49      *     var si= new CAAT.SpriteImage().
 50      *          initialize( an_image_instance, rows, columns ).
 51      *          setAnimationImageIndex( [2,1,0,1] ).                // cycle throwout image with these indexes
 52      *          setChangeFPS( 200 ).                                // change sprite every 200 ms.
 53      *          setSpriteTransformation( CAAT.SpriteImage.TR_xx);   // optionally draw images inverted, ...
 54      * </code>
 55      *
 56      * A SpriteImage is an sprite sheet. It encapsulates an Image and treates and references it as a two
 57      * dimensional array of row by columns sub-images. The access form will be sequential so if defined a
 58      * CompoundImage of more than one row, the subimages will be referenced by an index ranging from 0 to
 59      * rows*columns-1. Each sumimage will be of size (image.width/columns) by (image.height/rows).
 60      *
 61      * <p>
 62      * It is able to draw its sub-images in the following ways:
 63      * <ul>
 64      * <li>no transformed (default)
 65      * <li>flipped horizontally
 66      * <li>flipped vertically
 67      * <li>flipped both vertical and horizontally
 68      * </ul>
 69      *
 70      * <p>
 71      * It is supposed to be used in conjunction with <code>CAAT.SpriteActor</code> instances.
 72      *
 73      * @constructor
 74      *
 75      */
 76     CAAT.SpriteImage = function() {
 77         this.paint= this.paintN;
 78         this.setAnimationImageIndex([0]);
 79         this.mapInfo=   {};
 80         return this;
 81     };

183         /**
184          * Initialize a grid of subimages out of a given image.
185          * @param image {HTMLImageElement|Image} an image object.
186          * @param rows {number} number of rows.
187          * @param columns {number} number of columns
188          *
189          * @return this
190          */
191         initialize : function(image, rows, columns) {
192             this.image = image;
193             this.rows = rows;
194             this.columns = columns;
195             this.width = image.width;
196             this.height = image.height;
197             this.singleWidth = Math.floor(this.width / columns);
198             this.singleHeight = Math.floor(this.height / rows);
199             this.mapInfo= {};
200 
201             var i,sx0,sy0;
202             var helper;
203 
204             if (image.__texturePage) {
205                 image.__du = this.singleWidth / image.__texturePage.width;
206                 image.__dv = this.singleHeight / image.__texturePage.height;
207 
208 
209                 var w = this.singleWidth;
210                 var h = this.singleHeight;
211                 var mod = this.columns;
212                 if (image.inverted) {
213                     var t = w;
214                     w = h;
215                     h = t;
216                     mod = this.rows;
217                 }
218 
219                 var xt = this.image.__tx;
220                 var yt = this.image.__ty;
221 
222                 var tp = this.image.__texturePage;
223 
224                 for (i = 0; i < rows * columns; i++) {
225 
226 
227                     var c = ((i % mod) >> 0);
228                     var r = ((i / mod) >> 0);
229 
230                     var u = xt + c * w;  // esquina izq x
231                     var v = yt + r * h;
232 
233                     var u1 = u + w;
234                     var v1 = v + h;
235 
236                     helper= new CAAT.SpriteImageHelper().setGL(
237                         u / tp.width,
238                         v / tp.height,
239                         u1 / tp.width,
240                         v1 / tp.height );
241 
242                     this.mapInfo[i]= helper;
243                 }
244 
245             } else {
246                 for (i = 0; i < rows * columns; i++) {
247                     sx0 = ((i % this.columns) | 0) * this.singleWidth;
248                     sy0 = ((i / this.columns) | 0) * this.singleHeight;
249 
250                     helper= new CAAT.SpriteImageHelper( sx0, sy0, this.singleWidth, this.singleHeight );
251                     this.mapInfo[i]= helper;
252                 }
253             }
254 
255             return this;
256         },
257         /**
258          * Draws the subimage pointed by imageIndex horizontally inverted.
259          * @param canvas a canvas context.
260          * @param imageIndex {number} a subimage index.
261          * @param x {number} x position in canvas to draw the image.
262          * @param y {number} y position in canvas to draw the image.
263          *
264          * @return this
265          */
266         paintInvertedH : function(director, time, x, y) {
267 
268             this.setSpriteIndexAtTime(time);
269 
270             var el= this.mapInfo[this.spriteIndex];
271 
272             var ctx= director.ctx;
273             ctx.save();
274             ctx.translate(((0.5 + x) | 0) + el.width, (0.5 + y) | 0);
275             ctx.scale(-1, 1);
276 
277 
278             ctx.drawImage(
279                 this.image,
280                 el.x, el.y,
281                 el.width, el.height,
282                 this.offsetX>>0, this.offsetY>>0,
283                 el.width, el.height );
284 
285             ctx.restore();
286 
287             return this;

483         /**
484          * Set the transformation to apply to the Sprite image.
485          * Any value of
486          *  <li>TR_NONE
487          *  <li>TR_FLIP_HORIZONTAL
488          *  <li>TR_FLIP_VERTICAL
489          *  <li>TR_FLIP_ALL
490          *
491          * @param transformation an integer indicating one of the previous values.
492          * @return this
493          */
494         setSpriteTransformation : function( transformation ) {
495             this.transformation= transformation;
496             switch(transformation)	{
497 				case this.TR_FLIP_HORIZONTAL:
498 					this.paint= this.paintInvertedH;
499 					break;
500 				case this.TR_FLIP_VERTICAL:
501 					this.paint= this.paintInvertedV;
502 					break;
503 				case this.TR_FLIP_ALL:
504 					this.paint= this.paintInvertedHV;
505 					break;
506                 case this.TR_FIXED_TO_SIZE:
507                     this.paint= this.paintScaled;
508                     break;
509 				default:
510 					this.paint= this.paintN;
511 			}
512             return this;
513         },

516          * Set the sprite animation images index. This method accepts an array of objects which define indexes to

537         /**
538          * Draws the sprite image calculated and stored in spriteIndex.
539          *
540          * @param director the CAAT.Director object instance that contains the Scene the Actor is in.
541          * @param time an integer indicating the Scene time when the bounding box is to be drawn.
542          */
543 		setSpriteIndexAtTime : function(time) {
544 
545             if ( this.animationImageIndex.length>1 ) {
546                 if ( this.prevAnimationTime===-1 )	{
547                     this.prevAnimationTime= time;
548                     this.spriteIndex=0;
549                 }
550                 else	{
551                     var ttime= time;
552                     ttime-= this.prevAnimationTime;
553                     ttime/= this.changeFPS;
554                     ttime%= this.animationImageIndex.length;
555                     this.spriteIndex= this.animationImageIndex[Math.floor(ttime)];
556                 }
557             }
558         },

560         /**
561          * This method takes the output generated from the tool at http://labs.hyperandroid.com/static/texture/spriter.html
562          * and creates a map into that image.
563          * @param image {Image|HTMLImageElement|Canvas} an image
564          * @param map {object} the map into the image to define subimages.
565          */
566         initializeFromMap : function( image, map ) {
567             this.initialize( image, 1, 1 );
568 
569             var key;
570             var helper;
571             var count=0;
572 
573             for( key in map ) {
574                 var value= map[key];
575 
576                 helper= new CAAT.SpriteImageHelper(
577                     value.x,
578                     value.y,
579                     value.width,
580                     value.height
581                 );
582 
583                 this.mapInfo[key]= helper;
584 
585                 // set a default spriteIndex
586                 if ( !count ) {
587                     this.setAnimationImageIndex( [key] );
588                 }
589 
590                 count++;
591             }
592 
593             return this;
594         }
