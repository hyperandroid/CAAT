 43     /**
 44      *
 45      * This class is used by CAAT.Actor to draw images. It differs from CAAT.CompoundImage in that it
 46      * manages the subimage change based on time and a list of animation sub-image indexes.
 47      * A common use of this class will be:
 48      * 
 49      * <code>
 50      *     var si= new CAAT.SpriteImage().
 51      *          initialize( an_image_instance, rows, columns ).
 52      *          setAnimationImageIndex( [2,1,0,1] ).                // cycle throwout image with these indexes
 53      *          setChangeFPS( 200 ).                                // change sprite every 200 ms.
 54      *          setSpriteTransformation( CAAT.SpriteImage.TR_xx);   // optionally draw images inverted, ...
 55      * </code>
 56      *
 57      * A SpriteImage is an sprite sheet. It encapsulates an Image and treates and references it as a two
 58      * dimensional array of row by columns sub-images. The access form will be sequential so if defined a
 59      * CompoundImage of more than one row, the subimages will be referenced by an index ranging from 0 to
 60      * rows*columns-1. Each sumimage will be of size (image.width/columns) by (image.height/rows).
 61      *
 62      * <p>
 63      * It is able to draw its sub-images in the following ways:
 64      * <ul>
 65      * <li>no transformed (default)
 66      * <li>flipped horizontally
 67      * <li>flipped vertically
 68      * <li>flipped both vertical and horizontally
 69      * </ul>
 70      *
 71      * <p>
 72      * It is supposed to be used in conjunction with <code>CAAT.SpriteActor</code> instances.
 73      *
 74      * @constructor
 75      *
 76      */
 77     CAAT.SpriteImage = function() {
 78         this.paint= this.paintN;
 79         this.setAnimationImageIndex([0]);
 80         this.mapInfo=   {};
 81         return this;
 82     };

193         /**
194          * Initialize a grid of subimages out of a given image.
195          * @param image {HTMLImageElement|Image} an image object.
196          * @param rows {number} number of rows.
197          * @param columns {number} number of columns

201         initialize : function(image, rows, columns) {
202             this.image = image;
203             this.rows = rows;
204             this.columns = columns;
205             this.width = image.width;
206             this.height = image.height;
207             this.singleWidth = Math.floor(this.width / columns);
208             this.singleHeight = Math.floor(this.height / rows);
209             this.mapInfo= {};
210 
211             var i,sx0,sy0;
212             var helper;
213 
214             if (image.__texturePage) {
215                 image.__du = this.singleWidth / image.__texturePage.width;
216                 image.__dv = this.singleHeight / image.__texturePage.height;
217 
218 
219                 var w = this.singleWidth;
220                 var h = this.singleHeight;
221                 var mod = this.columns;
222                 if (image.inverted) {
223                     var t = w;
224                     w = h;
225                     h = t;
226                     mod = this.rows;
227                 }
228 
229                 var xt = this.image.__tx;
230                 var yt = this.image.__ty;
231 
232                 var tp = this.image.__texturePage;
233 
234                 for (i = 0; i < rows * columns; i++) {
235 
236 
237                     var c = ((i % mod) >> 0);
238                     var r = ((i / mod) >> 0);
239 
240                     var u = xt + c * w;  // esquina izq x
241                     var v = yt + r * h;
242 
243                     var u1 = u + w;
244                     var v1 = v + h;
245 
246                     helper= new CAAT.SpriteImageHelper(u,v,(u1-u),(v1-v),tp.width,tp.height).setGL(
247                         u / tp.width,
248                         v / tp.height,
249                         u1 / tp.width,
250                         v1 / tp.height );
251 
252                     this.mapInfo[i]= helper;
253                 }
254 
255             } else {
256                 for (i = 0; i < rows * columns; i++) {
257                     sx0 = ((i % this.columns) | 0) * this.singleWidth;
258                     sy0 = ((i / this.columns) | 0) * this.singleHeight;

260                     helper= new CAAT.SpriteImageHelper( sx0, sy0, this.singleWidth, this.singleHeight, image.width, image.height  );
