287                 var tm = this.matrix;

289                 var m00 = tm[0];
290                 var m01 = tm[1];
291                 var m02 = tm[2];
292                 var m10 = tm[3];
293                 var m11 = tm[4];
294                 var m12 = tm[5];
295                 var m20 = tm[6];
296                 var m21 = tm[7];
297                 var m22 = tm[8];
298 
299                 var newMatrix = new CAAT.Math.Matrix();
300 
301                 var determinant = m00 * (m11 * m22 - m21 * m12) - m10 * (m01 * m22 - m21 * m02) + m20 * (m01 * m12 - m11 * m02);
302                 if (determinant === 0) {
303                     return null;
304                 }
305 
306                 var m = newMatrix.matrix;
307 
308                 m[0] = m11 * m22 - m12 * m21;
309                 m[1] = m02 * m21 - m01 * m22;
310                 m[2] = m01 * m12 - m02 * m11;
311 
312                 m[3] = m12 * m20 - m10 * m22;
313                 m[4] = m00 * m22 - m02 * m20;
314                 m[5] = m02 * m10 - m00 * m12;
315 
316                 m[6] = m10 * m21 - m11 * m20;
317                 m[7] = m01 * m20 - m00 * m21;
318                 m[8] = m00 * m11 - m01 * m10;
319 
320                 newMatrix.multiplyScalar(1 / determinant);
321 
322                 return newMatrix;
