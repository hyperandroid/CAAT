357         debugInfo : function( statistics ) {
358             this.statistics= statistics;
359 
360             var cc= CAAT;
361 
362             this.frameTimeAcc+= cc.FRAME_TIME;
363             this.frameRAFAcc+= cc.REQUEST_ANIMATION_FRAME_TIME;
364 
365             /* Update the framerate counter */
366             this.framerate.frames++;
367             if ( cc.RAF > this.framerate.timeLastRefresh + this.framerate.refreshInterval ) {
368                 this.framerate.fps = ( ( this.framerate.frames * 1000 ) / ( cc.RAF - this.framerate.timeLastRefresh ) ) | 0;
369                 this.framerate.fpsMin = this.framerate.frames > 0 ? Math.min( this.framerate.fpsMin, this.framerate.fps ) : this.framerate.fpsMin;
370                 this.framerate.fpsMax = Math.max( this.framerate.fpsMax, this.framerate.fps );
371 
372                 this.textFPS.innerHTML= this.framerate.fps;
373 
374                 var value= ((this.frameTimeAcc*100/this.framerate.frames)|0)/100;
375                 this.frameTimeAcc=0;
376                 this.textDrawTime.innerHTML= value;
377 
378                 var value2= ((this.frameRAFAcc*100/this.framerate.frames)|0)/100;
379                 this.frameRAFAcc=0;
380                 this.textRAFTime.innerHTML= value2;
381 
382                 this.framerate.timeLastRefresh = cc.RAF;
383                 this.framerate.frames = 0;
384 
385                 this.paint(value2);
386             }
387 
388             this.textEntitiesTotal.innerHTML= this.statistics.size_total;
389             this.textEntitiesActive.innerHTML= this.statistics.size_active;
390             this.textDirtyRects.innerHTML= this.statistics.size_dirtyRects;
391             this.textDraws.innerHTML= this.statistics.draws;
392             this.textDiscardDR.innerHTML= this.statistics.size_discarded_by_dirty_rects;
393         },

395         paint : function( rafValue ) {
396             var ctx= this.ctx;
397             var t=0;
398 
399             ctx.drawImage(
400                 this.canvas,
401                 1, 0, this.width-1, this.height,
402                 0, 0, this.width-1, this.height );
403 
404             ctx.strokeStyle= 'black';
405             ctx.beginPath();
406             ctx.moveTo( this.width-.5, 0 );
407             ctx.lineTo( this.width-.5, this.height );
408             ctx.stroke();
409 
410             ctx.strokeStyle= '#a22';
411             ctx.beginPath();
412             t= this.height-((20/this.SCALE*this.height)>>0)-.5;
413             ctx.moveTo( .5, t );
414             ctx.lineTo( this.width+.5, t );
415             ctx.stroke();
416 
417             ctx.strokeStyle= '#aa2';
418             ctx.beginPath();
419             t= this.height-((30/this.SCALE*this.height)>>0)-.5;
420             ctx.moveTo( .5, t );
421             ctx.lineTo( this.width+.5, t );
422             ctx.stroke();
423 
424             var fps = Math.min( this.height-(this.framerate.fps/this.SCALE*this.height), 59 );
425             if (-1===this.framerate.prevFps) {
426                 this.framerate.prevFps= fps|0;
427             }
428 
429             ctx.strokeStyle= '#0ff';//this.framerate.fps<15 ? 'red' : this.framerate.fps<30 ? 'yellow' : 'green';
430             ctx.beginPath();
431             ctx.moveTo( this.width, (fps|0)-.5 );
432             ctx.lineTo( this.width, this.framerate.prevFps-.5 );
433             ctx.stroke();
434 
435             this.framerate.prevFps= fps;
436 
437 
438             var t1= ((this.height-(rafValue/this.SCALE*this.height))>>0)-.5;
439             ctx.strokeStyle= '#ff0';
440             ctx.beginPath();
441             ctx.moveTo( this.width, t1 );
442             ctx.lineTo( this.width, t1 );
443             ctx.stroke();
444         }
