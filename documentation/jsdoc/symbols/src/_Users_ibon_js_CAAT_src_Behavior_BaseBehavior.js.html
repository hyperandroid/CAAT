1 /**
  2  * See LICENSE file.
  3  *
  4  * Behaviors are keyframing elements.
  5  * By using a BehaviorContainer, you can specify different actions on any animation Actor.
  6  * An undefined number of Behaviors can be defined for each Actor.
  7  *
  8  * There're the following Behaviors:
  9  *  + AlphaBehavior:   controls container/actor global alpha.
 10  *  + RotateBehavior:  takes control of rotation affine transform.
 11  *  + ScaleBehavior:   takes control of scaling on x and y axis affine transform.
 12  *  + Scale1Behavior:  takes control of scaling on x or y axis affine transform.
 13  *  + PathBehavior:    takes control of translating an Actor/ActorContainer across a path [ie. pathSegment collection].
 14  *  + GenericBehavior: applies a behavior to any given target object's property, or notifies a callback.
 15  *
 16  *
 17  **/
 18 
 19 CAAT.Module({
 20     defines:        "CAAT.Behavior.BaseBehavior",
 21     constants:      {
 22         Status: {
 23             NOT_STARTED: 0,
 24             STARTED:    1,
 25             EXPIRED:    2
 26         }
 27     },
 28     depends:        ["CAAT.Behavior.Interpolator"],
 29     extendsWith:   function() {
 30 
 31         var DefaultInterpolator=    new CAAT.Behavior.Interpolator().createLinearInterpolator(false);
 32         var DefaultInterpolatorPP=  new CAAT.Behavior.Interpolator().createLinearInterpolator(true);
 33 
 34         return {
 35             /**
 36              * Behavior base class.
 37              *
 38              * <p>
 39              * A behavior is defined by a frame time (behavior duration) and a behavior application function called interpolator.
 40              * In its default form, a behaviour is applied linearly, that is, the same amount of behavior is applied every same
 41              * time interval.
 42              * <p>
 43              * A concrete Behavior, a rotateBehavior in example, will change a concrete Actor's rotationAngle during the specified
 44              * period.
 45              * <p>
 46              * A behavior is guaranteed to notify (if any observer is registered) on behavior expiration.
 47              * <p>
 48              * A behavior can keep an unlimited observers. Observers are objects of the form:
 49              * <p>
 50              * <code>
 51              * {
 52              *      behaviorExpired : function( behavior, time, actor);
 53              *      behaviorApplied : function( behavior, time, normalizedTime, actor, value);
 54              * }
 55              * </code>
 56              * <p>
 57              * <strong>behaviorExpired</strong>: function( behavior, time, actor). This method will be called for any registered observer when
 58              * the scene time is greater than behavior's startTime+duration. This method will be called regardless of the time
 59              * granurality.
 60              * <p>
 61              * <strong>behaviorApplied</strong> : function( behavior, time, normalizedTime, actor, value). This method will be called once per
 62              * frame while the behavior is not expired and is in frame time (behavior startTime>=scene time). This method can be
 63              * called multiple times.
 64              * <p>
 65              * Every behavior is applied to a concrete Actor.
 66              * Every actor must at least define an start and end value. The behavior will set start-value at behaviorStartTime and
 67              * is guaranteed to apply end-value when scene time= behaviorStartTime+behaviorDuration.
 68              * <p>
 69              * You can set behaviors to apply forever that is cyclically. When a behavior is cycle=true, won't notify
 70              * behaviorExpired to its registered observers.
 71              * <p>
 72              * Other Behaviors simply must supply with the method <code>setForTime(time, actor)</code> overriden.
 73              *
 74              * @constructor
 75              */
 76             __init:function () {
 77                 this.lifecycleListenerList = [];
 78                 this.setDefaultInterpolator();
 79                 return this;
 80             },
 81 
 82             lifecycleListenerList:null, // observer list.
 83             behaviorStartTime:-1, // scene time to start applying the behavior
 84             behaviorDuration:-1, // behavior duration in ms.
 85             cycleBehavior:false, // apply forever ?
 86 
 87             status: CAAT.Behavior.BaseBehavior.Status.NOT_STARTED, // Status.NOT_STARTED
 88 
 89             interpolator:null, // behavior application function. linear by default.
 90             actor:null, // actor the Behavior acts on.
 91             id:0, // an integer id suitable to identify this behavior by number.
 92 
 93             timeOffset:0,
 94 
 95             doValueApplication:true,
 96 
 97             solved:true,
 98 
 99             discardable:false, // is true, this behavior will be removed from the this.actor instance when it expires.
100 
101             setValueApplication:function (apply) {
102                 this.doValueApplication = apply;
103                 return this;
104             },
105 
106             setTimeOffset:function (offset) {
107                 this.timeOffset = offset;
108                 return this;
109             },
110 
111             setStatus : function(st) {
112                 this.status= st;
113                 return this;
114             },
115 
116             /**
117              * Sets this behavior id.
118              * @param id an integer.
119              *
120              */
121             setId:function (id) {
122                 this.id = id;
123                 return this;
124             },
125             /**
126              * Sets the default interpolator to a linear ramp, that is, behavior will be applied linearly.
127              * @return this
128              */
129             setDefaultInterpolator:function () {
130                 this.interpolator = DefaultInterpolator;
131                 return this;
132             },
133             /**
134              * Sets default interpolator to be linear from 0..1 and from 1..0.
135              * @return this
136              */
137             setPingPong:function () {
138                 this.interpolator = DefaultInterpolatorPP;
139                 return this;
140             },
141 
142             /**
143              * Sets behavior start time and duration.
144              * Scene time will be the time of the scene the behavior actor is bound to.
145              * @param startTime {number} an integer indicating behavior start time in scene time in ms..
146              * @param duration {number} an integer indicating behavior duration in ms.
147              */
148             setFrameTime:function (startTime, duration) {
149                 this.behaviorStartTime = startTime;
150                 this.behaviorDuration = duration;
151                 this.status =CAAT.Behavior.BaseBehavior.Status.NOT_STARTED;
152 
153                 return this;
154             },
155             /**
156              * Sets behavior start time and duration but instead as setFrameTime which sets initial time as absolute time
157              * regarding scene's time, it uses a relative time offset from current scene time.
158              * a call to
159              *   setFrameTime( scene.time, duration ) is equivalent to
160              *   setDelayTime( 0, duration )
161              * @param delay {number}
162              * @param duration {number}
163              */
164             setDelayTime:function (delay, duration) {
165                 this.behaviorStartTime = delay;
166                 this.behaviorDuration = duration;
167                 this.status =CAAT.Behavior.BaseBehavior.Status.NOT_STARTED;
168                 this.solved = false;
169                 this.expired = false;
170 
171                 return this;
172 
173             },
174             setOutOfFrameTime:function () {
175                 this.status =CAAT.Behavior.BaseBehavior.Status.EXPIRED;
176                 this.behaviorStartTime = Number.MAX_VALUE;
177                 this.behaviorDuration = 0;
178                 return this;
179             },
180             /**
181              * Changes behavior default interpolator to another instance of CAAT.Interpolator.
182              * If the behavior is not defined by CAAT.Interpolator factory methods, the interpolation function must return
183              * its values in the range 0..1. The behavior will only apply for such value range.
184              * @param interpolator a CAAT.Interpolator instance.
185              */
186             setInterpolator:function (interpolator) {
187                 this.interpolator = interpolator;
188                 return this;
189             },
190             /**
191              * This method must no be called directly.
192              * The director loop will call this method in orther to apply actor behaviors.
193              * @param time the scene time the behaviro is being applied at.
194              * @param actor a CAAT.Actor instance the behavior is being applied to.
195              */
196             apply:function (time, actor) {
197 
198                 if (!this.solved) {
199                     this.behaviorStartTime += time;
200                     this.solved = true;
201                 }
202 
203                 time += this.timeOffset * this.behaviorDuration;
204 
205                 var orgTime = time;
206                 if (this.isBehaviorInTime(time, actor)) {
207                     time = this.normalizeTime(time);
208                     this.fireBehaviorAppliedEvent(
209                         actor,
210                         orgTime,
211                         time,
212                         this.setForTime(time, actor));
213                 }
214             },
215 
216             /**
217              * Sets the behavior to cycle, ie apply forever.
218              * @param bool a boolean indicating whether the behavior is cycle.
219              */
220             setCycle:function (bool) {
221                 this.cycleBehavior = bool;
222                 return this;
223             },
224             /**
225              * Adds an observer to this behavior.
226              * @param behaviorListener an observer instance.
227              */
228             addListener:function (behaviorListener) {
229                 this.lifecycleListenerList.push(behaviorListener);
230                 return this;
231             },
232             /**
233              * Remove all registered listeners to the behavior.
234              */
235             emptyListenerList:function () {
236                 this.lifecycleListenerList = [];
237                 return this;
238             },
239             /**
240              * @return an integer indicating the behavior start time in ms..
241              */
242             getStartTime:function () {
243                 return this.behaviorStartTime;
244             },
245             /**
246              * @return an integer indicating the behavior duration time in ms.
247              */
248             getDuration:function () {
249                 return this.behaviorDuration;
250 
251             },
252             /**
253              * Chekcs whether the behaviour is in scene time.
254              * In case it gets out of scene time, and has not been tagged as expired, the behavior is expired and observers
255              * are notified about that fact.
256              * @param time the scene time to check the behavior against.
257              * @param actor the actor the behavior is being applied to.
258              * @return a boolean indicating whether the behavior is in scene time.
259              */
260             isBehaviorInTime:function (time, actor) {
261 
262                 var st= CAAT.Behavior.BaseBehavior.Status;
263 
264                 if (this.status === st.EXPIRED || this.behaviorStartTime < 0) {
265                     return false;
266                 }
267 
268                 if (this.cycleBehavior) {
269                     if (time >= this.behaviorStartTime) {
270                         time = (time - this.behaviorStartTime) % this.behaviorDuration + this.behaviorStartTime;
271                     }
272                 }
273 
274                 if (time > this.behaviorStartTime + this.behaviorDuration) {
275                     if (this.status !== st.EXPIRED) {
276                         this.setExpired(actor, time);
277                     }
278 
279                     return false;
280                 }
281 
282                 if (this.status === st.NOT_STARTED) {
283                     this.status = st.STARTED;
284                     this.fireBehaviorStartedEvent(actor, time);
285                 }
286 
287                 return this.behaviorStartTime <= time; // && time<this.behaviorStartTime+this.behaviorDuration;
288             },
289 
290             fireBehaviorStartedEvent:function (actor, time) {
291                 for (var i = 0, l = this.lifecycleListenerList.length; i < l; i++) {
292                     var b = this.lifecycleListenerList[i];
293                     if (b.behaviorStarted) {
294                         b.behaviorStarted(this, time, actor);
295                     }
296                 }
297             },
298 
299             /**
300              * Notify observers about expiration event.
301              * @param actor a CAAT.Actor instance
302              * @param time an integer with the scene time the behavior was expired at.
303              */
304             fireBehaviorExpiredEvent:function (actor, time) {
305                 for (var i = 0, l = this.lifecycleListenerList.length; i < l; i++) {
306                     var b = this.lifecycleListenerList[i];
307                     if (b.behaviorExpired) {
308                         b.behaviorExpired(this, time, actor);
309                     }
310                 }
311             },
312             /**
313              * Notify observers about behavior being applied.
314              * @param actor a CAAT.Actor instance the behavior is being applied to.
315              * @param time the scene time of behavior application.
316              * @param normalizedTime the normalized time (0..1) considering 0 behavior start time and 1
317              * behaviorStartTime+behaviorDuration.
318              * @param value the value being set for actor properties. each behavior will supply with its own value version.
319              */
320             fireBehaviorAppliedEvent:function (actor, time, normalizedTime, value) {
321                 for (var i = 0, l = this.lifecycleListenerList.length; i < l; i++) {
322                     var b = this.lifecycleListenerList[i];
323                     if (b.behaviorApplied) {
324                         b.behaviorApplied(this, time, normalizedTime, actor, value);
325                     }
326                 }
327             },
328             /**
329              * Convert scene time into something more manageable for the behavior.
330              * behaviorStartTime will be 0 and behaviorStartTime+behaviorDuration will be 1.
331              * the time parameter will be proportional to those values.
332              * @param time the scene time to be normalized. an integer.
333              */
334             normalizeTime:function (time) {
335                 time = time - this.behaviorStartTime;
336                 if (this.cycleBehavior) {
337                     time %= this.behaviorDuration;
338                 }
339 
340                 return this.interpolator.getPosition(time / this.behaviorDuration).y;
341             },
342             /**
343              * Sets the behavior as expired.
344              * This method must not be called directly. It is an auxiliary method to isBehaviorInTime method.
345              * @param actor {CAAT.Actor}
346              * @param time {integer} the scene time.
347              *
348              * @private
349              */
350             setExpired:function (actor, time) {
351                 // set for final interpolator value.
352                 this.status = CAAT.Behavior.BaseBehavior.Status.EXPIRED;
353                 this.setForTime(this.interpolator.getPosition(1).y, actor);
354                 this.fireBehaviorExpiredEvent(actor, time);
355 
356                 if (this.discardable) {
357                     this.actor.removeBehavior(this);
358                 }
359             },
360             /**
361              * This method must be overriden for every Behavior breed.
362              * Must not be called directly.
363              * @param actor {CAAT.Actor} a CAAT.Actor instance.
364              * @param time {number} an integer with the scene time.
365              *
366              * @private
367              */
368             setForTime:function (time, actor) {
369 
370             },
371             /**
372              * @param overrides
373              */
374             initialize:function (overrides) {
375                 if (overrides) {
376                     for (var i in overrides) {
377                         this[i] = overrides[i];
378                     }
379                 }
380 
381                 return this;
382             },
383 
384             getPropertyName:function () {
385                 return "";
386             }
387         }
388     }
389 });
