1 /**
  2  * See LICENSE file.
  3  *
  4  * Behaviors are keyframing elements.
  5  * By using a BehaviorContainer, you can specify different actions on any animation Actor.
  6  * An undefined number of Behaviors can be defined for each Actor.
  7  *
  8  * There're the following Behaviors:
  9  *  + AlphaBehavior:   controls container/actor global alpha.
 10  *  + RotateBehavior:  takes control of rotation affine transform.
 11  *  + ScaleBehavior:   takes control of scaling on x and y axis affine transform.
 12  *  + Scale1Behavior:  takes control of scaling on x or y axis affine transform.
 13  *  + PathBehavior:    takes control of translating an Actor/ActorContainer across a path [ie. pathSegment collection].
 14  *  + GenericBehavior: applies a behavior to any given target object's property, or notifies a callback.
 15  *
 16  *
 17  **/
 18 
 19 /**
 20  * @class
 21  */
 22 CAAT.Module({
 23     defines:        "CAAT.Behavior.BaseBehavior",
 24     constants:      {
 25 
 26         /**
 27          * @enum
 28          */
 29         Status: {
 30             NOT_STARTED: 0,
 31             STARTED:    1,
 32             EXPIRED:    2
 33         }
 34     },
 35     depends:        ["CAAT.Behavior.Interpolator"],
 36     extendsWith:   function() {
 37 
 38         var DefaultInterpolator=    new CAAT.Behavior.Interpolator().createLinearInterpolator(false);
 39         var DefaultInterpolatorPP=  new CAAT.Behavior.Interpolator().createLinearInterpolator(true);
 40 
 41         /** @lends CAAT.Behavior.BaseBehavior.prototype */
 42         return {
 43 
 44             /**
 45              * Behavior base class.
 46              *
 47              * <p>
 48              * A behavior is defined by a frame time (behavior duration) and a behavior application function called interpolator.
 49              * In its default form, a behaviour is applied linearly, that is, the same amount of behavior is applied every same
 50              * time interval.
 51              * <p>
 52              * A concrete Behavior, a rotateBehavior in example, will change a concrete Actor's rotationAngle during the specified
 53              * period.
 54              * <p>
 55              * A behavior is guaranteed to notify (if any observer is registered) on behavior expiration.
 56              * <p>
 57              * A behavior can keep an unlimited observers. Observers are objects of the form:
 58              * <p>
 59              * <code>
 60              * {
 61              *      behaviorExpired : function( behavior, time, actor);
 62              *      behaviorApplied : function( behavior, time, normalizedTime, actor, value);
 63              * }
 64              * </code>
 65              * <p>
 66              * <strong>behaviorExpired</strong>: function( behavior, time, actor). This method will be called for any registered observer when
 67              * the scene time is greater than behavior's startTime+duration. This method will be called regardless of the time
 68              * granurality.
 69              * <p>
 70              * <strong>behaviorApplied</strong> : function( behavior, time, normalizedTime, actor, value). This method will be called once per
 71              * frame while the behavior is not expired and is in frame time (behavior startTime>=scene time). This method can be
 72              * called multiple times.
 73              * <p>
 74              * Every behavior is applied to a concrete Actor.
 75              * Every actor must at least define an start and end value. The behavior will set start-value at behaviorStartTime and
 76              * is guaranteed to apply end-value when scene time= behaviorStartTime+behaviorDuration.
 77              * <p>
 78              * You can set behaviors to apply forever that is cyclically. When a behavior is cycle=true, won't notify
 79              * behaviorExpired to its registered observers.
 80              * <p>
 81              * Other Behaviors simply must supply with the method <code>setForTime(time, actor)</code> overriden.
 82              *
 83              *
 84              */
 85             __init:function () {
 86                 this.lifecycleListenerList = [];
 87                 this.setDefaultInterpolator();
 88                 return this;
 89             },
 90 
 91             lifecycleListenerList:null, // observer list.
 92             behaviorStartTime:-1, // scene time to start applying the behavior
 93             behaviorDuration:-1, // behavior duration in ms.
 94             cycleBehavior:false, // apply forever ?
 95 
 96             status: CAAT.Behavior.BaseBehavior.Status.NOT_STARTED, // Status.NOT_STARTED
 97 
 98             interpolator:null, // behavior application function. linear by default.
 99             actor:null, // actor the Behavior acts on.
100             id:0, // an integer id suitable to identify this behavior by number.
101 
102             timeOffset:0,
103 
104             doValueApplication:true,
105 
106             solved:true,
107 
108             discardable:false, // is true, this behavior will be removed from the this.actor instance when it expires.
109 
110             /*  @memberOf CAAT.Behavior.BaseBehavior */
111             setValueApplication:function (apply) {
112                 this.doValueApplication = apply;
113                 return this;
114             },
115 
116             setTimeOffset:function (offset) {
117                 this.timeOffset = offset;
118                 return this;
119             },
120 
121             setStatus : function(st) {
122                 this.status= st;
123                 return this;
124             },
125 
126             /**
127              * Sets this behavior id.
128              * @param id an integer.
129              *
130              */
131             setId:function (id) {
132                 this.id = id;
133                 return this;
134             },
135             /**
136              * Sets the default interpolator to a linear ramp, that is, behavior will be applied linearly.
137              * @return this
138              */
139             setDefaultInterpolator:function () {
140                 this.interpolator = DefaultInterpolator;
141                 return this;
142             },
143             /**
144              * Sets default interpolator to be linear from 0..1 and from 1..0.
145              * @return this
146              */
147             setPingPong:function () {
148                 this.interpolator = DefaultInterpolatorPP;
149                 return this;
150             },
151 
152             /**
153              * Sets behavior start time and duration.
154              * Scene time will be the time of the scene the behavior actor is bound to.
155              * @param startTime {number} an integer indicating behavior start time in scene time in ms..
156              * @param duration {number} an integer indicating behavior duration in ms.
157              */
158             setFrameTime:function (startTime, duration) {
159                 this.behaviorStartTime = startTime;
160                 this.behaviorDuration = duration;
161                 this.status =CAAT.Behavior.BaseBehavior.Status.NOT_STARTED;
162 
163                 return this;
164             },
165             /**
166              * Sets behavior start time and duration but instead as setFrameTime which sets initial time as absolute time
167              * regarding scene's time, it uses a relative time offset from current scene time.
168              * a call to
169              *   setFrameTime( scene.time, duration ) is equivalent to
170              *   setDelayTime( 0, duration )
171              * @param delay {number}
172              * @param duration {number}
173              */
174             setDelayTime:function (delay, duration) {
175                 this.behaviorStartTime = delay;
176                 this.behaviorDuration = duration;
177                 this.status =CAAT.Behavior.BaseBehavior.Status.NOT_STARTED;
178                 this.solved = false;
179                 this.expired = false;
180 
181                 return this;
182 
183             },
184             setOutOfFrameTime:function () {
185                 this.status =CAAT.Behavior.BaseBehavior.Status.EXPIRED;
186                 this.behaviorStartTime = Number.MAX_VALUE;
187                 this.behaviorDuration = 0;
188                 return this;
189             },
190             /**
191              * Changes behavior default interpolator to another instance of CAAT.Interpolator.
192              * If the behavior is not defined by CAAT.Interpolator factory methods, the interpolation function must return
193              * its values in the range 0..1. The behavior will only apply for such value range.
194              * @param interpolator a CAAT.Interpolator instance.
195              */
196             setInterpolator:function (interpolator) {
197                 this.interpolator = interpolator;
198                 return this;
199             },
200             /**
201              * This method must no be called directly.
202              * The director loop will call this method in orther to apply actor behaviors.
203              * @param time the scene time the behaviro is being applied at.
204              * @param actor a CAAT.Actor instance the behavior is being applied to.
205              */
206             apply:function (time, actor) {
207 
208                 if (!this.solved) {
209                     this.behaviorStartTime += time;
210                     this.solved = true;
211                 }
212 
213                 time += this.timeOffset * this.behaviorDuration;
214 
215                 var orgTime = time;
216                 if (this.isBehaviorInTime(time, actor)) {
217                     time = this.normalizeTime(time);
218                     this.fireBehaviorAppliedEvent(
219                         actor,
220                         orgTime,
221                         time,
222                         this.setForTime(time, actor));
223                 }
224             },
225 
226             /**
227              * Sets the behavior to cycle, ie apply forever.
228              * @param bool a boolean indicating whether the behavior is cycle.
229              */
230             setCycle:function (bool) {
231                 this.cycleBehavior = bool;
232                 return this;
233             },
234             /**
235              * Adds an observer to this behavior.
236              * @param behaviorListener an observer instance.
237              */
238             addListener:function (behaviorListener) {
239                 this.lifecycleListenerList.push(behaviorListener);
240                 return this;
241             },
242             /**
243              * Remove all registered listeners to the behavior.
244              */
245             emptyListenerList:function () {
246                 this.lifecycleListenerList = [];
247                 return this;
248             },
249             /**
250              * @return an integer indicating the behavior start time in ms..
251              */
252             getStartTime:function () {
253                 return this.behaviorStartTime;
254             },
255             /**
256              * @return an integer indicating the behavior duration time in ms.
257              */
258             getDuration:function () {
259                 return this.behaviorDuration;
260 
261             },
262             /**
263              * Chekcs whether the behaviour is in scene time.
264              * In case it gets out of scene time, and has not been tagged as expired, the behavior is expired and observers
265              * are notified about that fact.
266              * @param time the scene time to check the behavior against.
267              * @param actor the actor the behavior is being applied to.
268              * @return a boolean indicating whether the behavior is in scene time.
269              */
270             isBehaviorInTime:function (time, actor) {
271 
272                 var st= CAAT.Behavior.BaseBehavior.Status;
273 
274                 if (this.status === st.EXPIRED || this.behaviorStartTime < 0) {
275                     return false;
276                 }
277 
278                 if (this.cycleBehavior) {
279                     if (time >= this.behaviorStartTime) {
280                         time = (time - this.behaviorStartTime) % this.behaviorDuration + this.behaviorStartTime;
281                     }
282                 }
283 
284                 if (time > this.behaviorStartTime + this.behaviorDuration) {
285                     if (this.status !== st.EXPIRED) {
286                         this.setExpired(actor, time);
287                     }
288 
289                     return false;
290                 }
291 
292                 if (this.status === st.NOT_STARTED) {
293                     this.status = st.STARTED;
294                     this.fireBehaviorStartedEvent(actor, time);
295                 }
296 
297                 return this.behaviorStartTime <= time; // && time<this.behaviorStartTime+this.behaviorDuration;
298             },
299 
300             fireBehaviorStartedEvent:function (actor, time) {
301                 for (var i = 0, l = this.lifecycleListenerList.length; i < l; i++) {
302                     var b = this.lifecycleListenerList[i];
303                     if (b.behaviorStarted) {
304                         b.behaviorStarted(this, time, actor);
305                     }
306                 }
307             },
308 
309             /**
310              * Notify observers about expiration event.
311              * @param actor a CAAT.Actor instance
312              * @param time an integer with the scene time the behavior was expired at.
313              */
314             fireBehaviorExpiredEvent:function (actor, time) {
315                 for (var i = 0, l = this.lifecycleListenerList.length; i < l; i++) {
316                     var b = this.lifecycleListenerList[i];
317                     if (b.behaviorExpired) {
318                         b.behaviorExpired(this, time, actor);
319                     }
320                 }
321             },
322             /**
323              * Notify observers about behavior being applied.
324              * @param actor a CAAT.Actor instance the behavior is being applied to.
325              * @param time the scene time of behavior application.
326              * @param normalizedTime the normalized time (0..1) considering 0 behavior start time and 1
327              * behaviorStartTime+behaviorDuration.
328              * @param value the value being set for actor properties. each behavior will supply with its own value version.
329              */
330             fireBehaviorAppliedEvent:function (actor, time, normalizedTime, value) {
331                 for (var i = 0, l = this.lifecycleListenerList.length; i < l; i++) {
332                     var b = this.lifecycleListenerList[i];
333                     if (b.behaviorApplied) {
334                         b.behaviorApplied(this, time, normalizedTime, actor, value);
335                     }
336                 }
337             },
338             /**
339              * Convert scene time into something more manageable for the behavior.
340              * behaviorStartTime will be 0 and behaviorStartTime+behaviorDuration will be 1.
341              * the time parameter will be proportional to those values.
342              * @param time the scene time to be normalized. an integer.
343              */
344             normalizeTime:function (time) {
345                 time = time - this.behaviorStartTime;
346                 if (this.cycleBehavior) {
347                     time %= this.behaviorDuration;
348                 }
349 
350                 return this.interpolator.getPosition(time / this.behaviorDuration).y;
351             },
352             /**
353              * Sets the behavior as expired.
354              * This method must not be called directly. It is an auxiliary method to isBehaviorInTime method.
355              * @param actor {CAAT.Actor}
356              * @param time {integer} the scene time.
357              *
358              * @private
359              */
360             setExpired:function (actor, time) {
361                 // set for final interpolator value.
362                 this.status = CAAT.Behavior.BaseBehavior.Status.EXPIRED;
363                 this.setForTime(this.interpolator.getPosition(1).y, actor);
364                 this.fireBehaviorExpiredEvent(actor, time);
365 
366                 if (this.discardable) {
367                     this.actor.removeBehavior(this);
368                 }
369             },
370             /**
371              * This method must be overriden for every Behavior breed.
372              * Must not be called directly.
373              * @param actor {CAAT.Actor} a CAAT.Actor instance.
374              * @param time {number} an integer with the scene time.
375              *
376              * @private
377              */
378             setForTime:function (time, actor) {
379 
380             },
381             /**
382              * @param overrides
383              */
384             initialize:function (overrides) {
385                 if (overrides) {
386                     for (var i in overrides) {
387                         this[i] = overrides[i];
388                     }
389                 }
390 
391                 return this;
392             },
393 
394             getPropertyName:function () {
395                 return "";
396             }
397         }
398     }
