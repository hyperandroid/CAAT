1 /**
  2  * See LICENSE file.
  3  *
  4  * An actor to show the path and its handles in the scene graph. 
  5  *
  6  **/
  7 CAAT.Module( {
  8     defines : "CAAT.Foundation.UI.PathActor",
  9     aliases : ["CAAT.PathActor"],
 10     depends : [
 11         "CAAT.Foundation.Actor"
 12     ],
 13     extendsClass : "CAAT.Foundation.Actor",
 14     extendsWith : {
 15 
 16 		path                    : null,
 17 		pathBoundingRectangle   : null,
 18 		bOutline                : false,
 19         outlineColor            : 'black',
 20         onUpdateCallback        : null,
 21         interactive             : false,
 22         showBBox                : false,
 23 
 24         /**
 25          * Return the contained path.
 26          * @return {CAAT.Path}
 27          */
 28         getPath : function() {
 29             return this.path;
 30         },
 31 
 32         /**
 33          * Sets the path to manage.
 34          * @param path {CAAT.PathUtil.PathSegment}
 35          * @return this
 36          */
 37 		setPath : function(path) {
 38 			this.path= path;
 39             if ( path!=null ) {
 40 			    this.pathBoundingRectangle= path.getBoundingBox();
 41                 this.setInteractive( this.interactive );
 42             }
 43             return this;
 44 		},
 45         /**
 46          * Paint this actor.
 47          * @param director {CAAT.Foundation.Director}
 48          * @param time {number}. Scene time.
 49          */
 50 		paint : function(director, time) {
 51 
 52             CAAT.Foundation.UI.PathActor.superclass.paint.call( this, director, time );
 53 
 54             if ( !this.path ) {
 55                 return;
 56             }
 57 
 58             var ctx= director.ctx;
 59 
 60             ctx.strokeStyle='#000';
 61 			this.path.paint(director, this.interactive);
 62 
 63             if ( this.bOutline ) {
 64                 ctx.strokeStyle= this.outlineColor;
 65                 ctx.strokeRect(
 66                     this.pathBoundingRectangle.x,
 67                     this.pathBoundingRectangle.y,
 68                     this.pathBoundingRectangle.width,
 69                     this.pathBoundingRectangle.height
 70                 );
 71             }
 72 /*
 73 			if ( this.bOutline ) {
 74 				ctx.strokeStyle= this.outlineColor;
 75 				ctx.strokeRect(0,0,this.width,this.height);
 76 			}
 77 */
 78 		},
 79         /**
 80          * Enables/disables drawing of the contained path's bounding box.
 81          * @param show {boolean} whether to show the bounding box
 82          * @param color {=string} optional parameter defining the path's bounding box stroke style.
 83          */
 84         showBoundingBox : function(show, color) {
 85             this.bOutline= show;
 86             if ( show && color ) {
 87                 this.outlineColor= color;
 88             }
 89             return this;
 90         },
 91         /**
 92          * Set the contained path as interactive. This means it can be changed on the fly by manipulation
 93          * of its control points.
 94          * @param interactive
 95          */
 96         setInteractive : function(interactive) {
 97             this.interactive= interactive;
 98             if ( this.path ) {
 99                 this.path.setInteractive(interactive);
100             }
101             return this;
102         },
103         setOnUpdateCallback : function( fn ) {
104             this.onUpdateCallback= fn;
105             return this;
106         },
107         /**
108          * Route mouse dragging functionality to the contained path.
109          * @param mouseEvent {CAAT.Event.MouseEvent}
110          */
111 		mouseDrag : function(mouseEvent) {
112 			this.path.drag(mouseEvent.point.x, mouseEvent.point.y, this.onUpdateCallback);
113 		},
114         /**
115          * Route mouse down functionality to the contained path.
116          * @param mouseEvent {CAAT.Event.MouseEvent}
117          */
118 		mouseDown : function(mouseEvent) {
119 			this.path.press(mouseEvent.point.x, mouseEvent.point.y);
120 		},
121         /**
122          * Route mouse up functionality to the contained path.
123          * @param mouseEvent {CAAT.Event.MouseEvent}
124          */
125 		mouseUp : function(mouseEvent) {
126 			this.path.release();
127 		}
128 	}
129 });
