  7 CAAT.Module({
  8     defines:"CAAT.Module.Audio.AudioManager",
  9     depends:[
 10         "CAAT.Module.Runtime.BrowserInfo"

 12     extendsWith:function () {
 13         return {
 14             __init:function () {
 15                 this.browserInfo = CAAT.Module.Runtime.BrowserInfo;
 16                 return this;
 17             },
 18 
 19             musicChannel: null,
 20             browserInfo:null,
 21             musicEnabled:true,
 22             fxEnabled:true,
 23             audioCache:null, // audio elements.
 24             channels:null, // available playing channels.
 25             workingChannels:null, // currently playing channels.
 26             loopingChannels:[],
 27             audioTypes:{               // supported audio formats. Don't remember where i took them from :S
 28                 'mp3':'audio/mpeg;',
 29                 'ogg':'audio/ogg; codecs="vorbis"',
 30                 'wav':'audio/wav; codecs="1"',
 31                 'mp4':'audio/mp4; codecs="mp4a.40.2"'
 32             },

 34             /**
 35              * Initializes the sound subsystem by creating a fixed number of Audio channels.
 36              * Every channel registers a handler for sound playing finalization. If a callback is set, the
 37              * callback function will be called with the associated sound id in the cache.
 38              *
 39              * @param numChannels {number} number of channels to pre-create. 8 by default.
 40              *
 41              * @return this.
 42              */
 43             initialize:function (numChannels) {
 44 
 45                 this.audioCache = [];
 46                 this.channels = [];
 47                 this.workingChannels = [];
 48 
 49                 for (var i = 0; i <= numChannels; i++) {
 50                     var channel = document.createElement('audio');
 51 
 52                     if (null !== channel) {
 53                         channel.finished = -1;
 54                         this.channels.push(channel);
 55                         var me = this;
 56                         channel.addEventListener(
 57                             'ended',
 58                             // on sound end, set channel to available channels list.
 59                             function (audioEvent) {
 60                                 var target = audioEvent.target;
 61                                 var i;
 62 
 63                                 // remove from workingChannels
 64                                 for (i = 0; i < me.workingChannels.length; i++) {
 65                                     if (me.workingChannels[i] === target) {
 66                                         me.workingChannels.splice(i, 1);
 67                                         break;
 68                                     }
 69                                 }
 70 
 71                                 if (target.caat_callback) {
 72                                     target.caat_callback(target.caat_id);
 73                                 }
 74 
 75                                 // set back to channels.
 76                                 me.channels.push(target);
 77                             },
 78                             false
 79                         );
 80                     }
 81                 }
 82 
 83                 this.musicChannel= this.channels.pop();
 84 
 85                 return this;
 86             },
 87             /**
 88              * Tries to add an audio tag to the available list of valid audios. The audio is described by a url.
 89              * @param id {object} an object to associate the audio element (if suitable to be played).
 90              * @param url {string} a string describing an url.
 91              * @param endplaying_callback {function} callback to be called upon sound end.
 92              *
 93              * @return {boolean} a boolean indicating whether the browser can play this resource.
 94              *
 95              * @private
 96              */
 97             addAudioFromURL:function (id, url, endplaying_callback) {
 98                 var extension = null;
 99                 var audio = document.createElement('audio');
100 
101                 if (null !== audio) {
102 
103                     if (!audio.canPlayType) {
104                         return false;
105                     }
106 
107                     extension = url.substr(url.lastIndexOf('.') + 1);
108                     var canplay = audio.canPlayType(this.audioTypes[extension]);
109 
110                     if (canplay !== "" && canplay !== "no") {
111                         audio.src = url;
112                         audio.preload = "auto";
113                         audio.load();
114                         if (endplaying_callback) {
115                             audio.caat_callback = endplaying_callback;
116                             audio.caat_id = id;
117                         }
118                         this.audioCache.push({ id:id, audio:audio });
119 
120                         return true;
121                     }
122                 }
123 
124                 return false;
125             },
126             /**
127              * Tries to add an audio tag to the available list of valid audios. The audio element comes from
128              * an HTMLAudioElement.
129              * @param id {object} an object to associate the audio element (if suitable to be played).
130              * @param audio {HTMLAudioElement} a DOM audio node.
131              * @param endplaying_callback {function} callback to be called upon sound end.
132              *
133              * @return {boolean} a boolean indicating whether the browser can play this resource.
134              *
135              * @private
136              */
137             addAudioFromDomNode:function (id, audio, endplaying_callback) {
138 
139                 var extension = audio.src.substr(audio.src.lastIndexOf('.') + 1);
140                 if (audio.canPlayType(this.audioTypes[extension])) {
141                     if (endplaying_callback) {
142                         audio.caat_callback = endplaying_callback;
143                         audio.caat_id = id;
144                     }
145                     this.audioCache.push({ id:id, audio:audio });
146 
147                     return true;
148                 }
149 
150                 return false;
151             },
152             /**
153              * Adds an elements to the audio cache.
154              * @param id {object} an object to associate the audio element (if suitable to be played).
155              * @param element {URL|HTMLElement} an url or html audio tag.
156              * @param endplaying_callback {function} callback to be called upon sound end.
157              *
158              * @return {boolean} a boolean indicating whether the browser can play this resource.
159              *
160              * @private
161              */
162             addAudioElement:function (id, element, endplaying_callback) {
163                 if (typeof element === "string") {
164                     return this.addAudioFromURL(id, element, endplaying_callback);
165                 } else {
166                     try {
167                         if (element instanceof HTMLAudioElement) {
168                             return this.addAudioFromDomNode(id, element, endplaying_callback);
169                         }
170                     }
171                     catch (e) {
172                     }
173                 }
174 
175                 return false;
176             },
177             /**
178              * creates an Audio object and adds it to the audio cache.
179              * This function expects audio data described by two elements, an id and an object which will
180              * describe an audio element to be associated with the id. The object will be of the form
181              * array, dom node or a url string.
182              *
183              * <p>
184              * The audio element can be one of the two forms:
185              *
186              * <ol>
187              *  <li>Either an HTMLAudioElement/Audio object or a string url.
188              *  <li>An array of elements of the previous form.
189              * </ol>
190              *
191              * <p>
192              * When the audio attribute is an array, this function will iterate throught the array elements
193              * until a suitable audio element to be played is found. When this is the case, the other array
194              * elements won't be taken into account. The valid form of using this addAudio method will be:
195              *
196              * <p>
197              * 1.<br>
198              * addAudio( id, url } ). In this case, if the resource pointed by url is
199              * not suitable to be played (i.e. a call to the Audio element's canPlayType method return 'no')
200              * no resource will be added under such id, so no sound will be played when invoking the play(id)
201              * method.
202              * <p>
203              * 2.<br>
204              * addAudio( id, dom_audio_tag ). In this case, the same logic than previous case is applied, but
205              * this time, the parameter url is expected to be an audio tag present in the html file.
206              * <p>
207              * 3.<br>
208              * addAudio( id, [array_of_url_or_domaudiotag] ). In this case, the function tries to locate a valid
209              * resource to be played in any of the elements contained in the array. The array element's can
210              * be any type of case 1 and 2. As soon as a valid resource is found, it will be associated to the
211              * id in the valid audio resources to be played list.
212              *
213              * @return this
214              */
215             addAudio:function (id, array_of_url_or_domnodes, endplaying_callback) {
216 
217                 if (array_of_url_or_domnodes instanceof Array) {
218                     /*
219                      iterate throught array elements until we can safely add an audio element.
220                      */
221                     for (var i = 0; i < array_of_url_or_domnodes.length; i++) {
222                         if (this.addAudioElement(id, array_of_url_or_domnodes[i], endplaying_callback)) {
223                             break;
224                         }
225                     }
226                 } else {
227                     this.addAudioElement(id, array_of_url_or_domnodes, endplaying_callback);
228                 }
229 
230                 return this;
231             },
232             /**
233              * Returns an audio object.
234              * @param aId {object} the id associated to the target Audio object.
235              * @return {object} the HTMLAudioElement addociated to the given id.
236              */
237             getAudio:function (aId) {
238                 for (var i = 0; i < this.audioCache.length; i++) {
239                     if (this.audioCache[i].id === aId) {
240                         return this.audioCache[i].audio;
241                     }
242                 }
243 
244                 return null;
245             },
246 
247             stopMusic : function() {
248                 this.musicChannel.pause();
249             },
250 
251             playMusic : function(id) {
252                 if (!this.musicEnabled) {
253                     return this;
254                 }
255 
256                 var audio_in_cache = this.getAudio(id);
257                 // existe el audio, y ademas hay un canal de audio disponible.
258                 if (null !== audio_in_cache) {
259                     var audio =this.musicChannel;
260                     if (null !== audio) {
261                         audio.src = audio_in_cache.src;
262                         audio.preload = "auto";
263 
264                         if (this.browserInfo.browser === 'Firefox') {
265                             audio.addEventListener(
266                                 'ended',
267                                 // on sound end, set channel to available channels list.
268                                 function (audioEvent) {
269                                     var target = audioEvent.target;
270                                     target.currentTime = 0;
271                                 },
272                                 false
273                             );
274                         } else {
275                             audio.loop = true;
276                         }
277                         audio.load();
278                         audio.play();
279                         return audio;
280                     }
281                 }
282             },
283 
284             /**
285              * Set an audio object volume.
286              * @param id {object} an audio Id
287              * @param volume {number} volume to set. The volume value is not checked.
288              *
289              * @return this
290              */
291             setVolume:function (id, volume) {
292                 var audio = this.getAudio(id);
293                 if (null != audio) {
294                     audio.volume = volume;
295                 }
296 
297                 return this;
298             },
299 
300             /**
301              * Plays an audio file from the cache if any sound channel is available.
302              * The playing sound will occupy a sound channel and when ends playing will leave
303              * the channel free for any other sound to be played in.
304              * @param id {object} an object identifying a sound in the sound cache.
305              * @return this.
306              */
307             play:function (id) {
308                 if (!this.fxEnabled) {
309                     return this;
310                 }
311 
312                 var audio = this.getAudio(id);
313                 // existe el audio, y ademas hay un canal de audio disponible.
314                 if (null !== audio && this.channels.length > 0) {
315                     var channel = this.channels.shift();
316                     channel.src = audio.src;
317                     channel.load();
318                     channel.volume = audio.volume;
319                     channel.play();
320                     this.workingChannels.push(channel);
321                 }
322 
323                 return this;
324             },
325             /**
326              * This method creates a new AudioChannel to loop the sound with.
327              * It returns an Audio object so that the developer can cancel the sound loop at will.
328              * The user must call <code>pause()</code> method to stop playing a loop.
329              * <p>
330              * Firefox does not honor the loop property, so looping is performed by attending end playing
331              * event on audio elements.
332              *
333              * @return {HTMLElement} an Audio instance if a valid sound id is supplied. Null otherwise
334              */
335             loop:function (id) {
336 
337                 if (!this.musicEnabled) {
338                     return this;
339                 }
340 
341                 var audio_in_cache = this.getAudio(id);
342                 // existe el audio, y ademas hay un canal de audio disponible.
343                 if (null !== audio_in_cache) {
344                     var audio = document.createElement('audio');
345                     if (null !== audio) {
346                         audio.src = audio_in_cache.src;
347                         audio.preload = "auto";
348 
349                         if (this.browserInfo.browser === 'Firefox') {
350                             audio.addEventListener(
351                                 'ended',
352                                 // on sound end, set channel to available channels list.
353                                 function (audioEvent) {
354                                     var target = audioEvent.target;
355                                     target.currentTime = 0;
356                                 },
357                                 false
358                             );
359                         } else {
360                             audio.loop = true;
361                         }
362                         audio.load();
363                         audio.play();
364                         this.loopingChannels.push(audio);
365                         return audio;
366                     }
367                 }
368 
369                 return null;
370             },
371             /**
372              * Cancel all playing audio channels
373              * Get back the playing channels to available channel list.
374              *
375              * @return this
376              */
377             endSound:function () {
378                 var i;
379                 for (i = 0; i < this.workingChannels.length; i++) {
380                     this.workingChannels[i].pause();
381                     this.channels.push(this.workingChannels[i]);
382                 }
383 
384                 for (i = 0; i < this.loopingChannels.length; i++) {
385                     this.loopingChannels[i].pause();
386                 }
387 
388                 this.stopMusic();
389 
390                 return this;
391             },
392             setSoundEffectsEnabled:function (enable) {
393                 this.fxEnabled = enable;
394                 for (var i = 0; i < this.loopingChannels.length; i++) {
395                     if (enable) {
396                         this.loopingChannels[i].play();
397                     } else {
398                         this.loopingChannels[i].pause();
399                     }
400                 }
401                 return this;
402             },
403             isSoundEffectsEnabled:function () {
404                 return this.fxEnabled;
405             },
406             setMusicEnabled:function (enable) {
407                 this.musicEnabled = enable;
408                 this.stopMusic();
409                 return this;
410             },
411             isMusicEnabled:function () {
412                 return this.musicEnabled;
413             }
414         }
415     }
