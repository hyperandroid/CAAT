672             /**
673              * This is the entry point for the animation system of the Director.
674              * The director is fed with the elapsed time value to maintain a virtual timeline.
675              * This virtual timeline will provide each Scene with its own virtual timeline, and will only
676              * feed time when the Scene is the current Scene, or is being switched.
677              *
678              * If dirty rectangles are enabled and canvas is used for rendering, the dirty rectangles will be
679              * set up as a single clip area.
680              *
681              * @param time {number} integer indicating the elapsed time between two consecutive frames of the
682              * Director.
683              */
684             render:function (time) {
685 
686                 if (this.currentScene && this.currentScene.isPaused()) {
687                     return;
688                 }
689 
690                 this.time += time;
691 
692                 for (i = 0, l = this.childrenList.length; i < l; i++) {
693                     var c = this.childrenList[i];
694                     if (c.isInAnimationFrame(this.time) && !c.isPaused()) {
695                         var tt = c.time - c.start_time;
696                         c.timerManager.checkTimers(tt);
697                         c.timerManager.removeExpiredTimers();
698                     }
699                 }
700 
701 
702                 this.animate(this, this.time);
703 
704                 if (!navigator.isCocoonJS && CAAT.DEBUG) {
705                     this.resetStats();
706                 }
707 
708                 /**
709                  * draw director active scenes.
710                  */
711                 var ne = this.childrenList.length;
712                 var i, tt, c;
713                 var ctx = this.ctx;
714 
715                 if (this.glEnabled) {
716 
717                     this.gl.clear(this.gl.COLOR_BUFFER_BIT | this.gl.DEPTH_BUFFER_BIT);
718                     this.coordsIndex = 0;
719                     this.uvIndex = 0;
720 
721                     for (i = 0; i < ne; i++) {
722                         c = this.childrenList[i];
