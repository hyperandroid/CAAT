554             __setupRetina : function() {
555 
556                 if ( CAAT.RETINA_DISPLAY_ENABLED ) {
557 
558                     // The world is full of opensource awesomeness.

560                     // Source: http://www.html5rocks.com/en/tutorials/canvas/hidpi/

562                     var devicePixelRatio= CAAT.Module.Runtime.BrowserInfo.DevicePixelRatio;
563                     var backingStoreRatio = this.ctx.webkitBackingStorePixelRatio ||
564                                             this.ctx.mozBackingStorePixelRatio ||
565                                             this.ctx.msBackingStorePixelRatio ||
566                                             this.ctx.oBackingStorePixelRatio ||
567                                             this.ctx.backingStorePixelRatio ||
568                                             1;
569 
570                     var ratio = devicePixelRatio / backingStoreRatio;
571 
572                     if (devicePixelRatio !== backingStoreRatio) {
573

575                         var oldHeight = this.canvas.height;
576
