 66         /**
 67          * Enable at window level accelerometer events.
 68          */
 69         CAAT.enableDeviceMotion= function() {
 70 
 71             CAAT.prevOnDeviceMotion=    null;   // previous accelerometer callback function.
 72             CAAT.onDeviceMotion=        null;   // current accelerometer callback set for CAAT.
 73             CAAT.accelerationIncludingGravity= { x:0, y:0, z:0 };   // acceleration data.
 74             CAAT.rotationRate= { alpha: 0, beta:0, gamma: 0 };      // angles data.
 75 
 76             function tilt(data) {
 77                 CAAT.rotationRate= {
 78                         alpha : 0,
 79                         beta  : data[0],
 80                         gamma : data[1]
 81                     };
 82             }
 83 
 84             if (window.DeviceOrientationEvent) {
 85                 window.addEventListener("deviceorientation", function (event) {
 86                     tilt([event.beta, event.gamma]);
 87                 }, true);
 88             } else if (window.DeviceMotionEvent) {
 89                 window.addEventListener('devicemotion', function (event) {
 90                     tilt([event.acceleration.x * 2, event.acceleration.y * 2]);
 91                 }, true);
 92             } else {
 93                 window.addEventListener("MozOrientation", function (event) {
 94                     tilt([-event.y * 45, event.x * 45]);
 95                 }, true);
 96             }
 97 
 98         };

101         /**

103          */
104         (function() {
105 
106             window.addEventListener('keydown',
107                 function(evt) {
108                     var key = (evt.which) ? evt.which : evt.keyCode;
109 
110                     if ( key===CAAT.SHIFT_KEY ) {
111                         CAAT.KEY_MODIFIERS.shift= true;
112                     } else if ( key===CAAT.CONTROL_KEY ) {
113                         CAAT.KEY_MODIFIERS.control= true;
114                     } else if ( key===CAAT.ALT_KEY ) {
115                         CAAT.KEY_MODIFIERS.alt= true;
116                     } else {
117                         for( var i=0; i<CAAT.keyListeners.length; i++ ) {
118                             CAAT.keyListeners[i]( new CAAT.KeyEvent(
119                                 key,
120                                 'down',
121                                 {
122                                     alt:        CAAT.KEY_MODIFIERS.alt,
123                                     control:    CAAT.KEY_MODIFIERS.control,
124                                     shift:      CAAT.KEY_MODIFIERS.shift
125                                 },
126                                 evt)) ;
127                         }
128                     }
129                 },
130                 false);
131 
132             window.addEventListener('keyup',
133                 function(evt) {
134 
135                     var key = (evt.which) ? evt.which : evt.keyCode;
136                     if ( key===CAAT.SHIFT_KEY ) {
137                         CAAT.KEY_MODIFIERS.shift= false;
138                     } else if ( key===CAAT.CONTROL_KEY ) {
139                         CAAT.KEY_MODIFIERS.control= false;
140                     } else if ( key===CAAT.ALT_KEY ) {
141                         CAAT.KEY_MODIFIERS.alt= false;
142                     } else {
143 
144                         for( var i=0; i<CAAT.keyListeners.length; i++ ) {
145                             CAAT.keyListeners[i]( new CAAT.KeyEvent(
146                                 key,
147                                 'up',
148                                 {
149                                     alt:        CAAT.KEY_MODIFIERS.alt,
150                                     control:    CAAT.KEY_MODIFIERS.control,
151                                     shift:      CAAT.KEY_MODIFIERS.shift
152                                 },
153                                 evt));
154                         }
155                     }
156                 },
157                 false );
158 
159             window.addEventListener('resize',
160                 function(evt) {
161                     for( var i=0; i<CAAT.windowResizeListeners.length; i++ ) {
162                         CAAT.windowResizeListeners[i].windowResized(
163                                 window.innerWidth,
164                                 window.innerHeight);
165                     }
166                 },
167                 false);
168         })();
