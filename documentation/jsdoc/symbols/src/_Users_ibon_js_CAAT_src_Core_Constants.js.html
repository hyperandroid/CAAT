  6 CAAT.Module( {
  7     defines: "CAAT.Core.Constants",
  8 
  9     extendsWith: function() {
 10 
 11         /**
 12          * This function makes the system obey decimal point calculations for actor's position, size, etc.
 13          * This may speed things up in some browsers, but at the cost of affecting visuals (like in rotating
 14          * objects).
 15          *
 16          * Latest Chrome (20+) is not affected by this.
 17          *
 18          * Default CAAT.Matrix try to speed things up.
 19          *
 20          * @param clamp {boolean}
 21          */
 22         CAAT.setCoordinateClamping= function( clamp ) {
 23             if ( clamp ) {
 24                 CAAT.Matrix.prototype.transformRenderingContext= CAAT.Matrix.prototype.transformRenderingContext_Clamp;
 25                 CAAT.Matrix.prototype.transformRenderingContextSet= CAAT.Matrix.prototype.transformRenderingContextSet_Clamp;
 26                 CAAT.Math.Matrix.prototype.transformRenderingContext= CAAT.Matrix.prototype.transformRenderingContext_Clamp;
 27                 CAAT.Math.Matrix.prototype.transformRenderingContextSet= CAAT.Matrix.prototype.transformRenderingContextSet_Clamp;
 28             } else {
 29                 CAAT.Matrix.prototype.transformRenderingContext= CAAT.Matrix.prototype.transformRenderingContext_NoClamp;
 30                 CAAT.Matrix.prototype.transformRenderingContextSet= CAAT.Matrix.prototype.transformRenderingContextSet_NoClamp;
 31                 CAAT.Math.Matrix.prototype.transformRenderingContext= CAAT.Matrix.prototype.transformRenderingContext_NoClamp;
 32                 CAAT.Math.Matrix.prototype.transformRenderingContextSet= CAAT.Matrix.prototype.transformRenderingContextSet_NoClamp;
 33             }
 34         };
 35 
 36         /**
 37          * Log function which deals with window's Console object.
 38          */
 39         CAAT.log= function() {
 40             if(window.console){
 41                 window.console.log( Array.prototype.slice.call(arguments) );
 42             }
 43         };
 44 
 45         /**
 46          * Control how CAAT.Font and CAAT.TextActor control font ascent/descent values.
 47          * 0 means it will guess values from a font height
 48          * 1 means it will try to use css to get accurate ascent/descent values and fall back to the previous method
 49          *   in case it couldn't.
 50          *
 51          * @type {Number}
 52          */
 53         CAAT.CSS_TEXT_METRICS=      0;
 54 
 55         CAAT.GLRENDER= false;
 56 
 57         /**
 58          * DEBUGGING CONSTANTS
 59          */
 60         CAAT.DEBUG= false;              // set this variable before building CAAT.Director intances to
 61                                     // enable debug panel.
 62         CAAT.DEBUGBB= false;            // show Bounding Boxes
 63         CAAT.DEBUGBBBCOLOR = '#00f';      // Bounding Boxes color.
 64         CAAT.DEBUGAABB = false;         // debug axis aligned bounding boxes.
 65         CAAT.DEBUGAABBCOLOR = '#f00';
 66         CAAT.DEBUG_DIRTYRECTS= false;    // if CAAT.Director.setClear uses CLEAR_DIRTY_RECTS, this will show them
 67                                     // on screen.
 68 
 69         /**
 70          * Do not consider mouse drag gesture at least until you have dragged
 71          * DRAG_THRESHOLD_X and DRAG_THRESHOLD_Y pixels.
 72          * This is suitable for tablets, where just by touching, drag events are delivered.
 73          */
 74         CAAT.DRAG_THRESHOLD_X=      5;
 75         CAAT.DRAG_THRESHOLD_Y=      5;
 76 
 77         return {
 78         }
 79     }
 80 } );
