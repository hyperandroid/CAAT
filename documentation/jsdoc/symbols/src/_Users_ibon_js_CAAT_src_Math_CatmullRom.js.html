 15              */
 16             setCurve:function (p0, p1, p2, p3) {
 17 
 18                 this.coordlist = [];
 19                 this.coordlist.push(p0);
 20                 this.coordlist.push(p1);
 21                 this.coordlist.push(p2);
 22                 this.coordlist.push(p3);
 23 
 24                 this.update();
 25

 27             },
 28             /**
 29              * Paint the contour by solving again the entire curve.
 30              * @param director {CAAT.Director}
 31              */
 32             paint:function (director) {
 33 
 34                 var x1, y1;
 35 
 36                 // Catmull rom solves from point 1 !!!
 37 
 38                 x1 = this.coordlist[1].x;
 39                 y1 = this.coordlist[1].y;
 40 
 41                 var ctx = director.ctx;
 42 
 43                 ctx.save();
 44                 ctx.beginPath();
 45                 ctx.moveTo(x1, y1);
 46 
 47                 var point = new CAAT.Point();
 48 
 49                 for (var t = this.k; t <= 1 + this.k; t += this.k) {
 50                     this.solve(point, t);
 51                     ctx.lineTo(point.x, point.y);
 52                 }
 53 
 54                 ctx.stroke();
 55                 ctx.restore();
 56 
 57                 CAAT.Math.CatmullRom.superclass.paint.call(this, director);
 58             },
 59             /**
 60              * Solves the curve for any given parameter t.
 61              * @param point {CAAT.Point} the point to store the solved value on the curve.
 62              * @param t {number} a number in the range 0..1
 63              */
 64             solve:function (point, t) {
 65                 var c = this.coordlist;
 66 
 67                 // Handy from CAKE. Thanks.
 68                 var af = ((-t + 2) * t - 1) * t * 0.5
 69                 var bf = (((3 * t - 5) * t) * t + 2) * 0.5
 70                 var cf = ((-3 * t + 4) * t + 1) * t * 0.5
 71                 var df = ((t - 1) * t * t) * 0.5
 72 
 73                 point.x = c[0].x * af + c[1].x * bf + c[2].x * cf + c[3].x * df;
 74                 point.y = c[0].y * af + c[1].y * bf + c[2].y * cf + c[3].y * df;
 75 
 76                 return point;
 77 
 78             },
 79 
 80             applyAsPath:function (director) {
 81 
 82                 var ctx = director.ctx;
 83 
 84                 var point = new CAAT.Math.Point();
 85 
 86                 for (var t = this.k; t <= 1 + this.k; t += this.k) {
 87                     this.solve(point, t);
 88                     ctx.lineTo(point.x, point.y);
 89                 }
 90 
 91                 return this;
 92             },
 93 
 94             /**
 95              * Return the first curve control point.
 96              * @return {CAAT.Point}
 97              */
 98             endCurvePosition:function () {
 99                 return this.coordlist[ this.coordlist.length - 2 ];
100             },
101             /**
102              * Return the last curve control point.
