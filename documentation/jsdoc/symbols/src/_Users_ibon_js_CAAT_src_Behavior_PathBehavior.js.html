1 CAAT.Module({
  2     defines:"CAAT.Behavior.PathBehavior",
  3     aliases: ["CAAT.PathBehavior"],
  4     depends:[
  5         "CAAT.Behavior.BaseBehavior",
  6         "CAAT.Foundation.SpriteImage"
  7     ],
  8     constants : {
  9         autorotate: {
 10             LEFT_TO_RIGHT:  0,
 11             RIGHT_TO_LEFT:  1,
 12             FREE:           2
 13         }
 14     },
 15     extendsClass : "CAAT.Behavior.BaseBehavior",
 16     extendsWith:function () {
 17 
 18         return {
 19 
 20             path:null, // the path to traverse
 21             autoRotate:false, // set whether the actor must be rotated tangentially to the path.
 22             prevX:-1, // private, do not use.
 23             prevY:-1, // private, do not use.
 24 
 25             autoRotateOp: CAAT.Behavior.PathBehavior.autorotate.FREE,
 26 
 27             getPropertyName:function () {
 28                 return "translate";
 29             },
 30 
 31             /**
 32              * Sets an actor rotation to be heading from past to current path's point.
 33              * Take into account that this will be incompatible with rotation Behaviors

 35              * @param autorotate {boolean}

 38              */

 40                 this.autoRotate = autorotate;
 41                 if (autorotateOp !== undefined) {
 42                     this.autoRotateOp = autorotateOp;
 43                 }
 44                 return this;
 45             },

 47              * Set the behavior path.

 51              * @deprecated

 71              * @param tx a float with xoffset.

 78             calculateKeyFrameData:function (time) {
 79                 time = this.interpolator.getPosition(time).y;
 80                 var point = this.path.getPosition(time);
 81                 return "translateX(" + point.x + "px) translateY(" + point.y + "px)";
 82             },
 83 
 84             calculateKeyFramesData:function (prefix, name, keyframessize) {
 85 
 86                 if (typeof keyframessize === 'undefined') {
 87                     keyframessize = 100;
 88                 }
 89                 keyframessize >>= 0;
 90 
 91                 var i;
 92                 var kfr;
 93                 var time;
 94                 var kfd = "@-" + prefix + "-keyframes " + name + " {";
 95 
 96                 for (i = 0; i <= keyframessize; i++) {
 97                     kfr = "" +
 98                         (i / keyframessize * 100) + "%" + // percentage
 99                         "{" +
100                         "-" + prefix + "-transform:" + this.calculateKeyFrameData(i / keyframessize) +
101                         "}";
102 
103                     kfd += kfr;
104                 }
105 
106                 kfd += "}";
107 
108                 return kfd;
109             },
110 
111             /**
112              * Translates the Actor to the corresponding time path position.
113              * If autoRotate=true, the actor is rotated as well. The rotation anchor will (if set) always be ANCHOR_CENTER.
114              * @param time an integer indicating the time the behavior is being applied at.
115              * @param actor a CAAT.Actor instance to be translated.
116              * @return {object} an object of the form <code>{ x: {float}, y: {float}�}</code>.
117              */
118             setForTime:function (time, actor) {
119 
120                 if (!this.path) {
121                     return {
122                         x:actor.x,
123                         y:actor.y
124                     };
125                 }
126 
127                 var point = this.path.getPosition(time);
128 
129                 if (this.autoRotate) {
130 
131                     if (-1 === this.prevX && -1 === this.prevY) {
132                         this.prevX = point.x;
133                         this.prevY = point.y;
134                     }
135 
136                     var ax = point.x - this.prevX;
137                     var ay = point.y - this.prevY;
138 
139                     if (ax === 0 && ay === 0) {
140                         actor.setLocation(point.x, point.y);
141                         return { x:actor.x, y:actor.y };
142                     }
143 
144                     var angle = Math.atan2(ay, ax);
145                     var si = CAAT.Foundation.SpriteImage;
146                     var pba = CAAT.Behavior.PathBehavior.autorotate;
147 
148                     // actor is heading left to right
149                     if (this.autoRotateOp === pba.LEFT_TO_RIGHT) {
150                         if (this.prevX <= point.x) {
151                             actor.setImageTransformation(si.TR_NONE);
152                         }
153                         else {
154                             actor.setImageTransformation(si.TR_FLIP_HORIZONTAL);
155                             angle += Math.PI;
156                         }
157                     } else if (this.autoRotateOp === pba.RIGHT_TO_LEFT) {
158                         if (this.prevX <= point.x) {
159                             actor.setImageTransformation(si.TR_FLIP_HORIZONTAL);
160                         }
161                         else {
162                             actor.setImageTransformation(si.TR_NONE);
163                             angle -= Math.PI;
164                         }
165                     }
166 
167                     actor.setRotation(angle);
168 
169                     this.prevX = point.x;
170                     this.prevY = point.y;
171 
172                     var modulo = Math.sqrt(ax * ax + ay * ay);
173                     ax /= modulo;
174                     ay /= modulo;
175                 }
176 
177                 if (this.doValueApplication) {
178                     actor.setLocation(point.x, point.y);
179                     return { x:actor.x, y:actor.y };
180                 } else {
181                     return {
182                         x:point.x,
183                         y:point.y
184                     };
185                 }
186 
187 
188             },
189             /**
190              * Get a point on the path.
191              * If the time to get the point at is in behaviors frame time, a point on the path will be returned, otherwise
192              * a default {x:-1, y:-1} point will be returned.
193              *
194              * @param time {number} the time at which the point will be taken from the path.
195              * @return {object} an object of the form {x:float y:float}
196              */
197             positionOnTime:function (time) {
198                 if (this.isBehaviorInTime(time, null)) {
199                     time = this.normalizeTime(time);
200                     return this.path.getPosition(time);
201                 }
202 
203                 return {x:-1, y:-1};
204 
205             }
206         };
