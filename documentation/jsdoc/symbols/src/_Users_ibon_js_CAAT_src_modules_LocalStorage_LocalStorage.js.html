  6 (function() {
  7     /**
  8      * Local storage management.
  9      * @constructor
 10      */
 11     CAAT.modules.LocalStorage= function() {
 12         return this;
 13     };
 14 
 15     CAAT.modules.LocalStorage.prototype= {
 16         /**
 17          * Stores an object in local storage. The data will be saved as JSON.stringify.
 18          * @param key {string} key to store data under.
 19          * @param data {object} an object.
 20          * @return this
 21          *
 22          * @static
 23          */
 24         save : function( key, data ) {
 25             try {
 26                 localStorage.setItem( key, JSON.stringify(data) );
 27             } catch(e) {
 28                 // eat it
 29             }
 30             return this;
 31         },
 32         /**
 33          * Retrieve a value from local storage.
 34          * @param key {string} the key to retrieve.
 35          * @return {object} object stored under the key parameter.
 36          *
 37          * @static
 38          */
 39         load : function( key ) {
 40             try {
 41                 return JSON.parse(localStorage.getItem( key ));
 42             } catch(e) {
 43                 return null;
 44             }
 45         },
 46         /**
 47          * Removes a value stored in local storage.
 48          * @param key {string}
 49          * @return this
 50          *
 51          * @static
 52          */
 53         remove : function( key ) {
 54             try {
 55                 localStorage.removeItem(key);
 56             } catch(e) {
 57                 // eat it
 58             }
 59             return this;
 60         }
 61     };
 62 
 63 })();
