314         /**
315          * Custom paint method for TextActor instances.
316          * If the path attribute is set, the text will be drawn traversing the path.
317          *
318          * @param director a valid CAAT.Director instance.
319          * @param time an integer with the Scene time the Actor is being drawn.
320          */
321 		paint : function(director, time) {
322 
323             if (!this.text) {
324                 return;
325             }
326 
327             CAAT.Foundation.UI.TextActor.superclass.paint.call(this, director, time );
328 
329             if ( this.cached ) {
330                 // cacheAsBitmap sets this actor's background image as a representation of itself.
331                 // So if after drawing the background it was cached, we're done.
332                 return;
333             }
334 
335 			if ( null===this.text) {
336 				return;
337 			}
338 
339             if ( this.textWidth===0 || this.textHeight===0 ) {
340                 this.calcTextSize(director);
341             }
342 
343 			var ctx= director.ctx;
344 			
345 			if ( this.font instanceof CAAT.Module.Font.Font || this.font instanceof CAAT.Foundation.SpriteImage ) {
346 				this.drawSpriteText(director,time);
347                 return;
348 			}
349 
350 			if( null!==this.font ) {
351 				ctx.font= this.font;
352 			}
353 
354             /**
355              * always draw text with middle or bottom, top is buggy in FF.
356              * @type {String}
357              */
358             ctx.textBaseline="alphabetic";
359 
360 			if (null===this.path) {
361 
362                 if ( null!==this.textAlign ) {
363                     ctx.textAlign= this.textAlign;
364                 }
365 
366                 var tx=0;
367                 if ( this.textAlign==='center') {
368                     tx= (this.width/2)|0;
369                 } else if ( this.textAlign==='right' ) {
370                     tx= this.width;
371                 }
372 
373 				if ( this.fill ) {
374                     if ( null!==this.textFillStyle ) {
375                         ctx.fillStyle= this.textFillStyle;
376                     }
377 					ctx.fillText( this.text, tx, this.fontData.ascent  );
378 				}
379 
380                 if ( this.outline ) {
381                     if (null!==this.outlineColor ) {
382                         ctx.strokeStyle= this.outlineColor;
383                     }
384 
385                     ctx.lineWidth= this.lineWidth;
386                     ctx.beginPath();
387 					ctx.strokeText( this.text, tx, this.fontData.ascent );
388 				}
389 			}
390 			else {
391 				this.drawOnPath(director,time);
392 			}

394         /**
395          * Private.
396          * Draw the text traversing a path.
397          * @param director a valid CAAT.Director instance.
398          * @param time an integer with the Scene time the Actor is being drawn.
399          */
400 		drawOnPath : function(director, time) {
401 
402 			var ctx= director.ctx;
403 
404             if ( this.fill && null!==this.textFillStyle ) {
405                 ctx.fillStyle= this.textFillStyle;
406             }
407 
408             if ( this.outline && null!==this.outlineColor ) {
409                 ctx.strokeStyle= this.outlineColor;
410             }
411 
412 			var textWidth=this.sign * this.pathInterpolator.getPosition(
413                     (time%this.pathDuration)/this.pathDuration ).y * this.path.getLength() ;
414 			var p0= new CAAT.Math.Point(0,0,0);
415 			var p1= new CAAT.Math.Point(0,0,0);
416 
417 			for( var i=0; i<this.text.length; i++ ) {
418 				var caracter= this.text[i].toString();
419 				var charWidth= ctx.measureText( caracter ).width;
420 
421                 // guonjien: remove "+charWidth/2" since it destroys the kerning. and he's right!!!. thanks.
422 				var currentCurveLength= textWidth;
423 
424 				p0= this.path.getPositionFromLength(currentCurveLength).clone();
425 				p1= this.path.getPositionFromLength(currentCurveLength-0.1).clone();
426 
427 				var angle= Math.atan2( p0.y-p1.y, p0.x-p1.x );
428 
429 				ctx.save();
430 
431 					ctx.translate( p0.x>>0, p0.y>>0 );
432 					ctx.rotate( angle );
433                     if ( this.fill ) {
434 					    ctx.fillText(caracter,0,0);
435                     }
436                     if ( this.outline ) {
437                         ctx.beginPath();
438                         ctx.lineWidth= this.lineWidth;
439                         ctx.strokeText(caracter,0,0);
440                     }
441 
442 				ctx.restore();
443 
444 				textWidth+= charWidth;
445 			}
446 		},

462 		/**
463          * Private.
464          * Draw the text traversing a path using a sprited font.
465          * @param director a valid CAAT.Director instance.
466          * @param time an integer with the Scene time the Actor is being drawn.
467          */
468 		drawSpriteTextOnPath: function(director, time) {
469 			var context= director.ctx;
470 
471 			var textWidth=this.sign * this.pathInterpolator.getPosition(
472                     (time%this.pathDuration)/this.pathDuration ).y * this.path.getLength() ;
473 			var p0= new CAAT.Math.Point(0,0,0);
474 			var p1= new CAAT.Math.Point(0,0,0);
475 
476 			for( var i=0; i<this.text.length; i++ ) {
477 				var character= this.text[i].toString();
478 				var charWidth= this.font.stringWidth(character);
479 
480 				//var pathLength= this.path.getLength();
481 
482 				var currentCurveLength= charWidth/2 + textWidth;
483 
484 				p0= this.path.getPositionFromLength(currentCurveLength).clone();
485 				p1= this.path.getPositionFromLength(currentCurveLength-0.1).clone();
486 
487 				var angle= Math.atan2( p0.y-p1.y, p0.x-p1.x );
488 
489 				context.save();
490 
491 				context.translate( p0.x|0, p0.y|0 );
492 				context.rotate( angle );
493 				
494 				var y = this.textBaseline === "bottom" ? 0 - this.font.getHeight() : 0;
495 				
496 				this.font.drawText(character, context, 0, y);
497 
498 				context.restore();
499 
500 				textWidth+= charWidth;
501 			}
