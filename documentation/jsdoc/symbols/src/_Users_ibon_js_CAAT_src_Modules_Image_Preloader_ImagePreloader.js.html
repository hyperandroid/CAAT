  9 CAAT.Module( {
 10     defines : "CAAT.Module.Preloader.ImagePreloader",
 11     aliases : ["CAAT.ImagePreloader"],
 12     extendsWith : {
 13         __init : function()   {
 14             this.images = [];
 15             return this;
 16         },
 17 
 18         images:                 null,   // a list of elements to load
 19         notificationCallback:   null,   // notification callback invoked for each image loaded.
 20         imageCounter:           0,      // elements counter.
 21 
 22         /**
 23          * Start images loading asynchronous process. This method will notify every image loaded event
 24          * and is responsibility of the caller to count the number of loaded images to see if it fits his
 25          * needs.
 26          * 
 27          * @param aImages {{ id:{url}, id2:{url}, ...} an object with id/url pairs.
 28          * @param callback_loaded_one_image {function( imageloader {CAAT.ImagePreloader}, counter {number}, images {{ id:{string}, image: {Image}}} )}
 29          * function to call on every image load.
 30          */
 31         loadImages: function( aImages, callback_loaded_one_image, callback_error ) {
 32 
 33             if (!aImages) {
 34                 if (callback_loaded_one_image ) {
 35                     callback_loaded_one_image(0,[]);
 36                 }
 37             }
 38 
 39             var me= this, i;
 40             this.notificationCallback = callback_loaded_one_image;
 41             this.images= [];
 42             for( i=0; i<aImages.length; i++ ) {
 43                 this.images.push( {id:aImages[i].id, image: new Image() } );
 44             }
 45 
 46             for( i=0; i<aImages.length; i++ ) {
 47                 this.images[i].image.onload = function imageLoaded() {
 48                     me.imageCounter++;
 49                     me.notificationCallback(me.imageCounter, me.images);
 50                 };
 51 
 52                 this.images[i].image.onerror= (function(index) {
 53                         return function(e) {
 54                             if ( callback_error ) {
 55                                 callback_error( e, index );
 56                             }
 57                         }
 58                     })(i);
 59 
 60                 this.images[i].image.src= aImages[i].url;
 61             }
 62 
 63             if ( aImages.length===0 ) {
 64                 callback_loaded_one_image(0,[]);
 65             }
 66         }
 67 
 68     }
 69 });
