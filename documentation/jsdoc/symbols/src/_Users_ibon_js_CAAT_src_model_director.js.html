474                 }

476             } else {
477                 this.imagesCache.push( { id: id, image: image } );

482                 this.updateGLPages( );

563         /**
564          * This is the entry point for the animation system of the Director.
565          * The director is fed with the elapsed time value to maintain a virtual timeline.
566          * This virtual timeline will provide each Scene with its own virtual timeline, and will only
567          * feed time when the Scene is the current Scene, or is being switched.
568          *
569          * If dirty rectangles are enabled and canvas is used for rendering, the dirty rectangles will be
570          * set up as a single clip area.
571          *
572          * @param time {number} integer indicating the elapsed time between two consecutive frames of the
573          * Director.
574          */
575         render : function(time) {
576 
577             this.time += time;
578 
579             this.animate(this,time);
580 
581             if ( CAAT.DEBUG ) {
582                 this.resetStats();
583             }
584 
585             /**
586              * draw director active scenes.
587              */
588             var ne = this.childrenList.length;
589             var i, tt, c;
590             var ctx= this.ctx;
591 
592             if (this.glEnabled) {
593 
594                 this.gl.clear(this.gl.COLOR_BUFFER_BIT | this.gl.DEPTH_BUFFER_BIT);
595                 this.coordsIndex = 0;
596                 this.uvIndex = 0;
597 
598                 for (i = 0; i < ne; i++) {
599                     c = this.childrenList[i];
600                     if (c.isInAnimationFrame(this.time)) {
601                         tt = c.time - c.start_time;
602                         if ( c.onRenderStart ) {
603                             c.onRenderStart(tt);
604                         }
605                         c.paintActorGL(this, tt);
606                         if ( c.onRenderEnd ) {
607                             c.onRenderEnd(tt);
608                         }
609 
610                         if ( !c.isPaused() ) {
611                             c.time += time;
612                         }
613 
614                         if ( CAAT.DEBUG ) {
615                             this.statistics.size_total+= c.size_total;
616                             this.statistics.size_active+= c.size_active;
617                         }
618 
619                     }
620                 }
621 
622                 this.glFlush();
623 
624             } else {
625                 ctx.globalAlpha = 1;
626                 ctx.globalCompositeOperation = 'source-over';
627 
628                 ctx.save();
629                 if ( this.dirtyRectsEnabled ) {
630                     this.modelViewMatrix.transformRenderingContext( ctx );
631 
632                     if ( !CAAT.DEBUG_DIRTYRECTS ) {
633                         ctx.beginPath();
634                         this.nDirtyRects=0;
635                         var dr= this.cDirtyRects;
636                         for( i=0; i<dr.length; i++ ) {
637                             var drr= dr[i];
638                             if ( !drr.isEmpty() ) {
639                                 ctx.rect( drr.x|0, drr.y|0, 1+(drr.width|0), 1+(drr.height|0) );
640                                 this.nDirtyRects++;
641                             }
642                         }
643                         ctx.clip();
644                     } else {
645                         ctx.clearRect(0, 0, this.width, this.height);
646                     }
647 
648                 } else if (this.clear===true ) {
649                     ctx.clearRect(0, 0, this.width, this.height);
650                 }
651 
652                 for (i = 0; i < ne; i++) {
653                     c= this.childrenList[i];
654 
655                     if (c.isInAnimationFrame(this.time)) {
656                         tt = c.time - c.start_time;
657                         ctx.save();
658 
659                         if ( c.onRenderStart ) {
660                             c.onRenderStart(tt);
661                         }
662 
663                         if ( !CAAT.DEBUG_DIRTYRECTS && this.dirtyRectsEnabled ) {
664                             if ( this.nDirtyRects ) {
665                                 c.paintActor(this, tt);
666                             }
667                         } else {
668                             c.paintActor(this, tt);
669                         }
670 
671                         if ( c.onRenderEnd ) {
672                             c.onRenderEnd(tt);
673                         }
674                         ctx.restore();
675 
676                         if (CAAT.DEBUGAABB) {
677                             ctx.globalAlpha= 1;
678                             ctx.globalCompositeOperation= 'source-over';
679                             this.modelViewMatrix.transformRenderingContextSet( ctx );
680                             c.drawScreenBoundingBox(this, tt);
681                         }
682 
683                         if ( !c.isPaused() ) {
684                             c.time += time;
685                         }
686 
687                         if ( CAAT.DEBUG ) {
688                             this.statistics.size_total+= c.size_total;
689                             this.statistics.size_active+= c.size_active;
690                             this.statistics.size_dirtyRects= this.nDirtyRects;
691                         }
692 
693                     }
694                 }
695 
696                 if ( this.nDirtyRects>0 && CAAT.DEBUG && CAAT.DEBUG_DIRTYRECTS ) {
697                     ctx.beginPath();
698                     this.nDirtyRects=0;
699                     var dr= this.cDirtyRects;
700                     for( i=0; i<dr.length; i++ ) {
701                         var drr= dr[i];
702                         if ( !drr.isEmpty() ) {
703                             ctx.rect( drr.x|0, drr.y|0, 1+(drr.width|0), 1+(drr.height|0) );
704                             this.nDirtyRects++;
705                         }
706                     }
707 
708                     ctx.clip();
709                     ctx.fillStyle='rgba(160,255,150,.4)';
710                     ctx.fillRect(0,0,this.width, this.height);
711                 }
712 
713                 ctx.restore();
714             }
715 
716             this.frameCounter++;
717         },
718         /**

755          * rectangles will be removed and intersecting ones will be unioned.
