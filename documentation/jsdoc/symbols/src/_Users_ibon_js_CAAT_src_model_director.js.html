430          * If many handcrafted images are to be added to the director, some performance can be achieved by calling

432          * <code>director.addImage(id,image,true)</code> to finally command the director to create texture pages.
433          *
434          * @param id {string|object} an identitifier to retrieve the image with
435          * @param image {Image|HTMLCanvasElement} image to add to cache
436          * @param noUpdateGL {!boolean} unless otherwise stated, the director will

445                     }
446                 }

448             } else {
449                 this.imagesCache.push( { id: id, image: image } );
450                 this.imagesCache[id]= image;
451             }
452 
453             if ( !!!noUpdateGL ) {
454                 this.updateGLPages( );
455             }

535         /**
536          * This is the entry point for the animation system of the Director.
537          * The director is fed with the elapsed time value to maintain a virtual timeline.
538          * This virtual timeline will provide each Scene with its own virtual timeline, and will only
539          * feed time when the Scene is the current Scene, or is being switched.
540          *
541          * If dirty rectangles are enabled and canvas is used for rendering, the dirty rectangles will be
542          * set up as a single clip area.
543          *
544          * @param time {number} integer indicating the elapsed time between two consecutive frames of the
545          * Director.
546          */
547         render : function(time) {
548 
549             this.time += time;
550 
551             this.animate(this,time);
552 
553             if ( CAAT.DEBUG ) {
554                 this.resetStats();
555             }
556 
557             /**
558              * draw director active scenes.
559              */
560             var ne = this.childrenList.length;
561             var i, tt, c;
562             var ctx= this.ctx;
563 
564             if (this.glEnabled) {
565 
566                 this.gl.clear(this.gl.COLOR_BUFFER_BIT | this.gl.DEPTH_BUFFER_BIT);
567                 this.coordsIndex = 0;
568                 this.uvIndex = 0;
569 
570                 for (i = 0; i < ne; i++) {
571                     c = this.childrenList[i];
572                     if (c.isInAnimationFrame(this.time)) {
573                         tt = c.time - c.start_time;
574                         if ( c.onRenderStart ) {
575                             c.onRenderStart(tt);
576                         }
577                         c.paintActorGL(this, tt);
578                         if ( c.onRenderEnd ) {
579                             c.onRenderEnd(tt);
580                         }
581 
582                         if ( !c.isPaused() ) {
583                             c.time += time;
584                         }
585 
586                         if ( CAAT.DEBUG ) {
587                             this.statistics.size_total+= c.size_total;
588                             this.statistics.size_active+= c.size_active;
589                         }
590 
591                     }
592                 }
593 
594                 this.glFlush();
595 
596             } else {
597                 ctx.globalAlpha = 1;
598                 ctx.globalCompositeOperation = 'source-over';
599 
600                 ctx.save();
601                 if ( this.dirtyRectsEnabled ) {
602                     this.modelViewMatrix.transformRenderingContext( ctx );
603 
604                     if ( !CAAT.DEBUG_DIRTYRECTS ) {
605                         ctx.beginPath();
606                         this.nDirtyRects=0;
607                         var dr= this.cDirtyRects;
608                         for( i=0; i<dr.length; i++ ) {
609                             var drr= dr[i];
610                             if ( !drr.isEmpty() ) {
611                                 //ctx.rect( (drr.x|0)+.5, (drr.y|0)+.5, 1+(drr.width|0), 1+(drr.height|0) );
612                                 ctx.rect( drr.x|0, drr.y|0, 1+(drr.width|0), 1+(drr.height|0) );
613                                 this.nDirtyRects++;
614                             }
615                         }
616                         ctx.clip();
617                     } else {
618                         ctx.clearRect(0, 0, this.width, this.height);
619                     }
620 
621                 } else if (this.clear===true ) {
622                     ctx.clearRect(0, 0, this.width, this.height);
623                 }
624 
625                 for (i = 0; i < ne; i++) {
626                     c= this.childrenList[i];
627 
628                     if (c.isInAnimationFrame(this.time)) {
629                         tt = c.time - c.start_time;
630                         ctx.save();
631 
632                         if ( c.onRenderStart ) {
633                             c.onRenderStart(tt);
634                         }
635 
636                         if ( !CAAT.DEBUG_DIRTYRECTS && this.dirtyRectsEnabled ) {
637                             if ( this.nDirtyRects ) {
638                                 c.paintActor(this, tt);
639                             }
640                         } else {
641                             c.paintActor(this, tt);
642                         }
643 
644                         if ( c.onRenderEnd ) {
645                             c.onRenderEnd(tt);
646                         }
647                         ctx.restore();
648 
649                         if (CAAT.DEBUGAABB) {
650                             ctx.globalAlpha= 1;
651                             ctx.globalCompositeOperation= 'source-over';
652                             this.modelViewMatrix.transformRenderingContextSet( ctx );
653                             c.drawScreenBoundingBox(this, tt);
654                         }
655 
656                         if ( !c.isPaused() ) {
657                             c.time += time;
658                         }
659 
660                         if ( CAAT.DEBUG ) {
661                             this.statistics.size_total+= c.size_total;
662                             this.statistics.size_active+= c.size_active;
663                             this.statistics.size_dirtyRects= this.nDirtyRects;
664                         }
665 
666                     }
667                 }
668 
669                 if ( this.nDirtyRects>0 && CAAT.DEBUG && CAAT.DEBUG_DIRTYRECTS ) {
670                     ctx.beginPath();
671                     this.nDirtyRects=0;
672                     var dr= this.cDirtyRects;
673                     for( i=0; i<dr.length; i++ ) {
674                         var drr= dr[i];
675                         if ( !drr.isEmpty() ) {
676                             ctx.rect( drr.x|0, drr.y|0, 1+(drr.width|0), 1+(drr.height|0) );
677                             this.nDirtyRects++;
678                         }
679                     }
680 
681                     ctx.clip();
682                     ctx.fillStyle='rgba(160,255,150,.4)';
683                     ctx.fillRect(0,0,this.width, this.height);
684                 }
685 
686                 ctx.restore();
687             }
688 
689             this.frameCounter++;
690         },
