 28         /**
 29          * HSV to RGB color conversion
 30          * <p>
 31          * H runs from 0 to 360 degrees<br>
 32          * S and V run from 0 to 100
 33          * <p>
 34          * Ported from the excellent java algorithm by Eugene Vishnevsky at:
 35          * http://www.cs.rit.edu/~ncs/color/t_convert.html
 36          *
 37          * @static
 38          */
 39         hsvToRgb:function (h, s, v) {
 40             var r, g, b, i, f, p, q, t;
 41 
 42             // Make sure our arguments stay in-range
 43             h = Math.max(0, Math.min(360, h));
 44             s = Math.max(0, Math.min(100, s));
 45             v = Math.max(0, Math.min(100, v));
 46 
 47             // We accept saturation and value arguments from 0 to 100 because that's
 48             // how Photoshop represents those values. Internally, however, the
 49             // saturation and value are calculated from a range of 0 to 1. We make
 50             // That conversion here.
 51             s /= 100;
 52             v /= 100;
 53 
 54             if (s === 0) {
 55                 // Achromatic (grey)
 56                 r = g = b = v;
 57                 return [Math.round(r * 255), Math.round(g * 255), Math.round(b * 255)];
 58             }
 59 
 60             h /= 60; // sector 0 to 5
 61             i = Math.floor(h);
 62             f = h - i; // factorial part of h
 63             p = v * (1 - s);
 64             q = v * (1 - s * f);
 65             t = v * (1 - s * (1 - f));
 66 
 67             switch (i) {
 68                 case 0:
 69                     r = v;
 70                     g = t;
 71                     b = p;
 72                     break;
 73 
 74                 case 1:
 75                     r = q;
 76                     g = v;
 77                     b = p;
 78                     break;
 79 
 80                 case 2:
 81                     r = p;
 82                     g = v;
 83                     b = t;
 84                     break;
 85 
 86                 case 3:
 87                     r = p;
 88                     g = q;
 89                     b = v;
 90                     break;
 91 
 92                 case 4:
 93                     r = t;
 94                     g = p;
 95                     b = v;
 96                     break;
 97 
 98                 default: // case 5:
 99                     r = v;
100                     g = p;
101                     b = q;
102             }
103 
104             return new CAAT.Module.ColorUtil.Color(Math.round(r * 255), Math.round(g * 255), Math.round(b * 255));
105         },
106 
107         /**
108          * Interpolate the color between two given colors. The return value will be a calculated color
109          * among the two given initial colors which corresponds to the 'step'th color of the 'nsteps'
110          * calculated colors.
111          * @param r0 {number} initial color red component.
112          * @param g0 {number} initial color green component.
113          * @param b0 {number} initial color blue component.
114          * @param r1 {number} final color red component.
115          * @param g1 {number} final color green component.
116          * @param b1 {number} final color blue component.
117          * @param nsteps {number} number of colors to calculate including the two given colors. If 16 is passed as value,
118          * 14 colors plus the two initial ones will be calculated.
119          * @param step {number} return this color index of all the calculated colors.
120          *
121          * @return { r{number}, g{number}, b{number} } return an object with the new calculated color components.
122          * @static
123          */
124         interpolate:function (r0, g0, b0, r1, g1, b1, nsteps, step) {
125 
126             var r, g, b;
127 
128             if (step <= 0) {
129                 return {
130                     r:r0,
131                     g:g0,
132                     b:b0
133                 };
134             } else if (step >= nsteps) {
135                 return {
136                     r:r1,
137                     g:g1,
138                     b:b1
139                 };
140             }
141 
142             r = (r0 + (r1 - r0) / nsteps * step) >> 0;
143             g = (g0 + (g1 - g0) / nsteps * step) >> 0;
144             b = (b0 + (b1 - b0) / nsteps * step) >> 0;
145 
146             if (r > 255) {
147                 r = 255;
148             } else if (r < 0) {
149                 r = 0;
150             }
151             if (g > 255) {
152                 g = 255;
153             } else if (g < 0) {
154                 g = 0;
155             }
156             if (b > 255) {
157                 b = 255;
158             } else if (b < 0) {
159                 b = 0;
160             }
161 
162             return {
163                 r:r,
164                 g:g,
165                 b:b
166             };
167         },
168 
169         /**
170          * Generate a ramp of colors from an array of given colors.
171          * @param fromColorsArray {[number]} an array of colors. each color is defined by an integer number from which
172          * color components will be extracted. Be aware of the alpha component since it will also be interpolated for
173          * new colors.
174          * @param rampSize {number} number of colors to produce.
175          * @param returnType {CAAT.ColorUtils.RampEnumeration} a value of CAAT.ColorUtils.RampEnumeration enumeration.
176          *
177          * @return { [{number},{number},{number},{number}] } an array of integers each of which represents a color of
178          * the calculated color ramp.
179          *
180          * @static
181          */
182         makeRGBColorRamp:function (fromColorsArray, rampSize, returnType) {
183 
184             var ramp = [], nc = fromColorsArray.length - 1, chunk = rampSize / nc, i, j,
185                 na, nr, ng, nb,
186                 c, a0, r0, g0, b0,
187                 c1, a1, r1, g1, b1,
188                 da, dr, dg, db;
189 
190             for (i = 0; i < nc; i += 1) {
191                 c = fromColorsArray[i];
192                 a0 = (c >> 24) & 0xff;
193                 r0 = (c & 0xff0000) >> 16;
194                 g0 = (c & 0xff00) >> 8;
195                 b0 = c & 0xff;
196 
197                 c1 = fromColorsArray[i + 1];
198                 a1 = (c1 >> 24) & 0xff;
199                 r1 = (c1 & 0xff0000) >> 16;
200                 g1 = (c1 & 0xff00) >> 8;
201                 b1 = c1 & 0xff;
202 
203                 da = (a1 - a0) / chunk;
204                 dr = (r1 - r0) / chunk;
205                 dg = (g1 - g0) / chunk;
206                 db = (b1 - b0) / chunk;
207 
208                 for (j = 0; j < chunk; j += 1) {
209                     na = (a0 + da * j) >> 0;
210                     nr = (r0 + dr * j) >> 0;
211                     ng = (g0 + dg * j) >> 0;
212                     nb = (b0 + db * j) >> 0;
213 
214                     var re = CAAT.Module.ColorUtil.Color.RampEnumeration;
215 
216                     switch (returnType) {
217                         case re.RAMP_RGBA:
218                             ramp.push('argb(' + na + ',' + nr + ',' + ng + ',' + nb + ')');
219                             break;
220                         case re.RAMP_RGB:
221                             ramp.push('rgb(' + nr + ',' + ng + ',' + nb + ')');
222                             break;
223                         case re.RAMP_CHANNEL_RGB:
224                             ramp.push(0xff000000 | nr << 16 | ng << 8 | nb);
225                             break;
226                         case re.RAMP_CHANNEL_RGBA:
227                             ramp.push(na << 24 | nr << 16 | ng << 8 | nb);
228                             break;
229                         case re.RAMP_CHANNEL_RGBA_ARRAY:
230                             ramp.push([ nr, ng, nb, na ]);
231                             break;
232                         case re.RAMP_CHANNEL_RGB_ARRAY:
233                             ramp.push([ nr, ng, nb ]);
234                             break;
235                     }
236                 }
237             }
238 
239             return ramp;
240 
241         },
242 
243         random:function () {
244             var a = '0123456789abcdef';
245             var c = '#';
246             for (var i = 0; i < 3; i++) {
247                 c += a[ (Math.random() * a.length) >> 0 ];
248             }
249             return c;
250         }

265         /**
266          * Get color hexadecimal representation.
267          * @return {string} a string with color hexadecimal representation.
268          */
269         toHex:function () {
270             // See: http://jsperf.com/rgb-decimal-to-hex/5
271             return ('000000' + ((this.r << 16) + (this.g << 8) + this.b).toString(16)).slice(-6);
272         }
