 10 (function() {
 11     /**
 12      * This class is a image resource loader. It accepts an object of the form:
 13      *
 14      * {
 15      *   id1: string_url1,
 16      *   id2: string_url2,
 17      *   id3: string_url3,
 18      *   ...
 19      * }
 20      *
 21      * and on resources loaded correctly, will return an object of the form:
 22      *
 23      * {
 24      *   id1: HTMLImageElement,
 25      *   id2: HTMLImageElement,
 26      *   id3: HTMLImageElement,
 27      *   ...
 28      * }
 29      *
 30      * @constructor
 31      */
 32     CAAT.ImagePreloader = function()   {
 33         this.images = [];
 34         return this;
 35     };
 36 
 37     CAAT.ImagePreloader.prototype =   {
 38 
 39         images:                 null,   // a list of elements to load
 40         notificationCallback:   null,   // notification callback invoked for each image loaded.
 41         imageCounter:           0,      // elements counter.
 42 
 43         /**
 44          * Start images loading asynchronous process. This method will notify every image loaded event
 45          * and is responsibility of the caller to count the number of loaded images to see if it fits his
 46          * needs.
 47          * 
 48          * @param aImages {{ id:{url}, id2:{url}, ...} an object with id/url pairs.
 49          * @param callback_loaded_one_image {function( imageloader {CAAT.ImagePreloader}, counter {number}, images {{ id:{string}, image: {Image}}} )}
 50          * function to call on every image load.
 51          */
 52         loadImages: function( aImages, callback_loaded_one_image, callback_error ) {
 53 
 54             if (!aImages) {
 55                 if (callback_loaded_one_image ) {
 56                     callback_loaded_one_image(0,[]);
 57                 }
 58             }
 59 
 60             var me= this, i;
 61             this.notificationCallback = callback_loaded_one_image;
 62             this.images= [];
 63             for( i=0; i<aImages.length; i++ ) {
 64                 this.images.push( {id:aImages[i].id, image: new Image() } );
 65             }
 66 
 67             for( i=0; i<aImages.length; i++ ) {
 68                 this.images[i].image.onload = function imageLoaded() {
 69                     me.imageCounter++;
 70                     me.notificationCallback(me.imageCounter, me.images);
 71                 };
 72 
 73                 this.images[i].image.onerror= (function(index) {
 74                         return function(e) {
 75                             if ( callback_error ) {
 76                                 callback_error( e, index );
 77                             }
 78                         }
 79                     })(i);
 80 
 81                 this.images[i].image.src= aImages[i].url;
 82             }
 83 
 84             if ( aImages.length===0 ) {
 85                 callback_loaded_one_image(0,[]);
 86             }
 87         }
 88 
 89     };
 90 })();
