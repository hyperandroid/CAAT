1 CAAT.Module({
  2     defines : "CAAT.Module.TexturePacker.TextureScanMap",
  3     depends : [
  4         "CAAT.Module.TexturePacker.TextureScan"
  5     ],
  6     extendsWith : {
  7         __init : function(w,h) {
  8             this.scanMapHeight= h;
  9             this.scanMapWidth= w;
 10 
 11             this.scanMap= [];
 12             for( var i=0; i<this.scanMapHeight; i++ ) {
 13                 this.scanMap.push( new CAAT.Module.TexturePacker.TextureScan(this.scanMapWidth) );
 14             }
 15 
 16             return this;
 17         },
 18 
 19         scanMap:        null,
 20         scanMapWidth:   0,
 21         scanMapHeight:  0,
 22 
 23         /**
 24          * Always try to fit a chunk of size width*height pixels from left-top.
 25          * @param width
 26          * @param height
 27          */
 28         whereFitsChunk : function( width, height ) {
 29 
 30             // trivial rejection:
 31             if ( width>this.width||height>this.height) {
 32                 return null;
 33             }
 34 
 35             // find first fitting point
 36             var i,j,initialPosition= 0;
 37 
 38             while( initialPosition<=this.scanMapHeight-height) {
 39 
 40                 // para buscar sitio se buscara un sitio hasta el tamano de alto del trozo.
 41                 // mas abajo no va a caber.
 42 
 43                 // fitHorizontalPosition es un array con todas las posiciones de este scan donde
 44                 // cabe un chunk de tamano width.
 45                 var fitHorizontalPositions= null;
 46                 var foundPositionOnScan=    false;
 47 
 48                 for( ; initialPosition<=this.scanMapHeight-height; initialPosition++ ) {
 49                     fitHorizontalPositions= this.scanMap[ initialPosition ].findWhereFits( width );
 50 
 51                     // si no es nulo el array de resultados, quiere decir que en alguno de los puntos
 52                     // nos cabe un trozo de tamano width.
 53                     if ( null!==fitHorizontalPositions && fitHorizontalPositions.length>0 ) {
 54                         foundPositionOnScan= true;
 55                         break;
 56                     }
 57                 }
 58 
 59                 if ( foundPositionOnScan ) {
 60                     // j es el scan donde cabe un trozo de tamano width.
 61                     // comprobamos desde este scan que en todos los scan verticales cabe el trozo.
 62                     // se comprueba que cabe en alguno de los tamanos que la rutina de busqueda horizontal
 63                     // nos ha devuelto antes.
 64 
 65                     var minInitialPosition=Number.MAX_VALUE;
 66                     for( j=0; j<fitHorizontalPositions.length; j++ ) {
 67                         var fits= true;
 68                         for( i=initialPosition; i<initialPosition+height; i++ ) {
 69                             // hay un trozo que no cabe
 70                             if ( !this.scanMap[i].fits( fitHorizontalPositions[j], width ) ) {
 71                                 fits= false;
 72                                 break;
 73                             }
 74                         }
 75 
 76                         // se ha encontrado un trozo donde la imagen entra.
 77                         // d.p.m. incluirla en posicion, y seguir con otra.
 78                         if ( fits ) {
 79                             return { x: fitHorizontalPositions[j], y: initialPosition };
 80                         } 
 81                     }
 82 
 83                     initialPosition++;
 84                 } else {
 85                     // no hay sitio en ningun scan.
 86                     return null;
 87                 }
 88             }
 89 
 90             // no se ha podido encontrar un area en la textura para un trozo de tamano width*height
 91             return null;
 92         },
 93         substract : function( x,y, width, height ) {
 94             for( var i=0; i<height; i++ ) {
 95                 if ( !this.scanMap[i+y].substract(x,width) ) {
 96                     CAAT.log('Error: removing chunk ',width,height,' at ',x,y);
 97                 }
 98             }
 99         },
100         log : function() {
101             for( var i=0; i<this.scanMapHeight; i++ ) {
102                 this.scanMap[i].log(i);
103             }
104         }
105     }
