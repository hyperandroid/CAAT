 43              *
 44              * @param time the time to apply the scale for.
 45              * @param actor the target actor to Scale.
 46              * @return {object} an object of the form <code>{ scaleX: {float}, scaleY: {float}ï¿½}</code>
 47              */
 48             setForTime:function (time, actor) {
 49 
 50                 var scaleX = this.startScaleX + time * (this.endScaleX - this.startScaleX);
 51                 var scaleY = this.startScaleY + time * (this.endScaleY - this.startScaleY);
 52 
 53                 // Firefox 3.x & 4, will crash animation if either scaleX or scaleY equals 0.
 54                 if (0 === scaleX) {
 55                     scaleX = 0.01;
 56                 }
 57                 if (0 === scaleY) {
 58                     scaleY = 0.01;
 59                 }
 60 
 61                 if (this.doValueApplication) {
 62                     actor.setScaleAnchored(scaleX, scaleY, this.anchorX, this.anchorY);
 63                 }
 64 
 65                 return { scaleX:scaleX, scaleY:scaleY };
