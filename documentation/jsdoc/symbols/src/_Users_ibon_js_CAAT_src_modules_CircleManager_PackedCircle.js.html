 21 CAAT.Module( {
 22     defines : "CAAT.Module.CircleManager.PackedCircle",
 23     depends : [
 24         "CAAT.Module.CircleManager.PackedCircle",
 25         "CAAT.Math.Point"
 26     ],
 27     constants : {
 28         BOUNDS_RULE_WRAP:		1,      // Wrap to otherside
 29         BOUNDS_RULE_CONSTRAINT:	2,      // Constrain within bounds
 30         BOUNDS_RULE_DESTROY:	4,      // Destroy when it reaches the edge
 31         BOUNDS_RULE_IGNORE:		8		// Ignore when reaching bounds
 32     },
 33     extendsWith : {
 34         __init : function() {
 35             this.boundsRule = CAAT.Module.CircleManager.PackedCircle.BOUNDS_RULE_IGNORE;
 36             this.position = new CAAT.Math.Point(0,0,0);
 37             this.offset = new CAAT.Math.Point(0,0,0);
 38             this.targetPosition = new CAAT.Math.Point(0,0,0);
 39             return this;
 40         },
 41 
 42 		id:             0,
 43 		delegate:		null,
 44 		position:		null,
 45 		offset:			null,
 46 
 47 		targetPosition:	null,	// Where it wants to go
 48 		targetChaseSpeed: 0.02,
 49 
 50 		isFixed:		false,
 51 		boundsRule:		0,
 52 		collisionMask:	0,
 53 		collisionGroup:	0,
 54 
 55 		containsPoint: function(aPoint)
 56 		{
 57 			var distanceSquared = this.position.getDistanceSquared(aPoint);
 58 			return distanceSquared < this.radiusSquared;
 59 		},
 60 
 61 		getDistanceSquaredFromPosition: function(aPosition)
 62 		{
 63 			var distanceSquared = this.position.getDistanceSquared(aPosition);
 64 			// if it's shorter than either radius, we intersect
 65 			return distanceSquared < this.radiusSquared;
 66 		},
 67 
 68 		intersects: function(aCircle)
 69 		{
 70 			var distanceSquared = this.position.getDistanceSquared(aCircle.position);
 71 			return (distanceSquared < this.radiusSquared || distanceSquared < aCircle.radiusSquared);
 72 		},
 73 
 74 /**
 75  * ACCESSORS
 76  */
 77 		setPosition: function(aPosition)
 78 		{
 79 			this.position = aPosition;
 80 			return this;
 81 		},
 82 
 83 		setDelegate: function(aDelegate)
 84 		{
 85 			this.delegate = aDelegate;
 86 			return this;
 87 		},
 88 
 89 		setOffset: function(aPosition)
 90 		{
 91 			this.offset = aPosition;
 92 			return this;
 93 		},
 94 
 95 		setTargetPosition: function(aTargetPosition)
 96 		{
 97 			this.targetPosition = aTargetPosition;
 98 			return this;
 99 		},
100 
101 		setTargetChaseSpeed: function(aTargetChaseSpeed)
102 		{
103 			this.targetChaseSpeed = aTargetChaseSpeed;
104 			return this;
105 		},
106 
107 		setIsFixed: function(value)
108 		{
109 			this.isFixed = value;
110 			return this;
111 		},
112 
113 		setCollisionMask: function(aCollisionMask)
114 		{
115 			this.collisionMask = aCollisionMask;
116 			return this;
117 		},
118 
119 		setCollisionGroup: function(aCollisionGroup)
120 		{
121 			this.collisionGroup = aCollisionGroup;
122 			return this;
123 		},
124 
125 		setRadius: function(aRadius)
126 		{
127 			this.radius = aRadius;
128 			this.radiusSquared = this.radius*this.radius;
129 			return this;
130 		},
131 
132 		initialize : function(overrides)
133 		{
134 			if (overrides)
135 			{
136 				for (var i in overrides)
137 				{
138 					this[i] = overrides[i];
139 				}
140 			}
141 
142 			return this;
143 		},
144 
145 		dealloc: function()
146 		{
147 			this.position = null;
148 			this.offset = null;
149 			this.delegate = null;
150 			this.targetPosition = null;
151 		}
152 	}
153 });
