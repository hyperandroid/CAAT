  5 CAAT.Module({
  6     defines : "CAAT.Module.TexturePacker.TexturePageManager",
  7     depends : [
  8         "CAAT.Module.TexturePacker.TexturePage"
  9     ],
 10     extendsWith : {
 11         __init : function() {
 12             this.pages= [];
 13             return this;
 14         },
 15 
 16         pages:  null,
 17         createPages:    function(gl,width,height,imagesCache) {
 18 
 19             var end= false;
 20             while( !end ) {
 21                 var page= new CAAT.Module.TexturePacker.TexturePage(width,height);
 22                 page.create(imagesCache);
 23                 page.initialize(gl);
 24                 page.endCreation();
 25                 this.pages.push(page);
 26 
 27                 end= true;
 28                 for( var i=0; i<imagesCache.length; i++ ) {
 29                     // imagen sin asociacion de textura
 30                     if ( !imagesCache[i].image.__texturePage ) {
 31                         // cabe en la pagina ?? continua con otras paginas.
 32                         if ( imagesCache[i].image.width<=width && imagesCache[i].image.height<=height ) {
 33                             end= false;
 34                         }
 35                         break;
 36                     }
 37                 }
 38             }
 39         },
 40         deletePages : function() {
 41             for( var i=0; i<this.pages.length; i++ ) {
 42                 this.pages[i].deletePage();
 43             }
 44             this.pages= null;
 45         }
 46     }
 47 
 48 });
