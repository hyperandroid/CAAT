  9 CAAT.Module( {
 10     defines : "CAAT.Module.Preloader.Preloader",
 11     extendsWith : function() {
 12 
 13         var descriptor= function(id, path, loader) {
 14 
 15             var me= this;
 16 
 17             this.id=    id;
 18             this.path=  path;
 19             this.image= new Image();
 20 
 21             this.image.onload = function() {
 22                 loader.__onload(me);
 23             };
 24 
 25             this.image.onerror= function(e) {
 26                 loader.__onerror(me);
 27             } ;
 28 
 29             this.load= function() {
 30                 me.image.src= me.path;
 31             };
 32 
 33             return this;
 34         };
 35 
 36         return {
 37             __init : function()   {
 38                 this.elements= [];
 39                 return this;
 40             },
 41 
 42             elements:       null,   // a list of elements to load
 43             imageCounter:   0,      // elements counter.
 44             cfinished:      null,
 45             cloaded:        null,
 46             cerrored:       null,
 47             loadedCount:    0,
 48 
 49             addElement : function( id, path ) {
 50                 this.elements.push( new descriptor(id,path,this) );
 51                 return this;
 52             },
 53 
 54             __onload : function( d ) {
 55                 if ( this.cloaded ) {
 56                     this.cloaded(d.id);
 57                 }
 58 
 59                 this.loadedCount++;
 60                 if ( this.loadedCount===this.elements.length ) {
 61                     if ( this.cfinished ) {
 62                         this.cfinished( this.elements );
 63                     }
 64                 }
 65             },
 66 
 67             __onerror : function( d ) {
 68                 if ( this.cerrored ) {
 69                     this.cerrored(d.id);
 70                 }
 71             },
 72 
 73             load: function( onfinished, onload_one, onerror ) {
 74 
 75                 this.cfinished= onfinished;
 76                 this.cloaded= onload_one;
 77                 this.cerroed= onerror;
 78 
 79                 var i;
 80 
 81                 for( i=0; i<this.elements.length; i++ ) {
 82                     this.elements[i].load();
 83                 }
 84 
 85                 return this;
 86             }
 87         }
 88     }
 89 });
