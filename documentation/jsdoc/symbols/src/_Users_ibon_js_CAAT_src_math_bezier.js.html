  6 CAAT.Module( {
  7     defines:    "CAAT.Math.Bezier",
  8     depends:    ["CAAT.Math.Curve"],
  9     extendsClass:    "CAAT.Math.Curve",
 10     aliases:    ["CAAT.Bezier"],
 11     extendsWith:    function() {
 12             return {
 13 
 14             cubic:		false,
 15 
 16             applyAsPath : function( director ) {
 17 
 18                 var cc= this.coordlist;
 19 
 20                 if ( this.cubic ) {
 21                     director.ctx.bezierCurveTo(
 22                         cc[1].x,
 23                         cc[1].y,
 24                         cc[2].x,
 25                         cc[2].y,
 26                         cc[3].x,
 27                         cc[3].y
 28                     );
 29                 } else {
 30                     director.ctx.quadraticCurveTo(
 31                         cc[1].x,
 32                         cc[1].y,
 33                         cc[2].x,
 34                         cc[2].y
 35                     );
 36                 }
 37                 return this;
 38             },
 39             isQuadric : function() {
 40                 return !this.cubic;
 41             },
 42             isCubic : function() {
 43                 return this.cubic;
 44             },
 45             /**
 46              * Set this curve as a cubic bezier defined by the given four control points.
 47              * @param cp0x {number}
 48              * @param cp0y {number}
 49              * @param cp1x {number}
 50              * @param cp1y {number}
 51              * @param cp2x {number}
 52              * @param cp2y {number}
 53              * @param cp3x {number}
 54              * @param cp3y {number}
 55              */
 56             setCubic : function( cp0x,cp0y, cp1x,cp1y, cp2x,cp2y, cp3x,cp3y ) {
 57 
 58                 this.coordlist= [];
 59 
 60                 this.coordlist.push( new CAAT.Math.Point().set(cp0x, cp0y ) );
 61                 this.coordlist.push( new CAAT.Math.Point().set(cp1x, cp1y ) );
 62                 this.coordlist.push( new CAAT.Math.Point().set(cp2x, cp2y ) );
 63                 this.coordlist.push( new CAAT.Math.Point().set(cp3x, cp3y ) );
 64 
 65                 this.cubic= true;
 66                 this.update();
 67 
 68                 return this;
 69             },
 70             /**
 71              * Set this curve as a quadric bezier defined by the three control points.
 72              * @param cp0x {number}
 73              * @param cp0y {number}
 74              * @param cp1x {number}
 75              * @param cp1y {number}
 76              * @param cp2x {number}
 77              * @param cp2y {number}
 78              */
 79             setQuadric : function(cp0x,cp0y, cp1x,cp1y, cp2x,cp2y ) {
 80 
 81                 this.coordlist= [];
 82 
 83                 this.coordlist.push( new CAAT.Math.Point().set(cp0x, cp0y ) );
 84                 this.coordlist.push( new CAAT.Math.Point().set(cp1x, cp1y ) );
 85                 this.coordlist.push( new CAAT.Math.Point().set(cp2x, cp2y ) );
 86 
 87                 this.cubic= false;
 88                 this.update();
 89 
 90                 return this;
 91             },
 92             setPoints : function( points ) {
 93                 if ( points.length===3 ) {
 94                     this.coordlist= points;
 95                     this.cubic= false;
 96                     this.update();
 97                 } else if (points.length===4 ) {
 98                     this.coordlist= points;
 99                     this.cubic= true;
100                     this.update();
101                 } else {
102                     throw 'points must be an array of 3 or 4 CAAT.Point instances.'
103                 }
104 
105                 return this;
106             },
107             /**
108              * Paint this curve.
109              * @param director {CAAT.Director}
110              */
111             paint : function( director ) {
112                 if ( this.cubic ) {
113                     this.paintCubic(director);
114                 } else {
115                     this.paintCuadric( director );
116                 }
117 
118                 CAAT.Math.Bezier.superclass.paint.call(this,director);
119 
120             },
121             /**
122              * Paint this quadric Bezier curve. Each time the curve is drawn it will be solved again from 0 to 1 with
123              * CAAT.Bezier.k increments.
124              *
125              * @param director {CAAT.Director}
126              * @private
127              */
128             paintCuadric : function( director ) {
129                 var x1,y1;
130                 x1 = this.coordlist[0].x;
131                 y1 = this.coordlist[0].y;
132 
133                 var ctx= director.ctx;
134 
135                 ctx.save();
136                 ctx.beginPath();
137                 ctx.moveTo(x1,y1);
138 
139                 var point= new CAAT.Math.Point();
140                 for(var t=this.k;t<=1+this.k;t+=this.k){
141                     this.solve(point,t);
142                     ctx.lineTo(point.x, point.y );
143                 }
144 
145                 ctx.stroke();
146                 ctx.restore();
147 
148             },
149             /**
150              * Paint this cubic Bezier curve. Each time the curve is drawn it will be solved again from 0 to 1 with
151              * CAAT.Bezier.k increments.
152              *
153              * @param director {CAAT.Director}
154              * @private
155              */
156             paintCubic : function( director ) {
157 
158                 var x1,y1;
159                 x1 = this.coordlist[0].x;
160                 y1 = this.coordlist[0].y;
161 
162                 var ctx= director.ctx;
163 
164                 ctx.save();
165                 ctx.beginPath();
166                 ctx.moveTo(x1,y1);
167 
168                 var point= new CAAT.Math.Point();
169                 for(var t=this.k;t<=1+this.k;t+=this.k){
170                     this.solve(point,t);
171                     ctx.lineTo(point.x, point.y );
172                 }
173 
174                 ctx.stroke();
175                 ctx.restore();
176             },
177             /**
178              * Solves the curve for any given parameter t.
179              * @param point {CAAT.Point} the point to store the solved value on the curve.
180              * @param t {number} a number in the range 0..1
181              */
182             solve : function(point,t) {
183                 if ( this.cubic ) {
184                     return this.solveCubic(point,t);
185                 } else {
186                     return this.solveQuadric(point,t);
187                 }
188             },
189             /**
190              * Solves a cubic Bezier.
191              * @param point {CAAT.Point} the point to store the solved value on the curve.
192              * @param t {number} the value to solve the curve for.
193              */
194             solveCubic : function(point,t) {
195 
196                 var t2= t*t;
197                 var t3= t*t2;
198 
199                 var cl= this.coordlist;
200                 var cl0= cl[0];
201                 var cl1= cl[1];
202                 var cl2= cl[2];
203                 var cl3= cl[3];
204 
205                 point.x=(
206                     cl0.x + t * (-cl0.x * 3 + t * (3 * cl0.x-
207                     cl0.x*t)))+t*(3*cl1.x+t*(-6*cl1.x+
208                     cl1.x*3*t))+t2*(cl2.x*3-cl2.x*3*t)+
209                     cl3.x * t3;
210 
211                 point.y=(
212                         cl0.y+t*(-cl0.y*3+t*(3*cl0.y-
213                         cl0.y*t)))+t*(3*cl1.y+t*(-6*cl1.y+
214                         cl1.y*3*t))+t2*(cl2.y*3-cl2.y*3*t)+
215                         cl3.y * t3;
216 
217                 return point;
218             },
219             /**
220              * Solves a quadric Bezier.
221              * @param point {CAAT.Point} the point to store the solved value on the curve.
222              * @param t {number} the value to solve the curve for.
223              */
224             solveQuadric : function(point,t) {
225                 var cl= this.coordlist;
226                 var cl0= cl[0];
227                 var cl1= cl[1];
228                 var cl2= cl[2];
229                 var t1= 1-t;
230 
231                 point.x= t1*t1*cl0.x + 2*t1*t*cl1.x + t*t*cl2.x;
232                 point.y= t1*t1*cl0.y + 2*t1*t*cl1.y + t*t*cl2.y;
233 
234                 return point;
235             }
236         }
237     }
238 });
