 12 (function() {
 13     /**
 14      *
 15      * Curve class is the base for all curve solvers available in CAAT.
 16      *
 17      * @constructor
 18      */
 19 	CAAT.Curve= function() {
 20 		return this;
 21 	};
 22 	
 23 	CAAT.Curve.prototype= {
 24 		coordlist:		null,
 25 		k:				0.05,
 26 		length:			-1,
 27 		interpolator:	false,
 28 		HANDLE_SIZE:	20,
 29 		drawHandles:	true,
 30 
 31         /**
 32          * Paint the curve control points.
 33          * @param director {CAAT.Director}
 34          */
 35 		paint: function(director) {
 36             if ( false===this.drawHandles ) {
 37                 return;
 38             }
 39 
 40             var cl= this.coordlist;
 41 			var ctx= director.ctx;
 42 		
 43 			// control points
 44 			ctx.save();
 45 			ctx.beginPath();
 46 			
 47 			ctx.strokeStyle='#a0a0a0';
 48 			ctx.moveTo( cl[0].x, cl[0].y );
 49 			ctx.lineTo( cl[1].x, cl[1].y );
 50 			ctx.stroke();
 51 			if ( this.cubic ) {
 52 				ctx.moveTo( cl[2].x, cl[2].y );
 53 				ctx.lineTo( cl[3].x, cl[3].y );
 54 				ctx.stroke();
 55 			} 
 56 
 57 
 58             ctx.globalAlpha=0.5;
 59             for( var i=0; i<this.coordlist.length; i++ ) {
 60                 ctx.fillStyle='#7f7f00';
 61                 var w= CAAT.Curve.prototype.HANDLE_SIZE/2;
 62                 ctx.fillRect( cl[i].x-w, cl[i].y-w, w*2, w*2 );
 63             }
 64 
 65 			ctx.restore();
 66 		},
 67         /**
 68          * Signal the curve has been modified and recalculate curve length.
 69          */
 70 		update : function() {
 71 			this.calcLength();
 72 		},
 73         /**
 74          * This method must be overriden by subclasses. It is called whenever the curve must be solved for some time=t.
 75          * The t parameter must be in the range 0..1
 76          * @param point {CAAT.Point} to store curve solution for t.
 77          * @param t {number}
 78          * @return {CAAT.Point} the point parameter.
 79          */
 80 		solve: function(point,t) {
 81 		},
 82         /**
 83          * Get an array of points defining the curve contour.
 84          * @param numSamples {number} number of segments to get.
 85          */
 86         getContour : function(numSamples) {
 87             var contour= [], i;
 88 
 89             for( i=0; i<=numSamples; i++ ) {
 90                 var point= new CAAT.Point();
 91                 this.solve( point, i/numSamples );
 92                 contour.push(point);
 93             }
 94 
 95             return contour;
 96         },
 97         /**
 98          * Calculates a curve bounding box.
 99          *
100          * @param rectangle {CAAT.Rectangle} a rectangle to hold the bounding box.
101          * @return {CAAT.Rectangle} the rectangle parameter.
102          */
103 		getBoundingBox : function(rectangle) {
104 			if ( !rectangle ) {
105 				rectangle= new CAAT.Rectangle();
106 			}
107 
108             // thanks yodesoft.com for spotting the first point is out of the BB
109             rectangle.setEmpty();
110             rectangle.union( this.coordlist[0].x, this.coordlist[0].y );
111 
112 			var pt= new CAAT.Point();
113 			for(var t=this.k;t<=1+this.k;t+=this.k){
114 				this.solve(pt,t);
115 				rectangle.union( pt.x, pt.y );
116 			}			
117 			
118 			return rectangle;
119 		},
120         /**
121          * Calculate the curve length by incrementally solving the curve every substep=CAAT.Curve.k. This value defaults
122          * to .05 so at least 20 iterations will be performed.
123          *
124          * @return {number} the approximate curve length.
125          */
126 		calcLength : function() {
127 			var x1,y1;
128 			x1 = this.coordlist[0].x;
129 			y1 = this.coordlist[0].y;
130 			var llength=0;
131 			var pt= new CAAT.Point();
132 			for(var t=this.k;t<=1+this.k;t+=this.k){
133 				this.solve(pt,t);
134 				llength+= Math.sqrt( (pt.x-x1)*(pt.x-x1) + (pt.y-y1)*(pt.y-y1) );
135 				x1=pt.x;
136 				y1=pt.y;
137 			}
138 			
139 			this.length= llength;
140 			return llength;
141 		},
142         /**
143          * Return the cached curve length.
144          * @return {number} the cached curve length.
145          */
146 		getLength : function() {
147 			return this.length;
148 		},
149         /**
150          * Return the first curve control point.
151          * @return {CAAT.Point}
152          */
153 		endCurvePosition : function() {
154 			return this.coordlist[ this.coordlist.length-1 ];
155 		},
156         /**
157          * Return the last curve control point.
158          * @return {CAAT.Point}
159          */
160 		startCurvePosition : function() {
161 			return this.coordlist[ 0 ];
162 		},
163 
164         setPoints : function( points ) {
165         },
166 
167         setPoint : function( point, index ) {
168             if ( index>=0 && index<this.coordlist.length ) {
169                 this.coordlist[index]= point;
170             }
171         },
172         /**
173          *
174          * @param director <=CAAT.Director>
175          */
176         applyAsPath : function( director ) {
177         }
178 	};
179 })();

184     /**
185      * Bezier quadric and cubic curves implementation.
186      *
187      * @constructor
188      * @extends CAAT.Curve
189      */
190 	CAAT.Bezier= function() {
191 		CAAT.Bezier.superclass.constructor.call(this);
192 		return this;
193 	};
194 	
195 	CAAT.Bezier.prototype= {
196 		
197 		cubic:		false,
198 
199         applyAsPath : function( director ) {
200 
201             var cc= this.coordlist;
202 
203             if ( this.cubic ) {
204                 director.ctx.bezierCurveTo(
205                     cc[1].x,
206                     cc[1].y,
207                     cc[2].x,
208                     cc[2].y,
209                     cc[3].x,
210                     cc[3].y
211                 );
212             } else {
213                 director.ctx.quadraticCurveTo(
214                     cc[1].x,
215                     cc[1].y,
216                     cc[2].x,
217                     cc[2].y
218                 );
219             }
220             return this;
221         },
222         isQuadric : function() {
223             return !this.cubic;
224         },
225         isCubic : function() {
226             return this.cubic;
227         },
228         /**
229          * Set this curve as a cubic bezier defined by the given four control points.
230          * @param cp0x {number}
231          * @param cp0y {number}
232          * @param cp1x {number}
233          * @param cp1y {number}
234          * @param cp2x {number}
235          * @param cp2y {number}
236          * @param cp3x {number}
237          * @param cp3y {number}
238          */
239 		setCubic : function( cp0x,cp0y, cp1x,cp1y, cp2x,cp2y, cp3x,cp3y ) {
240 		
241 			this.coordlist= [];
242 		
243 			this.coordlist.push( new CAAT.Point().set(cp0x, cp0y ) );
244 			this.coordlist.push( new CAAT.Point().set(cp1x, cp1y ) );
245 			this.coordlist.push( new CAAT.Point().set(cp2x, cp2y ) );
246 			this.coordlist.push( new CAAT.Point().set(cp3x, cp3y ) );
247 			
248 			this.cubic= true;
249 			this.update();
250 
251             return this;
252 		},
253         /**
254          * Set this curve as a quadric bezier defined by the three control points.
255          * @param cp0x {number}
256          * @param cp0y {number}
257          * @param cp1x {number}
258          * @param cp1y {number}
259          * @param cp2x {number}
260          * @param cp2y {number}
261          */
262 		setQuadric : function(cp0x,cp0y, cp1x,cp1y, cp2x,cp2y ) {
263 		
264 			this.coordlist= [];
265 		
266 			this.coordlist.push( new CAAT.Point().set(cp0x, cp0y ) );
267 			this.coordlist.push( new CAAT.Point().set(cp1x, cp1y ) );
268 			this.coordlist.push( new CAAT.Point().set(cp2x, cp2y ) );
269 			
270 			this.cubic= false;
271 			this.update();
272 
273             return this;
274 		},
275         setPoints : function( points ) {
276             if ( points.length===3 ) {
277                 this.coordlist= points;
278                 this.cubic= false;
279                 this.update();
280             } else if (points.length===4 ) {
281                 this.coordlist= points;
282                 this.cubic= true;
283                 this.update();
284             } else {
285                 throw 'points must be an array of 3 or 4 CAAT.Point instances.'
286             }
287 
288             return this;
289         },
290         /**
291          * Paint this curve.
292          * @param director {CAAT.Director}
293          */
294 		paint : function( director ) {
295 			if ( this.cubic ) {
296 				this.paintCubic(director);
297 			} else {
298 				this.paintCuadric( director );
299 			}
300 			
301 			CAAT.Bezier.superclass.paint.call(this,director);
302 
303 		},
304         /**
305          * Paint this quadric Bezier curve. Each time the curve is drawn it will be solved again from 0 to 1 with
306          * CAAT.Bezier.k increments.
307          *
308          * @param director {CAAT.Director}
309          * @private
310          */
311 		paintCuadric : function( director ) {
312 			var x1,y1;
313 			x1 = this.coordlist[0].x;
314 			y1 = this.coordlist[0].y;
315 			
316 			var ctx= director.ctx;
317 			
318 			ctx.save();
319 			ctx.beginPath();
320 			ctx.moveTo(x1,y1);
321 			
322 			var point= new CAAT.Point();
323 			for(var t=this.k;t<=1+this.k;t+=this.k){
324 				this.solve(point,t);
325 				ctx.lineTo(point.x, point.y );
326 			}
327 			
328 			ctx.stroke();
329 			ctx.restore();
330 		
331 		},
332         /**
333          * Paint this cubic Bezier curve. Each time the curve is drawn it will be solved again from 0 to 1 with
334          * CAAT.Bezier.k increments.
335          *
336          * @param director {CAAT.Director}
337          * @private
338          */
339 		paintCubic : function( director ) {
340 
341 			var x1,y1;
342 			x1 = this.coordlist[0].x;
343 			y1 = this.coordlist[0].y;
344 			
345 			var ctx= director.ctx;
346 			
347 			ctx.save();
348 			ctx.beginPath();
349 			ctx.moveTo(x1,y1);
350 			
351 			var point= new CAAT.Point();
352 			for(var t=this.k;t<=1+this.k;t+=this.k){
353 				this.solve(point,t);
354 				ctx.lineTo(point.x, point.y );
355 			}
356 			
357 			ctx.stroke();
358 			ctx.restore();
359 		},
360         /**
361          * Solves the curve for any given parameter t.
362          * @param point {CAAT.Point} the point to store the solved value on the curve.
363          * @param t {number} a number in the range 0..1
364          */
365 		solve : function(point,t) {
366 			if ( this.cubic ) {
367 				return this.solveCubic(point,t);
368 			} else {
369 				return this.solveQuadric(point,t);
370 			}
371 		},
372         /**
373          * Solves a cubic Bezier.
374          * @param point {CAAT.Point} the point to store the solved value on the curve.
375          * @param t {number} the value to solve the curve for.
376          */
377 		solveCubic : function(point,t) {
378 			
379 			var t2= t*t;
380 			var t3= t*t2;
381 
382             var cl= this.coordlist;
383             var cl0= cl[0];
384             var cl1= cl[1];
385             var cl2= cl[2];
386             var cl3= cl[3];
387 
388 			point.x=(
389                 cl0.x + t * (-cl0.x * 3 + t * (3 * cl0.x-
390                 cl0.x*t)))+t*(3*cl1.x+t*(-6*cl1.x+
391                 cl1.x*3*t))+t2*(cl2.x*3-cl2.x*3*t)+
392                 cl3.x * t3;
393 				
394 			point.y=(
395                     cl0.y+t*(-cl0.y*3+t*(3*cl0.y-
396 					cl0.y*t)))+t*(3*cl1.y+t*(-6*cl1.y+
397 					cl1.y*3*t))+t2*(cl2.y*3-cl2.y*3*t)+
398 					cl3.y * t3;
399 			
400 			return point;
401 		},
402         /**
403          * Solves a quadric Bezier.
404          * @param point {CAAT.Point} the point to store the solved value on the curve.
405          * @param t {number} the value to solve the curve for.
406          */
407 		solveQuadric : function(point,t) {
408             var cl= this.coordlist;
409             var cl0= cl[0];
410             var cl1= cl[1];
411             var cl2= cl[2];
412             var t1= 1-t;
413 
414 			point.x= t1*t1*cl0.x + 2*t1*t*cl1.x + t*t*cl2.x;
415 			point.y= t1*t1*cl0.y + 2*t1*t*cl1.y + t*t*cl2.y;
416 			
417 			return point;
418 		}
419 	};
420 
421     extend(CAAT.Bezier, CAAT.Curve, null);
422 	
423 })();

427     /**
428      * CatmullRom curves solver implementation.
429      * <p>
430      * This object manages one single catmull rom segment, that is 4 points.
431      * A complete spline should be managed with CAAT.Path.setCatmullRom with a complete list of points.
432      *
433      * @constructor
434      * @extends CAAT.Curve
435      */
436 	CAAT.CatmullRom = function() {
437 		CAAT.CatmullRom.superclass.constructor.call(this);
438 		return this;
439 	};
440 	
441 	CAAT.CatmullRom.prototype= {
442 
443         /**
444          * Set curve control points.
445          * @param p0 <CAAT.Point>
446          * @param p1 <CAAT.Point>
447          * @param p2 <CAAT.Point>
448          * @param p3 <CAAT.Point>
449          */
450 		setCurve : function( p0, p1, p2, p3 ) {
451 
452 			this.coordlist= [];
453             this.coordlist.push( p0 );
454             this.coordlist.push( p1 );
455             this.coordlist.push( p2 );
456             this.coordlist.push( p3 );
457 
458 			this.update();
459 
460             return this;
461 		},
462         /**
463          * Paint the contour by solving again the entire curve.
464          * @param director {CAAT.Director}
465          */
466 		paint: function(director) {
467 			
468 			var x1,y1;
469 
470             // Catmull rom solves from point 1 !!!
471 
472 			x1 = this.coordlist[1].x;
473 			y1 = this.coordlist[1].y;
474 			
475 			var ctx= director.ctx;
476 			
477 			ctx.save();
478 			ctx.beginPath();
479 			ctx.moveTo(x1,y1);
480 			
481 			var point= new CAAT.Point();
482 
483 			for(var t=this.k;t<=1+this.k;t+=this.k){
484 				this.solve(point,t);
485 				ctx.lineTo(point.x,point.y);
486 			}
487 			
488 			ctx.stroke();
489 			ctx.restore();
490 			
491 			CAAT.CatmullRom.superclass.paint.call(this,director);
492 		},
493         /**
494          * Solves the curve for any given parameter t.
495          * @param point {CAAT.Point} the point to store the solved value on the curve.
496          * @param t {number} a number in the range 0..1
497          */
498 		solve: function(point,t) {
499 			var c= this.coordlist;
500 
501             // Handy from CAKE. Thanks.
502             var af = ((-t+2)*t-1)*t*0.5
503             var bf = (((3*t-5)*t)*t+2)*0.5
504             var cf = ((-3*t+4)*t+1)*t*0.5
505             var df = ((t-1)*t*t)*0.5
506 
507             point.x= c[0].x * af + c[1].x * bf + c[2].x * cf + c[3].x * df;
508             point.y= c[0].y * af + c[1].y * bf + c[2].y * cf + c[3].y * df;
509 
510 			return point;
511 
512 		},
513 
514         applyAsPath : function( director ) {
515 
516             var ctx= director.ctx;
517 
518             var point= new CAAT.Point();
519 
520             for(var t=this.k;t<=1+this.k;t+=this.k){
521                 this.solve(point,t);
522                 ctx.lineTo(point.x,point.y);
523             }
524 
525             return this;
526         },
527 
528         /**
529          * Return the first curve control point.
530          * @return {CAAT.Point}
531          */
532 		endCurvePosition : function() {
533 			return this.coordlist[ this.coordlist.length-2 ];
534 		},
535         /**
536          * Return the last curve control point.
537          * @return {CAAT.Point}
538          */
539 		startCurvePosition : function() {
540 			return this.coordlist[ 1 ];
541 		}
542 	};
543 
544     extend(CAAT.CatmullRom, CAAT.Curve, null);
545 })();
