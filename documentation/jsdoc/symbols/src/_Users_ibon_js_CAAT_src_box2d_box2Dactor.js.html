 40     /**
 41      *
 42      * Define a binding beween a box2D Body instance and a CAAT.Actor.
 43      * This function object should not be directly instantiated.
 44      * By now, one Box2D body can contain one single fixture and shape.
 45      *
 46      * @constructor
 47      */
 48     CAAT.B2DBodyActor= function() {
 49         CAAT.B2DBodyActor.superclass.constructor.call(this);
 50         return this;
 51     };
 52 
 53     CAAT.B2DBodyActor.prototype= {
 54 
 55         /**
 56          * Body restitution.
 57          */
 58         restitution:        .5,
 59 
 60         /**
 61          * Body friction.
 62          */
 63         friction:           .5,
 64 
 65         /**
 66          * Body dentisy
 67          */
 68         density:            1,
 69 
 70         /**
 71          * Dynamic bodies by default
 72          */
 73         bodyType:           Box2D.Dynamics.b2Body.b2_dynamicBody,
 74 
 75         /**
 76          * Box2D body
 77          */
 78         worldBody:          null,
 79 
 80         world:              null,
 81 
 82         /**
 83          * Box2d fixture
 84          */
 85         worldBodyFixture:   null,
 86         bodyDef:            null,
 87         fixtureDef:         null,
 88 
 89         /**
 90          * BodyData object linked to the box2D body.
 91          */
 92         bodyData:           null,
 93 
 94         recycle:            false,
 95 
 96         /**
 97          * set this actor to recycle its body, that is, do not destroy it.
 98          */
 99         setRecycle : function() {
100             this.recycle= true;
101             return this;
102         },
103         destroy : function() {
104 
105             CAAT.B2DBodyActor.superclass.destroy.call(this);
106             if ( this.recycle ) {
107                 this.setLocation(-Number.MAX_VALUE, -Number.MAX_VALUE);
108                 this.setAwake(false);
109             } else {
110                 var body= this.worldBody;
111                 body.DestroyFixture( this.worldBodyFixture );
112                 this.world.DestroyBody(body);
113             }
114 
115             return this;
116         },
117         setAwake : function( bool ) {
118             this.worldBody.SetAwake(bool);
119             return this;
120         },
121         setSleepingAllowed : function(bool) {
122             this.worldBody.SetSleepingAllowed(bool);
123             return this;
124         },
125         setLocation : function(x,y) {
126             this.worldBody.SetPosition(
127                 new Box2D.Common.Math.b2Vec2(
128                     (x+this.width/2)/CAAT.PMR,
129                     (y+this.height/2)/CAAT.PMR) );
130             return this;
131         },
132         /**
133          * Set this body's
134          * density.
135          * @param d {number}
136          */
137         setDensity : function(d) {
138             this.density= d;
139             return this;
140         },
141 
142         /**
143          * Set this body's friction.
144          * @param f {number}
145          */
146         setFriction : function(f) {
147             this.friction= f;
148             return this;
149         },
150 
151         /**
152          * Set this body's restitution coeficient.
153          * @param r {number}
154          */
155         setRestitution : function(r) {
156             this.restitution= r;
157             return this;
158         },
159 
160         /**
161          * Set this body's type:
162          * @param bodyType {Box2D.Dynamics.b2Body.b2_*}
163          */
164         setBodyType : function(bodyType) {
165             this.bodyType= bodyType;
166             return this;
167         },
168 
169         /**
170          * Helper method to check whether this js object contains a given property and if it doesn't exist
171          * create and set it to def value.
172          * @param obj {object}
173          * @param prop {string}
174          * @param def {object}
175          */
176         check : function(obj, prop, def) {
177             if ( !obj[prop] ) {
178                 obj[prop]= def;
179             }
180         },
181 
182         /**
183          * Create an actor as a box2D body binding, create it on the given world and with
184          * the initialization data set in bodyData object.
185          * @param world {Box2D.Dynamics.b2World} a Box2D world instance
186          * @param bodyData {object} An object with body info.
187          */
188         createBody : function(world, bodyData) {
189 
190             if ( bodyData ) {
191                 this.check( bodyData, 'density', 1 );
192                 this.check( bodyData, 'friction', .5 );
193                 this.check( bodyData, 'restitution', .2 );
194                 this.check( bodyData, 'bodyType', Box2D.Dynamics.b2Body.b2_staticBody );
195                 this.check( bodyData, 'userData', {} );
196                 this.check( bodyData, 'image', null );
197 
198                 this.density=       bodyData.density;
199                 this.friction=      bodyData.friction;
200                 this.restitution=   bodyData.restitution;
201                 this.bodyType=      bodyData.bodyType;
202                 this.image=         bodyData.image;
203 
204             }
205 
206             this.world= world;
207 
208             return this;
209         },
210 
211         /**
212          * Get this body's center on screen regardless of its shape.
213          * This method will return box2d body's centroid.
214          */
215         getCenter : function() {
216             return {x:0, y:0};
217         },
218 
219         /**
220          * Get a distance joint's position on pixels.
221          */
222         getDistanceJointLocalAnchor : function() {
223             return {x:0, y:0};
224         }
225     };
226 
227     extend( CAAT.B2DBodyActor, CAAT.Actor );

234      *
235      * Create a polygon shaped body in box2D and its CAAT.Actor counterpart.

239     CAAT.B2DPolygonBody= function() {
240         CAAT.B2DPolygonBody.superclass.constructor.call(this);
241         return this;
242     };
243 
244     /**
245      *
246      * Define different types of polygon body types.
247      *
248      * @enum
249      */
250     CAAT.B2DPolygonBody.Type= {
251         EDGE:   'edge',
252         BOX:    'box',
253         POLYGON:'polygon'
254     };
255 
256     CAAT.B2DPolygonBody.prototype= {
257 
258         /**
259          * Measured body's bounding box.
260          */
261         boundingBox: null,
262 
263         /**
264          * Get on-screen distance joint coordinate.
265          */
266         getDistanceJointLocalAnchor : function() {
267             var b= this.worldBody;
268             var xf= b.m_xf;
269             var poly= b.GetFixtureList().GetShape();
270             return poly.m_centroid;
271         },
272 
273         /**
274          * Get this polygon's centroid on screen coordinates.
275          */
276         getCenter : function() {
277             var b= this.worldBody;
278             var xf= b.m_xf;
279             var poly= b.GetFixtureList().GetShape();
280             return Box2D.Common.Math.b2Math.MulX(xf, poly.m_centroid);
281         },
282 
283         /**
284          * Method override to get position and rotation angle from box2d body.
285          * @param director {CAAT.Director}
286          * @param time {number}
287          */
288         animate: function(director, time) {
289 
290             var b= this.worldBody;
291             var xf= b.m_xf;
292 
293             var poly= this.worldBodyFixture.GetShape();
294             if ( poly ) {
295                 var v= Box2D.Common.Math.b2Math.MulX(xf, poly.m_centroid);
296                 //this.setLocation(
297                 CAAT.Actor.prototype.setLocation.call( this,
298                         v.x*CAAT.PMR - this.width/2,
299                         v.y*CAAT.PMR - this.height/2 );
300                 this.setRotation( b.GetAngle() );
301             }
302             
303             return CAAT.B2DPolygonBody.superclass.animate.call(this,director,time);
304         },
305 
306         /**
307          * Create a box2d body and link it to this CAAT.Actor.
308          * @param world {Box2D.Dynamics.b2World}
309          * @param bodyData {object}
310          */
311         createBody : function(world, bodyData) {
312             CAAT.B2DPolygonBody.superclass.createBody.call(this,world,bodyData);
313 
314             var box2D_data= CAAT.B2DPolygonBody.createPolygonBody(world,bodyData);
315 
316             bodyData.userData.actor = this;
317 
318             this.worldBody=         box2D_data.worldBody;
319             this.worldBodyFixture=  box2D_data.worldBodyFixture;
320             this.fixtureDef=        box2D_data.fixDef;
321             this.bodyDef=           box2D_data.bodyDef;
322             this.bodyData=          bodyData;
323             this.boundingBox=       box2D_data.boundingBox;
324 
325             this.setBackgroundImage( bodyData.image ).
326                 setSize(
327                     box2D_data.boundingBox[1].x-box2D_data.boundingBox[0].x+1,
328                     box2D_data.boundingBox[1].y-box2D_data.boundingBox[0].y+1 ).
329                 setFillStyle( box2D_data.worldBodyFixture.IsSensor() ? 'red' : 'green').
330                 setImageTransformation(CAAT.SpriteImage.prototype.TR_FIXED_TO_SIZE);
331 
332             return this;
333         }
334     };

336     /**
337      * Helper function to aid in box2d polygon shaped bodies.
338      * @param world
339      * @param bodyData
340      */
341     CAAT.B2DPolygonBody.createPolygonBody= function(world, bodyData) {
342         var fixDef = new Box2D.Dynamics.b2FixtureDef();
343         fixDef.density = bodyData.density;
344         fixDef.friction = bodyData.friction;
345         fixDef.restitution = bodyData.restitution;
346         fixDef.shape = new Box2D.Collision.Shapes.b2PolygonShape();
347 
348         var minx = Number.MAX_VALUE;
349         var maxx = -Number.MAX_VALUE;
350         var miny = Number.MAX_VALUE;
351         var maxy = -Number.MAX_VALUE;
352 
353         var vec = [];
354 
355         var scale = (bodyData.bodyDefScale || 1);
356         scale = scale + (bodyData.bodyDefScaleTolerance || 0) * Math.random();
357 
358         for (var i = 0; i < bodyData.bodyDef.length; i++) {
359             var x = bodyData.bodyDef[i].x * scale;
360             var y = bodyData.bodyDef[i].y * scale;
361             if (x < minx) {
362                 minx = x;
363             }
364             if (x > maxx) {
365                 maxx = x;
366             }
367             if (y < miny) {
368                 miny = y;
369             }
370             if (y > maxy) {
371                 maxy = y;
372             }
373 
374             x += bodyData.x || 0;
375             y += bodyData.y || 0;
376             vec.push(new Box2D.Common.Math.b2Vec2(x / CAAT.PMR, y / CAAT.PMR));
377         }
378 
379         var boundingBox = [
380             {x:minx, y:miny},
381             {x:maxx, y:maxy}
382         ];
383 
384         var bodyDef = new Box2D.Dynamics.b2BodyDef();
385         bodyDef.type = bodyData.bodyType;
386 
387         if (bodyData.polygonType === CAAT.B2DPolygonBody.Type.EDGE) {
388 
389             fixDef.shape.SetAsEdge(vec[0], vec[1]);
390 
391         } else if (bodyData.polygonType === CAAT.B2DPolygonBody.Type.BOX) {
392 
393             fixDef.shape.SetAsBox(
394                 (maxx - minx) / 2 / CAAT.PMR,
395                 (maxy - miny) / 2 / CAAT.PMR);
396             bodyDef.position.x = ((maxx - minx) / 2 + (bodyData.x || 0)) / CAAT.PMR;
397             bodyDef.position.y = ((maxy - miny) / 2 + (bodyData.y || 0)) / CAAT.PMR;
398 
399         } else if (bodyData.polygonType === CAAT.B2DPolygonBody.Type.POLYGON ) {
400 
401             fixDef.shape.SetAsArray(vec, vec.length);
402 
403         } else {
404             throw 'Unkown bodyData polygonType: '+bodyData.polygonType;
405         }
406 
407         // link entre cuerpo fisico box2d y caat.
408         fixDef.userData = bodyData.userData;
409         bodyDef.userData = bodyData.userData;
410 
411         var worldBody = world.CreateBody(bodyDef);
412         var worldBodyFixture = worldBody.CreateFixture(fixDef);
413 
414 
415         if (bodyData.isSensor) {
416             worldBodyFixture.SetSensor(true);
417         }
418 
419         return {
420             worldBody:          worldBody,
421             worldBodyFixture:   worldBodyFixture,
422             fixDef:             fixDef,
423             bodyDef:            bodyDef,
424             boundingBox:        boundingBox
425         };
426     };
427 
428     extend( CAAT.B2DPolygonBody, CAAT.B2DBodyActor );

444     CAAT.B2DCircularBody.prototype= {
445 
446         /**
447          * Default radius.
448          */
449         radius: 1,
450 
451         /**
452          * Get a given dinstance joint's local anchor position.
453          * For a circular body, it is its center.
454          */
455         getDistanceJointLocalAnchor : function() {
456             return new Box2D.Common.Math.b2Vec2(0,0);
457         },
458 
459         /**
460          * Get this box2d body's center position on screen.
461          */
462         getCenter : function() {
463             return this.worldBody.m_xf.position;
464         },
465 
466         /**
467          * Method override to get position and rotation angle from box2d body.
468          * @param director {CAAT.Director}
469          * @param time {number}
470          */
471         animate: function(director, time) {
472 
473             var b= this.worldBody;
474             var xf= b.m_xf;
475             //this.setLocation(
476             CAAT.Actor.prototype.setLocation.call( this,
477                     CAAT.PMR*xf.position.x - this.width/2,
478                     CAAT.PMR*xf.position.y - this.height/2 );
479             this.setRotation( b.GetAngle() );
480 
481             return CAAT.B2DCircularBody.superclass.animate.call(this,director,time);
482         },
483 
484         /**
485          * Create a box2d body and link it to this CAAT.Actor instance.
486          * @param world {Box2D.Dynamics.b2World} a Box2D world instance
487          * @param bodyData {object}
488          */
489         createBody : function(world, bodyData) {
490 
491             var scale= (bodyData.radius || 1);
492             scale= scale+ (bodyData.bodyDefScaleTolerance || 0)*Math.random();
493             bodyData.radius= scale;
494 
495             CAAT.B2DCircularBody.superclass.createBody.call(this,world,bodyData);
496 
497             if ( bodyData.radius )  this.radius= bodyData.radius;
498 
499             var fixDef=             new Box2D.Dynamics.b2FixtureDef();
500             fixDef.density=         this.density;
501             fixDef.friction=        this.friction;
502             fixDef.restitution=     this.restitution;
503             fixDef.shape =          new Box2D.Collision.Shapes.b2CircleShape(this.radius/CAAT.PMR);
504 
505             var bodyDef =           new Box2D.Dynamics.b2BodyDef();
506             bodyDef.type =          this.bodyType;
507             bodyDef.position.x=     bodyData.x/CAAT.PMR;
508             bodyDef.position.y=     bodyData.y/CAAT.PMR;
509 
510             // link entre cuerpo fisico box2d y caat.
511             bodyData.userData.actor=         this;
512             fixDef.userData=        bodyData.userData;
513             bodyDef.userData=       bodyData.userData;
514 
515             var worldBody=          world.CreateBody(bodyDef);
516             var worldBodyFixture=   worldBody.CreateFixture(fixDef);
517 
518             if ( bodyData.isSensor ) {
519                 worldBodyFixture.SetSensor(true);
520             }
521 
522             this.worldBody=         worldBody;
523             this.worldBodyFixture=  worldBodyFixture;
524             this.fixtureDef=        fixDef;
525             this.bodyDef=           bodyDef;
526             this.bodyData=          bodyData;
527 
528 
529             this.setFillStyle(this.worldBodyFixture.IsSensor() ? 'red' : 'blue').
530                     setBackgroundImage(this.image).
531                     setSize(2*this.radius,2*this.radius).
532                     setImageTransformation(CAAT.SpriteImage.prototype.TR_FIXED_TO_SIZE);
533 
534 
535             return this;
536         }
537     };
538 
539     extend( CAAT.B2DCircularBody, CAAT.B2DBodyActor );
