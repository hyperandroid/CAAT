 35     Class.extend = function (extendingProt, constants, name, aliases, flags) {
 36 
 37         var _super = this.prototype;
 38 
 39         // Instantiate a base class (but only create the instance,
 40         // don't run the init constructor)
 41         initializing = true;
 42         var prototype = new this();
 43         initializing = false;
 44 
 45         // The dummy class constructor
 46         function CAATClass() {
 47             // All construction is actually done in the init method
 48             if (!initializing && this.__init) {
 49                 this.__init.apply(this, arguments);
 50             }
 51         }
 52 
 53         // Populate our constructed prototype object
 54         CAATClass.prototype = prototype;
 55         // Enforce the constructor to be what we expect
 56         CAATClass.prototype.constructor = CAATClass;
 57         CAATClass.superclass = _super;
 58         // And make this class extendable
 59         CAATClass.extend = Class.extend;
 60 
 61         assignNamespace( name, CAATClass );
 62         if ( constants ) {
 63             constants= (isFunction(constants) ? constants() : constants);
 64             for( var constant in constants ) {
 65                 if ( constants.hasOwnProperty(constant) ) {
 66                     CAATClass[ constant ]= constants[constant];
 67                 }
 68             }
 69         }
 70 
 71         if ( aliases ) {
 72             if ( !isArray(aliases) ) {
 73                 aliases= [aliases];
 74             }
 75             for( var i=0; i<aliases.length; i++ ) {
 76                 ensureNamespace( aliases[i] );
 77                 var ns= assignNamespace( aliases[i], CAATClass );
 78 
 79                 // assign constants to alias classes.
 80                 if ( constants ) {
 81                     for( var constant in constants ) {
 82                         if ( constants.hasOwnProperty(constant) ) {
 83                             ns[ constant ]= constants[constant];
 84                         }
 85                     }
 86                 }
 87             }
 88         }
 89 
 90         extendingProt= (isFunction(extendingProt) ? extendingProt() : extendingProt);
 91 
 92         // Copy the properties over onto the new prototype
 93         for (var fname in extendingProt) {
 94             // Check if we're overwriting an existing function
 95             prototype[fname] = ( (fname === "__init" || (flags && flags.decorated) ) && isFunction(extendingProt[fname]) && isFunction(_super[fname]) ) ?
 96                 (function (name, fn) {
 97                     return function () {
 98                         var tmp = this.__super;
 99                         this.__super = _super[name];
100                         var ret = fn.apply(this, arguments);
101                         this.__super = tmp;
102                         return ret;
103                     };
104                 })(fname, extendingProt[fname]) :
105 
106                 extendingProt[fname];
107         }
108 
109         return CAATClass;
