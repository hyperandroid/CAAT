 15         /**
 16          * Helper function to aid in box2d polygon shaped bodies.
 17          * @param world
 18          * @param bodyData
 19          */
 20         createPolygonBody : function(world, bodyData) {
 21             var fixDef = new Box2D.Dynamics.b2FixtureDef();
 22             fixDef.density = bodyData.density;
 23             fixDef.friction = bodyData.friction;
 24             fixDef.restitution = bodyData.restitution;
 25             fixDef.shape = new Box2D.Collision.Shapes.b2PolygonShape();
 26 
 27             var minx = Number.MAX_VALUE;
 28             var maxx = -Number.MAX_VALUE;
 29             var miny = Number.MAX_VALUE;
 30             var maxy = -Number.MAX_VALUE;
 31 
 32             var vec = [];
 33 
 34             var scale = (bodyData.bodyDefScale || 1);
 35             scale = scale + (bodyData.bodyDefScaleTolerance || 0) * Math.random();
 36 
 37             for (var i = 0; i < bodyData.bodyDef.length; i++) {
 38                 var x = bodyData.bodyDef[i].x * scale;
 39                 var y = bodyData.bodyDef[i].y * scale;
 40                 if (x < minx) {
 41                     minx = x;
 42                 }
 43                 if (x > maxx) {
 44                     maxx = x;
 45                 }
 46                 if (y < miny) {
 47                     miny = y;
 48                 }
 49                 if (y > maxy) {
 50                     maxy = y;
 51                 }
 52 /*
 53                 x += bodyData.x || 0;
 54                 y += bodyData.y || 0;
 55                 */
 56                 vec.push(new Box2D.Common.Math.b2Vec2(x / CAAT.PMR, y / CAAT.PMR));
 57             }
 58 
 59             var boundingBox = [
 60                 {x:minx, y:miny},
 61                 {x:maxx, y:maxy}
 62             ];
 63 
 64             var bodyDef = new Box2D.Dynamics.b2BodyDef();
 65             bodyDef.type = bodyData.bodyType;
 66             bodyDef.position.Set(
 67                 ((maxx - minx) / 2 + (bodyData.x || 0)) / CAAT.PMR,
 68                 ((maxy - miny) / 2 + (bodyData.y || 0)) / CAAT.PMR );
 69 
 70             if (bodyData.polygonType === CAAT.Foundation.Box2D.B2DPolygonBody.TYPE.EDGE) {
 71 
 72                 var v0= new Box2D.Common.Math.b2Vec2(vec[0].x, vec[0].y );
 73                 var v1= new Box2D.Common.Math.b2Vec2(vec[1].x-vec[0].x, vec[1].y-vec[0].y );
 74 
 75                 bodyDef.position.Set(v0.x, v0.y);
 76                 fixDef.shape.SetAsEdge(new Box2D.Common.Math.b2Vec2(0,0), v1);
 77 
 78 
 79             } else if (bodyData.polygonType === CAAT.Foundation.Box2D.B2DPolygonBody.TYPE.BOX) {
 80 
 81                 fixDef.shape.SetAsBox(
 82                     (maxx - minx) / 2 / CAAT.PMR,
 83                     (maxy - miny) / 2 / CAAT.PMR);
 84 
 85             } else if (bodyData.polygonType === CAAT.Foundation.Box2D.B2DPolygonBody.TYPE.POLYGON ) {
 86 
 87                 fixDef.shape.SetAsArray(vec, vec.length);
 88 
 89             } else {
 90                 throw 'Unkown bodyData polygonType: '+bodyData.polygonType;
 91             }
 92 
 93             // link entre cuerpo fisico box2d y caat.
 94             fixDef.userData = bodyData.userData;
 95             bodyDef.userData = bodyData.userData;
 96 
 97             var worldBody = world.CreateBody(bodyDef);
 98             var worldBodyFixture = worldBody.CreateFixture(fixDef);
 99 
100 
101             if (bodyData.isSensor) {
102                 worldBodyFixture.SetSensor(true);
103             }
104 
105             return {
106                 worldBody:          worldBody,
107                 worldBodyFixture:   worldBodyFixture,
108                 fixDef:             fixDef,
109                 bodyDef:            bodyDef,
110                 boundingBox:        boundingBox
111             };
112         }
