  7 CAAT.Module( {
  8     defines:        "CAAT.Math.Rectangle",
  9     aliases:        ["CAAT.Rectangle"],
 10     extendsWith:    function() {
 11         return {
 12 
 13             __init : function( x,y,w,h ) {
 14                 this.setLocation(x,y);
 15                 this.setDimension(w,h);
 16             },
 17 
 18             x:		0,
 19             y:		0,
 20             x1:		0,
 21             y1:		0,
 22             width:	-1,
 23             height:	-1,
 24 
 25             setEmpty : function() {
 26                 this.width=     -1;
 27                 this.height=    -1;
 28                 this.x=         0;
 29                 this.y=         0;
 30                 this.x1=        0;
 31                 this.y1=        0;
 32                 return this;
 33             },
 34             /**
 35              * Set this rectangle's location.
 36              * @param x {number}
 37              * @param y {number}
 38              */
 39             setLocation: function( x,y ) {
 40                 this.x= x;
 41                 this.y= y;
 42                 this.x1= this.x+this.width;
 43                 this.y1= this.y+this.height;
 44                 return this;
 45             },
 46             /**
 47              * Set this rectangle's dimension.
 48              * @param w {number}
 49              * @param h {number}
 50              */
 51             setDimension : function( w,h ) {
 52                 this.width= w;
 53                 this.height= h;
 54                 this.x1= this.x+this.width;
 55                 this.y1= this.y+this.height;
 56                 return this;
 57             },
 58             setBounds : function( x,y,w,h ) {
 59                 this.setLocation( x, y );
 60                 this.setDimension( w, h );
 61                 return this;
 62             },
 63             /**
 64              * Return whether the coordinate is inside this rectangle.
 65              * @param px {number}
 66              * @param py {number}
 67              *
 68              * @return {boolean}
 69              */
 70             contains : function(px,py) {
 71                 //return px>=0 && px<this.width && py>=0 && py<this.height;
 72                 return px>=this.x && px<this.x1 && py>=this.y && py<this.y1;
 73             },
 74             /**
 75              * Return whether this rectangle is empty, that is, has zero dimension.
 76              * @return {boolean}
 77              */
 78             isEmpty : function() {
 79                 return this.width===-1 && this.height===-1;
 80             },
 81             /**
 82              * Set this rectangle as the union of this rectangle and the given point.
 83              * @param px {number}
 84              * @param py {number}
 85              */
 86             union : function(px,py) {
 87 
 88                 if ( this.isEmpty() ) {
 89                     this.x= px;
 90                     this.x1= px;
 91                     this.y= py;
 92                     this.y1= py;
 93                     this.width=0;
 94                     this.height=0;
 95                     return;
 96                 }
 97 
 98                 this.x1= this.x+this.width;
 99                 this.y1= this.y+this.height;
100 
101                 if ( py<this.y ) {
102                     this.y= py;
103                 }
104                 if ( px<this.x ) {
105                     this.x= px;
106                 }
107                 if ( py>this.y1 ) {
108                     this.y1= py;
109                 }
110                 if ( px>this.x1 ){
111                     this.x1= px;
112                 }
113 
114                 this.width= this.x1-this.x;
115                 this.height= this.y1-this.y;
116             },
117             unionRectangle : function( rectangle ) {
118                 this.union( rectangle.x , rectangle.y  );
119                 this.union( rectangle.x1, rectangle.y  );
120                 this.union( rectangle.x,  rectangle.y1 );
121                 this.union( rectangle.x1, rectangle.y1 );
122                 return this;
123             },
124             intersects : function( r ) {
125                 if ( r.isEmpty() || this.isEmpty() ) {
126                     return false;
127                 }
128 
129                 if ( r.x1<= this.x ) {
130                     return false;
131                 }
132                 if ( r.x >= this.x1 ) {
133                     return false;
134                 }
135                 if ( r.y1<= this.y ) {
136                     return false;
137                 }
138 
139                 return r.y < this.y1;
140             },
141 
142             intersectsRect : function( x,y,w,h ) {
143                 if ( -1===w || -1===h ) {
144                     return false;
145                 }
146 
147                 var x1= x+w-1;
148                 var y1= y+h-1;
149 
150                 if ( x1< this.x ) {
151                     return false;
152                 }
153                 if ( x > this.x1 ) {
154                     return false;
155                 }
156                 if ( y1< this.y ) {
157                     return false;
158                 }
159                 return y <= this.y1;
160 
161             },
162 
163             intersect : function( i, r ) {
164                 if ( typeof r==='undefined' ) {
165                     r= new CAAT.Math.Rectangle();
166                 }
167 
168                 r.x= Math.max( this.x, i.x );
169                 r.y= Math.max( this.y, i.y );
170                 r.x1=Math.min( this.x1, i.x1 );
171                 r.y1=Math.min( this.y1, i.y1 );
172                 r.width= r.x1-r.x;
173                 r.height=r.y1-r.y;
174 
175                 return r;
176             }
177         }
178 	}
179 });
