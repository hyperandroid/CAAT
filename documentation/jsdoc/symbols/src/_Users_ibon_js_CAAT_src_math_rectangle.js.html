1 /**

  3  *
  4  * Rectangle Class.
  5  * Needed to compute Curve bounding box.
  6  * Needed to compute Actor affected area on change.
  7  *
  8  **/
  9 
 10 
 11 (function() {
 12     /**
 13      * A Rectangle implementation, which defines an area positioned somewhere.
 14      *
 15      * @constructor
 16      */
 17 	CAAT.Rectangle= function() {
 18 		return this;
 19 	};
 20 	
 21 	CAAT.Rectangle.prototype= {
 22 		x:		0,
 23 		y:		0,
 24 		x1:		0,
 25 		y1:		0,
 26 		width:	-1,
 27 		height:	-1,
 28 
 29         setEmpty : function() {
 30             this.width=     -1;
 31             this.height=    -1;
 32             this.x=         0;
 33             this.y=         0;
 34             this.x1=        0;
 35             this.y1=        0;
 36             return this;
 37         },
 38         /**
 39          * Set this rectangle's location.
 40          * @param x {number}
 41          * @param y {number}
 42          */
 43         setLocation: function( x,y ) {
 44             this.x= x;
 45             this.y= y;
 46             this.x1= this.x+this.width;
 47             this.y1= this.y+this.height;
 48             return this;
 49         },
 50         /**
 51          * Set this rectangle's dimension.
 52          * @param w {number}
 53          * @param h {number}
 54          */
 55         setDimension : function( w,h ) {
 56             this.width= w;
 57             this.height= h;
 58             this.x1= this.x+this.width;
 59             this.y1= this.y+this.height;
 60             return this;
 61         },
 62         setBounds : function( x,y,w,h ) {
 63             this.setLocation( x, y )
 64             this.setDimension( w, h );
 65             return this;
 66         },
 67         /**
 68          * Return whether the coordinate is inside this rectangle.
 69          * @param px {number}
 70          * @param py {number}
 71          *
 72          * @return {boolean}
 73          */
 74 		contains : function(px,py) {
 75 			return px>=0 && px<this.width && py>=0 && py<this.height; 
 76 		},
 77         /**
 78          * Return whether this rectangle is empty, that is, has zero dimension.
 79          * @return {boolean}
 80          */
 81 		isEmpty : function() {
 82 			return this.width===-1 && this.height===-1;
 83 		},
 84         /**
 85          * Set this rectangle as the union of this rectangle and the given point.
 86          * @param px {number}
 87          * @param py {number}
 88          */
 89 		union : function(px,py) {
 90 			
 91 			if ( this.isEmpty() ) {
 92 				this.x= px;
 93                 this.x1= px;
 94 				this.y= py;
 95                 this.y1= py;
 96                 this.width=0;
 97                 this.height=0;
 98 				return;
 99 			}
100 			
101 			this.x1= this.x+this.width;
102 			this.y1= this.y+this.height;
103 			
104 			if ( py<this.y ) {
105 				this.y= py;
106 			}
107 			if ( px<this.x ) {
108 				this.x= px;
109 			}
110 			if ( py>this.y1 ) {
111 				this.y1= py;
112 			}
113 			if ( px>this.x1 ){
114 				this.x1= px;
115 			}
116 			
117 			this.width= this.x1-this.x;
118 			this.height= this.y1-this.y;
119 		},
120         unionRectangle : function( rectangle ) {
121             this.union( rectangle.x , rectangle.y  );
122             this.union( rectangle.x1, rectangle.y  );
123             this.union( rectangle.x,  rectangle.y1 );
124             this.union( rectangle.x1, rectangle.y1 );
125             return this;
126         },
127         intersects : function( r ) {
128             if ( r.isEmpty() || this.isEmpty() ) {
129                 return false;
130             }
131 
132             if ( r.x1< this.x ) {
133                 return false;
134             }
135             if ( r.x > this.x1 ) {
136                 return false;
137             }
138             if ( r.y1< this.y ) {
139                 return false;
140             }
141             if ( r.y> this.y1 ) {
142                 return false;
143             }
144 
145             return true;
146         },
147 
148         intersectsRect : function( x,y,w,h ) {
149             if ( -1===w || -1===h ) {
150                 return false;
151             }
152 
153             var x1= x+w-1;
154             var y1= y+h-1;
155 
156             if ( x1< this.x ) {
157                 return false;
158             }
159             if ( x > this.x1 ) {
160                 return false;
161             }
162             if ( y1< this.y ) {
163                 return false;
164             }
165             if ( y> this.y1 ) {
166                 return false;
167             }
168 
169             return true;
170         },
171 
172         intersect : function( i, r ) {
173             if ( typeof r==='undefined' ) {
174                 r= new CAAT.Rectangle();
175             }
176 
177             r.x= Math.max( this.x, i.x );
178             r.y= Math.max( this.y, i.y );
179             r.x1=Math.min( this.x1, i.x1 );
180             r.y1=Math.min( this.y1, i.y1 );
181             r.width= r.x1-r.x;
182             r.height=r.y1-r.y;
183 
184             return r;
185         }
186 	};
187 })();
