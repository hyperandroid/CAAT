  6 CAAT.Module( {
  7     defines : "CAAT.Foundation.UI.InterpolatorActor",
  8     aliases : ["CAAT.InterpolatorActor"],
  9     depends : [
 10         "CAAT.Foundation.Actor"
 11     ],
 12     extendsClass : "CAAT.Foundation.Actor",
 13     extendsWith : {
 14 
 15         interpolator:   null,   // CAAT.Interpolator instance.
 16         contour:        null,   // interpolator contour cache
 17         S:              50,     // contour samples.
 18         gap:            5,      // border size in pixels.
 19 
 20         /**
 21          * Sets a padding border size. By default is 5 pixels.
 22          * @param gap {number} border size in pixels.
 23          * @return this
 24          */
 25         setGap : function( gap ) {
 26             this.gap= gap;
 27             return this;
 28         },
 29         /**
 30          * Sets the CAAT.Interpolator instance to draw.
 31          *
 32          * @param interpolator a CAAT.Interpolator instance.
 33          * @param size an integer indicating the number of polyline segments so draw to show the CAAT.Interpolator
 34          * instance.
 35          *
 36          * @return this
 37          */
 38         setInterpolator : function( interpolator, size ) {
 39             this.interpolator= interpolator;
 40             this.contour= interpolator.getContour(size || this.S);
 41 
 42             return this;
 43         },
 44         /**
 45          * Paint this actor.
 46          * @param director {CAAT.Director}
 47          * @param time {number} scene time.
 48          */
 49         paint : function( director, time ) {
 50 
 51             CAAT.InterpolatorActor.superclass.paint.call(this,director,time);
 52 
 53             if ( this.backgroundImage ) {
 54                 return this;
 55             }
 56 
 57             if ( this.interpolator ) {
 58 
 59                 var canvas= director.ctx;
 60 
 61                 var xs= (this.width-2*this.gap);
 62                 var ys= (this.height-2*this.gap);
 63 
 64                 canvas.beginPath();
 65                 canvas.moveTo(
 66                         this.gap +  xs*this.contour[0].x,
 67                         -this.gap + this.height - ys*this.contour[0].y);
 68 
 69                 for( var i=1; i<this.contour.length; i++ ) {
 70                     canvas.lineTo(
 71                              this.gap + xs*this.contour[i].x,
 72                             -this.gap + this.height - ys*this.contour[i].y);
 73                 }
 74 
 75                 canvas.strokeStyle= this.strokeStyle;
 76                 canvas.stroke();
 77             }
 78         },
 79         /**
 80          * Return the represented interpolator.
 81          * @return {CAAT.Interpolator}
 82          */
 83         getInterpolator : function() {
 84             return this.interpolator;
 85         }
 86     }
 87 
 88 
 89 });
