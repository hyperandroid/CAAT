1 /**

  3  *
  4  * This file contains the definition for objects QuadTree and HashMap.
  5  * Quadtree offers an exact list of collisioning areas, while HashMap offers a list of potentially colliding elements.
  6  *
  7  **/
  8 
  9 CAAT.Module({
 10 
 11     defines:"CAAT.Module.Collision.QuadTree",
 12     depends:[
 13         "CAAT.Math.Rectangle"
 14     ],
 15     extendsClass:"CAAT.Math.Rectangle",
 16     extendsWith:function () {
 17 
 18         var QT_MAX_ELEMENTS = 1;
 19         var QT_MIN_WIDTH = 32;
 20 
 21         return {
 22 
 23             bgActors:null,
 24 
 25             quadData:null,
 26 
 27             create:function (l, t, r, b, backgroundElements, minWidth, maxElements) {
 28 
 29                 if (typeof minWidth === 'undefined') {
 30                     minWidth = QT_MIN_WIDTH;
 31                 }
 32                 if (typeof maxElements === 'undefined') {
 33                     maxElements = QT_MAX_ELEMENTS;
 34                 }
 35 
 36                 var cx = (l + r) / 2;
 37                 var cy = (t + b) / 2;
 38 
 39                 this.x = l;
 40                 this.y = t;
 41                 this.x1 = r;
 42                 this.y1 = b;
 43                 this.width = r - l;
 44                 this.height = b - t;
 45 
 46                 this.bgActors = this.__getOverlappingActorList(backgroundElements);
 47 
 48                 if (this.bgActors.length <= maxElements || this.width <= minWidth) {
 49                     return this;
 50                 }
 51 
 52                 this.quadData = new Array(4);
 53                 this.quadData[0] = new CAAT.Module.Collision.QuadTree().create(l, t, cx, cy, this.bgActors);  // TL
 54                 this.quadData[1] = new CAAT.Module.Collision.QuadTree().create(cx, t, r, cy, this.bgActors);  // TR
 55                 this.quadData[2] = new CAAT.Module.Collision.QuadTree().create(l, cy, cx, b, this.bgActors);  // BL
 56                 this.quadData[3] = new CAAT.Module.Collision.QuadTree().create(cx, cy, r, b, this.bgActors);
 57 
 58                 return this;
 59             },
 60 
 61             __getOverlappingActorList:function (actorList) {
 62                 var tmpList = [];
 63                 for (var i = 0, l = actorList.length; i < l; i++) {
 64                     var actor = actorList[i];
 65                     if (this.intersects(actor.AABB)) {
 66                         tmpList.push(actor);
 67                     }
 68                 }
 69                 return tmpList;
 70             },
 71 
 72             getOverlappingActors:function (rectangle) {
 73                 var i, j, l;
 74                 var overlappingActors = [];
 75                 var qoverlappingActors;
 76                 var actors = this.bgActors;
 77                 var actor;
 78 
 79                 if (this.quadData) {
 80                     for (i = 0; i < 4; i++) {
 81                         if (this.quadData[i].intersects(rectangle)) {
 82                             qoverlappingActors = this.quadData[i].getOverlappingActors(rectangle);
 83                             for (j = 0, l = qoverlappingActors.length; j < l; j++) {
 84                                 overlappingActors.push(qoverlappingActors[j]);
 85                             }
 86                         }
 87                     }
 88                 } else {
 89                     for (i = 0, l = actors.length; i < l; i++) {
 90                         actor = actors[i];
 91                         if (rectangle.intersects(actor.AABB)) {
 92                             overlappingActors.push(actor);
 93                         }
 94                     }
 95                 }
 96 
 97                 return overlappingActors;
 98             }
 99         }
100     }
101 });
