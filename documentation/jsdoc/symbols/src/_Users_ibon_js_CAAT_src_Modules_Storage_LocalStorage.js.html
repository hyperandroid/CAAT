  5 CAAT.Module({
  6     defines : "CAAT.Module.Storage.LocalStorage",
  7     constants : {
  8 
  9         /**
 10          * Stores an object in local storage. The data will be saved as JSON.stringify.
 11          * @param key {string} key to store data under.
 12          * @param data {object} an object.
 13          * @return this
 14          *
 15          * @static
 16          */
 17         save : function( key, data ) {
 18             try {
 19                 localStorage.setItem( key, JSON.stringify(data) );
 20             } catch(e) {
 21                 // eat it
 22             }
 23             return this;
 24         },
 25         /**
 26          * Retrieve a value from local storage.
 27          * @param key {string} the key to retrieve.
 28          * @return {object} object stored under the key parameter.
 29          *
 30          * @static
 31          */
 32         load : function( key, defValue ) {
 33             try {
 34                 var v= localStorage.getItem( key );
 35 
 36                 return null===v ? defValue : JSON.parse(v);
 37             } catch(e) {
 38                 return null;
 39             }
 40         },
 41 
 42         /**
 43          * Removes a value stored in local storage.
 44          * @param key {string}
 45          * @return this
 46          *
 47          * @static
 48          */
 49         remove : function( key ) {
 50             try {
 51                 localStorage.removeItem(key);
 52             } catch(e) {
 53                 // eat it
 54             }
 55             return this;
 56         }
 57     },
 58     extendsWith : {
 59 
 60     }
 61 
 62 });
