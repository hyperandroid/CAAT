1 /**
  2  * See LICENSE file.
  3  *
  4  * This class generates an in-memory image with the representation of a drawn list of characters.
  5  *
  6  **/
  7 (function() {
  8 
  9     /**
 10      * @constructor
 11      */
 12     CAAT.Font= function( ) {
 13         return this;
 14     };
 15 
 16     var UNKNOWN_CHAR_WIDTH= 10;
 17 
 18     CAAT.Font.prototype= {
 19 
 20         fontSize    :   10,
 21         fontSizeUnit:   "px",
 22         font        :   'Sans-Serif',
 23         fontStyle   :   '',
 24         fillStyle   :   '#fff',
 25         strokeStyle :   null,
 26         padding     :   0,
 27         image       :   null,
 28         charMap     :   null,
 29 
 30         height      :   0,
 31 
 32         setPadding : function( padding ) {
 33             this.padding= padding;
 34             return this;
 35         },
 36 
 37         setFontStyle : function( style ) {
 38             this.fontStyle= style;
 39             return this;
 40         },
 41 
 42         setFontSize : function( fontSize ) {
 43             this.fontSize=      fontSize;
 44             this.fontSizeUnit=  'px';
 45             return this;
 46         },
 47 
 48         setFont : function( font ) {
 49             this.font= font;
 50             return this;
 51         },
 52 
 53         setFillStyle : function( style ) {
 54             this.fillStyle= style;
 55             return this;
 56         },
 57 
 58         setStrokeStyle : function( style ) {
 59             this.strokeStyle= style;
 60             return this;
 61         },
 62 
 63         createDefault : function( padding ) {
 64             var str="";
 65             for( var i=32; i<128; i++ ) {
 66                 str= str+String.fromCharCode(i);
 67             }
 68 
 69             return this.create( str, padding );
 70         },
 71 
 72         create : function( chars, padding ) {
 73 
 74             padding= padding | 0;
 75             this.padding= padding;
 76 
 77             var canvas= document.createElement('canvas');
 78             canvas.width=   1;
 79             canvas.height=  1;
 80             var ctx= canvas.getContext('2d');
 81 
 82             ctx.textBaseline= 'top';
 83             ctx.font= this.fontStyle+' '+this.fontSize+""+this.fontSizeUnit+" "+ this.font;
 84 
 85             var textWidth= 0;
 86             var charWidth= [];
 87             var i;
 88             var x;
 89             var cchar;
 90 
 91             for( i=0; i<chars.length; i++ ) {
 92                 var cw= Math.max( 1, (ctx.measureText( chars.charAt(i) ).width>>0)+1 ) + 2 * padding ;
 93                 charWidth.push(cw);
 94                 textWidth+= cw;
 95             }
 96 
 97             canvas.width= textWidth;
 98             canvas.height= (this.fontSize*1.5)>>0;
 99             ctx= canvas.getContext('2d');
100 
101             ctx.textBaseline= 'top';
102             ctx.font= this.fontStyle+' '+this.fontSize+""+this.fontSizeUnit+" "+ this.font;
103             ctx.fillStyle= this.fillStyle;
104             ctx.strokeStyle= this.strokeStyle;
105 
106             this.charMap= {};
107 
108             x=0;
109             for( i=0; i<chars.length; i++ ) {
110                 cchar= chars.charAt(i);
111                 ctx.fillText( cchar, x+padding, 0 );
112                 if ( this.strokeStyle ) {
113                     ctx.beginPath();
114                     ctx.strokeText( cchar, x+padding,  0 );
115                 }
116                 this.charMap[cchar]= {
117                     x:      x,
118                     width:  charWidth[i]
119                 };
120                 x+= charWidth[i];
121             }
122 
123             this.image= CAAT.modules.ImageUtil.optimize( canvas, 32, { top: true, bottom: true, left: false, right: false } );
124             this.height= this.image.height;
125 
126             return this;
127         },
128 
129         setAsSpriteImage : function() {
130             var cm= [];
131             var _index= 0;
132             for( var i in this.charMap ) {
133                 var _char= i;
134                 var charData= this.charMap[i];
135 
136                 cm[i]={
137                     id: _index++,
138                     height: this.height,
139                     xoffset: 0,
140                     letter: _char,
141                     yoffset: 0,
142                     width: charData.width,
143                     xadvance: charData.width,
144                     x: charData.x,
145                     y: 0
146                 };
147             }
148             return new CAAT.SpriteImage().initializeAsGlyphDesigner( this.image, cm );
149         },
150 
151         stringWidth : function( str ) {
152             var i, l,  w=0, c;
153 
154             for( i=0, l=str.length; i<l; i++ ) {
155                 c= this.charMap[ str.charAt(i) ];
156                 if ( c ) {
157                     w+= c.width;
158                 } else {
159                     w+= UNKNOWN_CHAR_WIDTH;
160                 }
161             }
162 
163             return w;
164         },
165 
166         drawText : function( str, ctx, x, y ) {
167             var i,l,charInfo,w;
168             var height= this.image.height;
169 
170             for( i=0, l=str.length; i<l; i++ ) {
171                 charInfo= this.charMap[ str.charAt(i) ];
172                 if ( charInfo ) {
173                     w= charInfo.width;
174                     ctx.drawImage(
175                         this.image,
176                         charInfo.x, 0,
177                         w, height,
178                         x, y,
179                         w, height);
180 
181                     x+= w;
182                 } else {
183                     ctx.strokeStyle='#f00';
184                     ctx.strokeRect( x,y,UNKNOWN_CHAR_WIDTH,height );
185                     x+= UNKNOWN_CHAR_WIDTH;
186                 }
187             }
188         },
189 
190         save : function() {
191             var str= "image/png";
192             var strData= this.image.toDataURL(str);
193             document.location.href= strData.replace( str, "image/octet-stream" );
194         }
195 
196     };
197 
198 })();
