 13         createAlphaSpriteSheet:function (maxAlpha, minAlpha, sheetSize, image, bg_fill_style) {
 14 
 15             if (maxAlpha < minAlpha) {
 16                 var t = maxAlpha;
 17                 maxAlpha = minAlpha;
 18                 minAlpha = t;
 19             }
 20 
 21             var canvas = document.createElement('canvas');
 22             canvas.width = image.width;
 23             canvas.height = image.height * sheetSize;
 24             var ctx = canvas.getContext('2d');
 25             ctx.fillStyle = bg_fill_style ? bg_fill_style : 'rgba(255,255,255,0)';
 26             ctx.fillRect(0, 0, image.width, image.height * sheetSize);
 27 
 28             var i;
 29             for (i = 0; i < sheetSize; i++) {
 30                 ctx.globalAlpha = 1 - (maxAlpha - minAlpha) / sheetSize * (i + 1);
 31                 ctx.drawImage(image, 0, i * image.height);
 32             }
 33 
 34             return canvas;
 35         },
 36 
 37         /**
 38          * Creates a rotated canvas image element.
 39          */
 40         rotate:function (image, angle) {
 41 
 42             angle = angle || 0;
 43             if (!angle) {
 44                 return image;
 45             }
 46 
 47             var canvas = document.createElement("canvas");
 48             canvas.width = image.height;
 49             canvas.height = image.width;
 50             var ctx = canvas.getContext('2d');
 51             ctx.globalAlpha = 1;
 52             ctx.fillStyle = 'rgba(0,0,0,0)';
 53             ctx.clearRect(0, 0, canvas.width, canvas.height);
 54 
 55             var m = new CAAT.Math.Matrix();
 56             m.multiply(new CAAT.Math.Matrix().setTranslate(canvas.width / 2, canvas.width / 2));
 57             m.multiply(new CAAT.Math.Matrix().setRotation(angle * Math.PI / 180));
 58             m.multiply(new CAAT.Math.Matrix().setTranslate(-canvas.width / 2, -canvas.width / 2));
 59             m.transformRenderingContext(ctx);
 60             ctx.drawImage(image, 0, 0);
 61 
 62             return canvas;
 63         },

 65         /**
 66          * Remove an image's padding transparent border.
 67          * Transparent means that every scan pixel is alpha=0.
 68          */
 69         optimize:function (image, threshold, areas) {
 70             threshold >>= 0;
 71 
 72             var atop = true;
 73             var abottom = true;
 74             var aleft = true;
 75             var aright = true;
 76             if (typeof areas !== 'undefined') {
 77                 if (typeof areas.top !== 'undefined') {
 78                     atop = areas.top;
 79                 }
 80                 if (typeof areas.bottom !== 'undefined') {
 81                     abottom = areas.bottom;
 82                 }
 83                 if (typeof areas.left !== 'undefined') {
 84                     aleft = areas.left;
 85                 }
 86                 if (typeof areas.right !== 'undefined') {
 87                     aright = areas.right;
 88                 }
 89             }
 90 
 91 
 92             var canvas = document.createElement('canvas');
 93             canvas.width = image.width;
 94             canvas.height = image.height;
 95             var ctx = canvas.getContext('2d');
 96 
 97             ctx.fillStyle = 'rgba(0,0,0,0)';
 98             ctx.fillRect(0, 0, image.width, image.height);
 99             ctx.drawImage(image, 0, 0);
100 
101             var imageData = ctx.getImageData(0, 0, image.width, image.height);
102             var data = imageData.data;
103 
104             var i, j;
105             var miny = 0, maxy = canvas.height - 1;
106             var minx = 0, maxx = canvas.width - 1;
107 
108             var alpha = false;
109 
110             if (atop) {
111                 for (i = 0; i < canvas.height; i++) {
112                     for (j = 0; j < canvas.width; j++) {
113                         if (data[i * canvas.width * 4 + 3 + j * 4] > threshold) {
114                             alpha = true;
115                             break;
116                         }
117                     }
118 
119                     if (alpha) {
120                         break;
121                     }
122                 }
123                 // i contiene el indice del ultimo scan que no es transparente total.
124                 miny = i;
125             }
126 
127             if (abottom) {
128                 alpha = false;
129                 for (i = canvas.height - 1; i >= miny; i--) {
130                     for (j = 0; j < canvas.width; j++) {
131                         if (data[i * canvas.width * 4 + 3 + j * 4] > threshold) {
132                             alpha = true;
133                             break;
134                         }
135                     }
136 
137                     if (alpha) {
138                         break;
139                     }
140                 }
141                 maxy = i;
142             }
143 
144             if (aleft) {
145                 alpha = false;
146                 for (j = 0; j < canvas.width; j++) {
147                     for (i = miny; i <= maxy; i++) {
148                         if (data[i * canvas.width * 4 + 3 + j * 4 ] > threshold) {
149                             alpha = true;
150                             break;
151                         }
152                     }
153                     if (alpha) {
154                         break;
155                     }
156                 }
157                 minx = j;
158             }
159 
160             if (aright) {
161                 alpha = false;
162                 for (j = canvas.width - 1; j >= minx; j--) {
163                     for (i = miny; i <= maxy; i++) {
164                         if (data[i * canvas.width * 4 + 3 + j * 4 ] > threshold) {
165                             alpha = true;
166                             break;
167                         }
168                     }
169                     if (alpha) {
170                         break;
171                     }
172                 }
173                 maxx = j;
174             }
175 
176             if (0 === minx && 0 === miny && canvas.width - 1 === maxx && canvas.height - 1 === maxy) {
177                 return canvas;
178             }
179 
180             var width = maxx - minx + 1;
181             var height = maxy - miny + 1;
182             var id2 = ctx.getImageData(minx, miny, width, height);
183 
184             canvas.width = width;
185             canvas.height = height;
186             ctx = canvas.getContext('2d');
187             ctx.putImageData(id2, 0, 0);
188 
189             return canvas;
190         },
191

193         createThumb:function (image, w, h, best_fit) {
194             w = w || 24;
195             h = h || 24;
196             var canvas = document.createElement('canvas');
197             canvas.width = w;
198             canvas.height = h;
199             var ctx = canvas.getContext('2d');
200 
201             if (best_fit) {
202                 var max = Math.max(image.width, image.height);
203                 var ww = image.width / max * w;
204                 var hh = image.height / max * h;
205                 ctx.drawImage(image, (w - ww) / 2, (h - hh) / 2, ww, hh);
206             } else {
207                 ctx.drawImage(image, 0, 0, w, h);
208             }
209 
210             return canvas;
211         }
