  5 CAAT.Module( {
  6     defines : "CAAT.WebGL.GLU",
  7     depends : [
  8         "CAAT.Math.Matrix3"
  9     ],
 10     onCreate : function() {
 11         makePerspective= function (fovy, aspect, znear, zfar, viewportHeight) {
 12             var ymax = znear * Math.tan(fovy * Math.PI / 360.0);
 13             var ymin = -ymax;
 14             var xmin = ymin * aspect;
 15             var xmax = ymax * aspect;
 16 
 17             return makeFrustum(xmin, xmax, ymin, ymax, znear, zfar, viewportHeight);
 18         }
 19 
 20         //
 21         // glFrustum
 22         //
 23         makeFrustum= function (left, right, bottom, top, znear, zfar, viewportHeight) {
 24             var X = 2*znear/(right-left);
 25             var Y = 2*znear/(top-bottom);
 26             var A = (right+left)/(right-left);
 27             var B = (top+bottom)/(top-bottom);
 28             var C = -(zfar+znear)/(zfar-znear);
 29             var D = -2*zfar*znear/(zfar-znear);
 30 
 31             return new CAAT.Math.Matrix3().initWithMatrix(
 32                     [
 33                         [X,  0,  A, -viewportHeight/2 ],
 34                         [0, -Y,  B,  viewportHeight/2 ],
 35                         [0,  0,  C,                 D ],
 36                         [0,  0, -1,                 0 ]
 37                     ]);
 38         }
 39 
 40         makeOrtho= function (left, right, bottom, top, znear, zfar) {
 41             var tx = - (right + left) / (right - left) ;
 42             var ty = - (top + bottom) / (top - bottom) ;
 43             var tz = - (zfar + znear) / (zfar - znear);
 44 
 45             return new CAAT.Math.Matrix3().initWithMatrix(
 46                     [
 47                         [2 / (right - left), 0, 0, tx ],
 48                         [0, 2 / (top - bottom), 0, ty ],
 49                         [0, 0, -2 / (zfar- znear), tz ],
 50                         [0, 0, 0,                  1  ]
 51                     ]);
 52         }
 53 
 54     }
 55 });
