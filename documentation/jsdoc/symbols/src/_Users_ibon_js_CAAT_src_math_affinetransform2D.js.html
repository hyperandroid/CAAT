 41 
 42             point.x= x*this.matrix[0][0] + y*this.matrix[0][1] + z*this.matrix[0][2] + this.matrix[0][3];
 43             point.y= x*this.matrix[1][0] + y*this.matrix[1][1] + z*this.matrix[1][2] + this.matrix[1][3];
 44             point.z= x*this.matrix[2][0] + y*this.matrix[2][1] + z*this.matrix[2][2] + this.matrix[2][3];

157         /**
158          * Creates a matrix to represent arbitrary rotations around the given planes.
159          * @param xy {number} radians to rotate around xy plane.
160          * @param xz {number} radians to rotate around xz plane.
161          * @param yz {number} radians to rotate around yz plane.
162          *
163          * @return {CAAT.Matrix3} a newly allocated matrix.
164          * @static
165          */
166 	    rotate : function( xy, xz, yz ) {
167 		    var res=new CAAT.Matrix3();
168 		    var s,c,m;
169 
170 		    if (xy!==0) {
171 			    m =new CAAT.Matrix3( );
172 			    s=Math.sin(xy);
173 			    c=Math.cos(xy);
174 			    m.matrix[1][1]=c;
175 			    m.matrix[1][2]=-s;
176 			    m.matrix[2][1]=s;
177 			    m.matrix[2][2]=c;
178 			    res.multiply(m);
179 		    }
180 
181 		    if (xz!==0) {
182 			    m =new CAAT.Matrix3( );
183 			    s=Math.sin(xz);
184 			    c=Math.cos(xz);
185 			    m.matrix[0][0]=c;
186 			    m.matrix[0][2]=-s;
187 			    m.matrix[2][0]=s;
188 			    m.matrix[2][2]=c;
189 			    res.multiply(m);
190 		    }
191 
192 		    if (yz!==0) {
193 			    m =new CAAT.Matrix3( );
194 			    s=Math.sin(yz);
195 			    c=Math.cos(yz);
196 			    m.matrix[0][0]=c;
197 			    m.matrix[0][1]=-s;
198 			    m.matrix[1][0]=s;
199 			    m.matrix[1][1]=c;
200 			    res.multiply(m);
201 		    }
202 
203 		    return res;

245             var m00= mm[0][0];
246             var m01= mm[0][1];
247             var m02= mm[0][2];

389         /**

397 	    rotateModelView : function( xy, xz, yz ) {
398 		    var sxy= Math.sin( xy );
399             var sxz= Math.sin( xz );
400             var syz= Math.sin( yz );
401             var cxy= Math.cos( xy );
402             var cxz= Math.cos( xz );
403             var cyz= Math.cos( yz );
404 
405             this.matrix[0][0]= cxz*cxy;
406             this.matrix[0][1]= -cxz*sxy;
407             this.matrix[0][2]= sxz;
408             this.matrix[0][3]= 0;
409             this.matrix[1][0]= syz*sxz*cxy+sxy*cyz;
410             this.matrix[1][1]= cyz*cxy-syz*sxz*sxy;
411             this.matrix[1][2]= -syz*cxz;
412             this.matrix[1][3]= 0;
413             this.matrix[2][0]= syz*sxy-cyz*sxz*cxy;
414             this.matrix[2][1]= cyz*sxz*sxy+syz*cxy;
415             this.matrix[2][2]= cyz*cxz;
416             this.matrix[2][3]= 0;
417             this.matrix[3][0]= 0;
418             this.matrix[3][1]= 0;
419             this.matrix[3][2]= 0;
420             this.matrix[3][3]= 1;
421 
422             return this;

439         /**
440          * Calculate this matrix's determinant.
441          * @return {number} matrix determinant.
442          */
443         calculateDeterminant: function () {
444 
445             var mm= this.matrix;
446 		    var m11= mm[0][0], m12= mm[0][1], m13= mm[0][2], m14= mm[0][3],
447 		        m21= mm[1][0], m22= mm[1][1], m23= mm[1][2], m24= mm[1][3],
448 		        m31= mm[2][0], m32= mm[2][1], m33= mm[2][2], m34= mm[2][3],
449 		        m41= mm[3][0], m42= mm[3][1], m43= mm[3][2], m44= mm[3][3];
450 
451             return  m14 * m22 * m33 * m41 +
452                     m12 * m24 * m33 * m41 +
453                     m14 * m23 * m31 * m42 +
454                     m13 * m24 * m31 * m42 +
455 
456                     m13 * m21 * m34 * m42 +
457                     m11 * m23 * m34 * m42 +
458                     m14 * m21 * m32 * m43 +
459                     m11 * m24 * m32 * m43 +
460 
461                     m13 * m22 * m31 * m44 +
462                     m12 * m23 * m31 * m44 +
463                     m12 * m21 * m33 * m44 +
464                     m11 * m22 * m33 * m44 +
465 
466                     m14 * m23 * m32 * m41 -
467                     m13 * m24 * m32 * m41 -
468                     m13 * m22 * m34 * m41 -
469                     m12 * m23 * m34 * m41 -
470 
471                     m14 * m21 * m33 * m42 -
472                     m11 * m24 * m33 * m42 -

485         getInverse : function() {
486             var mm= this.matrix;
487 		    var m11 = mm[0][0], m12 = mm[0][1], m13 = mm[0][2], m14 = mm[0][3],
488 		        m21 = mm[1][0], m22 = mm[1][1], m23 = mm[1][2], m24 = mm[1][3],
489 		        m31 = mm[2][0], m32 = mm[2][1], m33 = mm[2][2], m34 = mm[2][3],
490 		        m41 = mm[3][0], m42 = mm[3][1], m43 = mm[3][2], m44 = mm[3][3];
491             
492             var m2= new CAAT.Matrix3();
493             m2.matrix[0][0]= m23*m34*m42 + m24*m32*m43 + m22*m33*m44 - m24*m33*m42 - m22*m34*m43 - m23*m32*m44;
494             m2.matrix[0][1]= m14*m33*m42 + m12*m34*m43 + m13*m32*m44 - m12*m33*m44 - m13*m34*m42 - m14*m32*m43;
495             m2.matrix[0][2]= m13*m24*m42 + m12*m23*m44 + m14*m22*m43 - m12*m24*m43 - m13*m22*m44 - m14*m23*m42;
496             m2.matrix[0][3]= m14*m23*m32 + m12*m24*m33 + m13*m22*m34 - m13*m24*m32 - m14*m22*m33 - m12*m23*m34;
497 
498             m2.matrix[1][0]= m24*m33*m41 + m21*m34*m43 + m23*m31*m44 - m23*m34*m41 - m24*m31*m43 - m21*m33*m44;
499             m2.matrix[1][1]= m13*m34*m41 + m14*m31*m43 + m11*m33*m44 - m14*m33*m41 - m11*m34*m43 - m13*m31*m44;
500             m2.matrix[1][2]= m14*m23*m41 + m11*m24*m43 + m13*m21*m44 - m13*m24*m41 - m14*m21*m43 - m11*m23*m44;
501             m2.matrix[1][3]= m13*m24*m31 + m14*m21*m33 + m11*m23*m34 - m14*m23*m31 - m11*m24*m33 - m13*m21*m34;
502 
503             m2.matrix[2][0]= m22*m34*m41 + m24*m31*m42 + m21*m32*m44 - m24*m32*m41 - m21*m34*m42 - m22*m31*m44;
504             m2.matrix[2][1]= m14*m32*m41 + m11*m34*m42 + m12*m31*m44 - m11*m32*m44 - m12*m34*m41 - m14*m31*m42;
505             m2.matrix[2][2]= m13*m24*m41 + m14*m21*m42 + m11*m22*m44 - m14*m22*m41 - m11*m24*m42 - m12*m21*m44;
506             m2.matrix[2][3]= m14*m22*m31 + m11*m24*m32 + m12*m21*m34 - m11*m22*m34 - m12*m24*m31 - m14*m21*m32;
507 
508             m2.matrix[3][0]= m23*m32*m41 + m21*m33*m42 + m22*m31*m43 - m22*m33*m41 - m23*m31*m42 - m21*m32*m43;
509             m2.matrix[3][1]= m12*m33*m41 + m13*m31*m42 + m11*m32*m43 - m13*m32*m41 - m11*m33*m42 - m12*m31*m43;
510             m2.matrix[3][2]= m13*m22*m41 + m11*m23*m42 + m12*m21*m43 - m11*m22*m43 - m12*m23*m41 - m13*m21*m42;
511             m2.matrix[3][3]= m12*m23*m31 + m13*m21*m32 + m11*m22*m33 - m13*m22*m31 - m11*m23*m32 - m12*m21*m33;

513             return m2.multiplyScalar( 1/this.calculateDeterminant() );
514         },

564         /**
565          * Transform a point by this matrix. The parameter point will be modified with the transformation values.
566          * @param point {CAAT.Point}.
567          * @return {CAAT.Point} the parameter point.
568          */

763 
764             this.matrix[0]= m00;
765             this.matrix[1]= m01;
766             this.matrix[2]= m02;

781          * @return {CAAT.Matrix} an inverse matrix.
782          */
783 	    getInverse : function() {
784             var tm= this.matrix;
785 
786 			var m00= tm[0];
787 			var m01= tm[1];
788             var m02= tm[2];
789 			var m10= tm[3];
790 			var m11= tm[4];
791             var m12= tm[5];
792             var m20= tm[6];
793             var m21= tm[7];
794             var m22= tm[8];
795 
796             var newMatrix= new CAAT.Matrix();
797 
798             var determinant= m00* (m11*m22 - m21*m12) - m10*(m01*m22 - m21*m02) + m20 * (m01*m12 - m11*m02);
799             if ( determinant===0 ) {
800                 return null;
801             }
802 
803             var m= newMatrix.matrix;
804 
805             m[0]= m11*m22-m12*m21;
806             m[1]= m02*m21-m01*m22;
807             m[2]= m01*m12-m02*m11;
808 
809             m[3]= m12*m20-m10*m22;
810             m[4]= m00*m22-m02*m20;
811             m[5]= m02*m10-m00*m12;
812 
813             m[6]= m10*m21-m11*m20;
814             m[7]= m01*m20-m00*m21;
815             m[8]= m00*m11-m01*m10;
816 
817             newMatrix.multiplyScalar( 1/determinant );
818 
819 			return newMatrix;
