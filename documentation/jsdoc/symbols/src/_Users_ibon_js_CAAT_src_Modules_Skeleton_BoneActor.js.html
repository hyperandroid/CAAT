 76                         var sdkf= this.skinDataKeyframes[i];

 79                         }
 80                     }

100                     ctx.save();
101                         ctx.translate(-w+skinInfo.x, -h+skinInfo.y );

103                         ctx.translate(w, h);
104                         ctx.rotate(skinInfo.angle);
105                         ctx.translate( -w, -h);

107                         ctx.drawImage( skinInfo.image, 0, 0, skinInfo.image.width, skinInfo.image.height );
