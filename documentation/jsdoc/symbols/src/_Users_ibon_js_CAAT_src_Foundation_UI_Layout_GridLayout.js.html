 30             var ncols = this.columns;
 31 
 32             if (nrows > 0) {
 33                 ncols = Math.floor( (nactors + nrows - 1) / nrows );
 34             } else {
 35                 nrows = Math.floor( (nactors + ncols - 1) / ncols );
 36             }
 37 
 38             var totalGapsWidth = (ncols - 1) * this.hgap;
 39             var widthWOInsets = container.width - (this.padding.left + this.padding.right);
 40             var widthOnComponent = Math.floor( (widthWOInsets - totalGapsWidth) / ncols );
 41             var extraWidthAvailable = Math.floor( (widthWOInsets - (widthOnComponent * ncols + totalGapsWidth)) / 2 );
 42 
 43             var totalGapsHeight = (nrows - 1) * this.vgap;
 44             var heightWOInsets = container.height - (this.padding.top + this.padding.bottom);
 45             var heightOnComponent = Math.floor( (heightWOInsets - totalGapsHeight) / nrows );
 46             var extraHeightAvailable = Math.floor( (heightWOInsets - (heightOnComponent * nrows + totalGapsHeight)) / 2 );
 47 
 48             for (var c = 0, x = this.padding.left + extraWidthAvailable; c < ncols ; c++, x += widthOnComponent + this.hgap) {
 49                 for (var r = 0, y = this.padding.top + extraHeightAvailable; r < nrows ; r++, y += heightOnComponent + this.vgap) {
 50                     var i = r * ncols + c;
