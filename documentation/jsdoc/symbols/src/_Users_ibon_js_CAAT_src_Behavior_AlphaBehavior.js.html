1 CAAT.Module({

  8 
  9             startAlpha:0,

 12             getPropertyName:function () {
 13                 return "opacity";

 15 
 16             /**
 17              * Applies corresponding alpha transparency value for a given time.
 18              *
 19              * @param time the time to apply the scale for.
 20              * @param actor the target actor to set transparency for.
 21              * @return {number} the alpha value set. Normalized from 0 (total transparency) to 1 (total opacity)

 23             setForTime:function (time, actor) {
 24 
 25                 CAAT.Behavior.AlphaBehavior.superclass.setForTime.call(this, time, actor);
 26 
 27                 var alpha = (this.startAlpha + time * (this.endAlpha - this.startAlpha));
 28                 if (this.doValueApplication) {
 29                     actor.setAlpha(alpha);
 30                 }
 31                 return alpha;
 32             },
 33             /**
 34              * Set alpha transparency minimum and maximum value.

 36              *
 37              * @param start {number} a float indicating the starting alpha value.
 38              * @param end {number} a float indicating the ending alpha value.
 39              */
 40             setValues:function (start, end) {
 41                 this.startAlpha = start;
 42                 this.endAlpha = end;
 43                 return this;
 44             },
 45 
 46             calculateKeyFrameData:function (time) {
 47                 time = this.interpolator.getPosition(time).y;
 48                 return  (this.startAlpha + time * (this.endAlpha - this.startAlpha));
 49             },
 50 
 51             /**
 52              * @param prefix {string} browser vendor prefix
 53              * @param name {string} keyframes animation name
 54              * @param keyframessize {number} number of keyframes to generate
 55              * @override
 56              */
 57             calculateKeyFramesData:function (prefix, name, keyframessize) {
 58 
 59                 if (typeof keyframessize === 'undefined') {
 60                     keyframessize = 100;
 61                 }
 62                 keyframessize >>= 0;
 63 
 64                 var i;
 65                 var kfr;
 66                 var kfd = "@-" + prefix + "-keyframes " + name + " {";
 67 
 68                 for (i = 0; i <= keyframessize; i++) {
 69                     kfr = "" +
 70                         (i / keyframessize * 100) + "%" + // percentage
 71                         "{" +
 72                         "opacity: " + this.calculateKeyFrameData(i / keyframessize) +
 73                         "}";
 74 
 75                     kfd += kfr;
 76                 }
 77 
 78                 kfd += "}";
 79 
 80                 return kfd;
 81             }
 82         }
 83     }
