612         /**
613          * Indicates that some path control point has changed, and that the path must recalculate
614          * its internal data, ie: length and bbox.
615          */
616 		updatePath : function(point, callback) {
617             var i,j;
618 
619             this.length=0;
620             this.bbox.setEmpty();
621             this.points= [];
622 
623             var xmin= Number.MAX_VALUE, ymin= Number.MAX_VALUE;
624 			for( i=0; i<this.pathSegments.length; i++ ) {
625 				this.pathSegments[i].updatePath(point);
626                 this.length+= this.pathSegments[i].getLength();
627                 this.bbox.unionRectangle( this.pathSegments[i].bbox );
628 
629                 for( j=0; j<this.pathSegments[i].numControlPoints(); j++ ) {
630                     var pt= this.pathSegments[i].getControlPoint( j );
631                     this.points.push( pt );
632                     if ( pt.x < xmin ) {
633                         xmin= pt.x;
634                     }
635                     if ( pt.y < ymin ) {
636                         ymin= pt.y;
637                     }
638                 }
639 			}
640 
641             this.clipOffsetX= -xmin;
642             this.clipOffsetY= -ymin;
643 
644             this.width= this.bbox.width;
645             this.height= this.bbox.height;
646             this.setLocation( this.bbox.x, this.bbox.y );
647 
648             this.pathSegmentStartTime=      [];
649             this.pathSegmentDurationTime=   [];
650             
651             var i;
652             for( i=0; i<this.pathSegments.length; i++) {
653                 this.pathSegmentStartTime.push(0);
654                 this.pathSegmentDurationTime.push(0);
655             }
656 
657             for( i=0; i<this.pathSegments.length; i++) {
658                 this.pathSegmentDurationTime[i]= this.getLength() ? this.pathSegments[i].getLength()/this.getLength() : 0;
659                 if ( i>0 ) {
660                     this.pathSegmentStartTime[i]= this.pathSegmentStartTime[i-1]+this.pathSegmentDurationTime[i-1];
661                 } else {
662                     this.pathSegmentStartTime[0]= 0;
663                 }
664 
665                 this.pathSegments[i].endPath();
666             }
667 
668             this.extractPathPoints();
669 
670             if ( callback ) {
671                 callback(this);
672             }
673 
674             return this;

863         setATMatrix : function() {
864             this.matrix.identity();
865 
866             var m= this.tmpMatrix.identity();
867             var mm= this.matrix.matrix;
868             var c,s,_m00,_m01,_m10,_m11;
869             var mm0, mm1, mm2, mm3, mm4, mm5;
870 
871             var bbox= this.bbox;
872             var bbw= bbox.width  ;
873             var bbh= bbox.height ;
874             var bbx= bbox.x;
875             var bby= bbox.y
876 
877             mm0= 1;
878             mm1= 0;
879             mm3= 0;
880             mm4= 1;
881 
882             mm2= this.tb_x - bbx - this.tAnchorX * bbw;
883             mm5= this.tb_y - bby - this.tAnchorY * bbh;
884 
885             if ( this.rb_angle ) {
886 
887                 var rbx= (this.rb_rotateAnchorX*bbw + bbx);
888                 var rby= (this.rb_rotateAnchorY*bbh + bby);
889 
890                 mm2+= mm0*rbx + mm1*rby;
891                 mm5+= mm3*rbx + mm4*rby;
892 
893                 c= Math.cos( this.rb_angle );
894                 s= Math.sin( this.rb_angle);
895                 _m00= mm0;
896                 _m01= mm1;
897                 _m10= mm3;
898                 _m11= mm4;
899                 mm0=  _m00*c + _m01*s;
900                 mm1= -_m00*s + _m01*c;
901                 mm3=  _m10*c + _m11*s;
902                 mm4= -_m10*s + _m11*c;
903 
904                 mm2+= -mm0*rbx - mm1*rby;
905                 mm5+= -mm3*rbx - mm4*rby;
906             }
907 
908             if ( this.sb_scaleX!=1 || this.sb_scaleY!=1 ) {
909 
910                 var sbx= (this.sb_scaleAnchorX*bbw + bbx);
911                 var sby= (this.sb_scaleAnchorY*bbh + bby);
912 
913                 mm2+= mm0*sbx + mm1*sby;
914                 mm5+= mm3*sbx + mm4*sby;
915 
916                 mm0= mm0*this.sb_scaleX;
917                 mm1= mm1*this.sb_scaleY;
918                 mm3= mm3*this.sb_scaleX;
919                 mm4= mm4*this.sb_scaleY;
920 
921                 mm2+= -mm0*sbx - mm1*sby;
922                 mm5+= -mm3*sbx - mm4*sby;
923             }
924 
925             mm[0]= mm0;
926             mm[1]= mm1;
927             mm[2]= mm2;
928             mm[3]= mm3;
929             mm[4]= mm4;
930             mm[5]= mm5;
931 
932             return this;
933
