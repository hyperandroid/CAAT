  7 (function() {
  8 
  9     /**
 10      * This class is a sound manager implementation which can play at least 'numChannels' sounds at the same time.
 11      * By default, CAAT.Director instances will set eight channels to play sound.
 12      * <p>
 13      * If more than 'numChannels' sounds want to be played at the same time the requests will be dropped,
 14      * so no more than 'numChannels' sounds can be concurrently played.
 15      * <p>
 16      * Available sounds to be played must be supplied to every CAAT.Director instance by calling <code>addSound</code>
 17      * method. The default implementation will accept a URL/URI or a HTMLAudioElement as source.
 18      * <p>
 19      * The cached elements can be played, or looped. The <code>loop</code> method will return a handler to
 20      * give the opportunity of cancelling the sound.
 21      * <p>
 22      * Be aware of Audio.canPlay, is able to return 'yes', 'no', 'maybe', ..., so anything different from
 23      * '' and 'no' will do.
 24      *
 25      * @constructor
 26      *
 27      */
 28     CAAT.AudioManager= function() {
 29         this.browserInfo= new CAAT.BrowserDetect();
 30         return this;
 31     };
 32 
 33     CAAT.AudioManager.prototype= {
 34 
 35         browserInfo:        null,
 36         musicEnabled:       true,
 37         fxEnabled:          true,
 38         audioCache:         null,   // audio elements.
 39         channels:           null,   // available playing channels.
 40         workingChannels:    null,   // currently playing channels.
 41         loopingChannels:    [],
 42         audioTypes: {               // supported audio formats. Don't remember where i took them from :S
 43 	        'mp3': 'audio/mpeg;',
 44             'ogg': 'audio/ogg; codecs="vorbis"',
 45             'wav': 'audio/wav; codecs="1"',
 46             'mp4': 'audio/mp4; codecs="mp4a.40.2"'
 47 		},
 48 
 49         /**
 50          * Initializes the sound subsystem by creating a fixed number of Audio channels.
 51          * Every channel registers a handler for sound playing finalization. If a callback is set, the
 52          * callback function will be called with the associated sound id in the cache.
 53          *
 54          * @param numChannels {number} number of channels to pre-create. 8 by default.
 55          *
 56          * @return this.
 57          */
 58         initialize : function(numChannels) {
 59 
 60             this.audioCache=      [];
 61             this.channels=        [];
 62             this.workingChannels= [];
 63 
 64             for( var i=0; i<numChannels; i++ ) {
 65                 var channel= document.createElement('audio');
 66 
 67                 if ( null!==channel ) {
 68                     channel.finished= -1;
 69                     this.channels.push( channel );
 70                     var me= this;
 71                     channel.addEventListener(
 72                             'ended',
 73                             // on sound end, set channel to available channels list.
 74                             function(audioEvent) {
 75                                 var target= audioEvent.target;
 76                                 var i;
 77 
 78                                 // remove from workingChannels
 79                                 for( i=0; i<me.workingChannels.length; i++ ) {
 80                                     if (me.workingChannels[i]===target ) {
 81                                         me.workingChannels.splice(i,1);
 82                                         break;
 83                                     }
 84                                 }
 85 
 86                                 if ( target.caat_callback ) {
 87                                     target.caat_callback(target.caat_id);
 88                                 }
 89 
 90                                 // set back to channels.
 91                                 me.channels.push(target);
 92                             },
 93                             false
 94                     );
 95                 }
 96             }
 97 
 98             return this;
 99         },
100         /**
101          * Tries to add an audio tag to the available list of valid audios. The audio is described by a url.
102          * @param id {object} an object to associate the audio element (if suitable to be played).
103          * @param url {string} a string describing an url.
104          * @param endplaying_callback {function} callback to be called upon sound end.
105          *
106          * @return {boolean} a boolean indicating whether the browser can play this resource.
107          *
108          * @private
109          */
110         addAudioFromURL : function( id, url, endplaying_callback ) {
111             var extension= null;
112             var audio= document.createElement('audio');
113 
114             if ( null!==audio ) {
115 
116                 if(!audio.canPlayType) {
117                     return false;
118                 }
119 
120                 extension= url.substr(url.lastIndexOf('.')+1);
121                 var canplay= audio.canPlayType(this.audioTypes[extension]);
122 
123                 if(canplay!=="" && canplay!=="no") {
124                     audio.src= url;
125                     audio.preload = "auto";
126                     audio.load();
127                     if ( endplaying_callback ) {
128                         audio.caat_callback= endplaying_callback;
129                         audio.caat_id= id;
130                     }
131                     this.audioCache.push( { id:id, audio:audio } );
132 
133                     return true;
134                 }
135             }
136 
137             return false;
138         },
139         /**
140          * Tries to add an audio tag to the available list of valid audios. The audio element comes from
141          * an HTMLAudioElement.
142          * @param id {object} an object to associate the audio element (if suitable to be played).
143          * @param audio {HTMLAudioElement} a DOM audio node.
144          * @param endplaying_callback {function} callback to be called upon sound end.
145          *
146          * @return {boolean} a boolean indicating whether the browser can play this resource.
147          *
148          * @private
149          */
150         addAudioFromDomNode : function( id, audio, endplaying_callback ) {
151 
152             var extension= audio.src.substr(audio.src.lastIndexOf('.')+1);
153             if ( audio.canPlayType(this.audioTypes[extension]) ) {
154                 if ( endplaying_callback ) {
155                     audio.caat_callback= endplaying_callback;
156                     audio.caat_id= id;
157                 }
158                 this.audioCache.push( { id:id, audio:audio } );
159 
160                 return true;
161             }
162 
163             return false;
164         },
165         /**
166          * Adds an elements to the audio cache.
167          * @param id {object} an object to associate the audio element (if suitable to be played).
168          * @param element {URL|HTMLElement} an url or html audio tag.
169          * @param endplaying_callback {function} callback to be called upon sound end.
170          *
171          * @return {boolean} a boolean indicating whether the browser can play this resource.
172          *
173          * @private
174          */
175         addAudioElement : function( id, element, endplaying_callback ) {
176             if ( typeof element === "string" ) {
177                 return this.addAudioFromURL( id, element, endplaying_callback );
178             } else {
179                 try {
180                     if ( element instanceof HTMLAudioElement ) {
181                         return this.addAudioFromDomNode( id, element, endplaying_callback );
182                     }
183                 }
184                 catch(e) {
185                 }
186             }
187 
188             return false;
189         },
190         /**
191          * creates an Audio object and adds it to the audio cache.
192          * This function expects audio data described by two elements, an id and an object which will
193          * describe an audio element to be associated with the id. The object will be of the form
194          * array, dom node or a url string.
195          *
196          * <p>
197          * The audio element can be one of the two forms:
198          *
199          * <ol>
200          *  <li>Either an HTMLAudioElement/Audio object or a string url.
201          *  <li>An array of elements of the previous form.
202          * </ol>
203          *
204          * <p>
205          * When the audio attribute is an array, this function will iterate throught the array elements
206          * until a suitable audio element to be played is found. When this is the case, the other array
207          * elements won't be taken into account. The valid form of using this addAudio method will be:
208          *
209          * <p>
210          * 1.<br>
211          * addAudio( id, url } ). In this case, if the resource pointed by url is
212          * not suitable to be played (i.e. a call to the Audio element's canPlayType method return 'no')
213          * no resource will be added under such id, so no sound will be played when invoking the play(id)
214          * method.
215          * <p>
216          * 2.<br>
217          * addAudio( id, dom_audio_tag ). In this case, the same logic than previous case is applied, but
218          * this time, the parameter url is expected to be an audio tag present in the html file.
219          * <p>
220          * 3.<br>
221          * addAudio( id, [array_of_url_or_domaudiotag] ). In this case, the function tries to locate a valid
222          * resource to be played in any of the elements contained in the array. The array element's can
223          * be any type of case 1 and 2. As soon as a valid resource is found, it will be associated to the
224          * id in the valid audio resources to be played list.
225          *
226          * @return this
227          */
228         addAudio : function( id, array_of_url_or_domnodes, endplaying_callback ) {
229 
230             if ( array_of_url_or_domnodes instanceof Array ) {
231                 /*
232                  iterate throught array elements until we can safely add an audio element.
233                  */
234                 for( var i=0; i<array_of_url_or_domnodes.length; i++ ) {
235                     if ( this.addAudioElement(id, array_of_url_or_domnodes[i], endplaying_callback) ) {
236                         break;
237                     }
238                 }
239             } else {
240                 this.addAudioElement(id, array_of_url_or_domnodes, endplaying_callback);
241             }
242 
243             return this;
244         },
245         /**
246          * Returns an audio object.
247          * @param aId {object} the id associated to the target Audio object.
248          * @return {object} the HTMLAudioElement addociated to the given id.
249          */
250         getAudio : function(aId) {
251             for( var i=0; i<this.audioCache.length; i++ ) {
252                 if ( this.audioCache[i].id===aId ) {
253                     return this.audioCache[i].audio;
254                 }
255             }
256 
257             return null;
258         },
259 
260         /**
261          * Set an audio object volume.
262          * @param id {object} an audio Id
263          * @param volume {number} volume to set. The volume value is not checked.
264          *
265          * @return this
266          */
267         setVolume : function( id, volume ) {
268             var audio= this.getAudio(id);
269             if ( null!=audio ) {
270                 audio.volume= volume;
271             }
272 
273             return this;
274         },
275 
276         /**
277          * Plays an audio file from the cache if any sound channel is available.
278          * The playing sound will occupy a sound channel and when ends playing will leave
279          * the channel free for any other sound to be played in.
280          * @param id {object} an object identifying a sound in the sound cache.
281          * @return this.
282          */
283         play : function( id ) {
284             if ( !this.fxEnabled ) {
285                 return this;
286             }
287 
288             var audio= this.getAudio(id);
289             // existe el audio, y ademas hay un canal de audio disponible.
290             if ( null!==audio && this.channels.length>0 ) {
291                 var channel= this.channels.shift();
292                 channel.src= audio.src;
293                 channel.load();
294                 channel.volume= audio.volume;
295                 channel.play();
296                 this.workingChannels.push(channel);
297             }
298 
299             return this;
300         },
301         /**
302          * This method creates a new AudioChannel to loop the sound with.
303          * It returns an Audio object so that the developer can cancel the sound loop at will.
304          * The user must call <code>pause()</code> method to stop playing a loop.
305          * <p>
306          * Firefox does not honor the loop property, so looping is performed by attending end playing
307          * event on audio elements.
308          *
309          * @return {HTMLElement} an Audio instance if a valid sound id is supplied. Null otherwise
310          */
311         loop : function( id ) {
312 
313             if (!this.musicEnabled) {
314                 return this;
315             }
316 
317             var audio_in_cache= this.getAudio(id);
318             // existe el audio, y ademas hay un canal de audio disponible.
319             if ( null!==audio_in_cache ) {
320                 var audio= document.createElement('audio');
321                 if ( null!==audio ) {
322                     audio.src= audio_in_cache.src;
323                     audio.preload = "auto";
324 
325                     if ( this.browserInfo.browser==='Firefox') {
326                         audio.addEventListener(
327                             'ended',
328                             // on sound end, set channel to available channels list.
329                             function(audioEvent) {
330                                 var target= audioEvent.target;
331                                 target.currentTime=0;
332                             },
333                             false
334                         );
335                     } else {
336                         audio.loop= true;
337                     }
338                     audio.load();
339                     audio.play();
340                     this.loopingChannels.push(audio);
341                     return audio;
342                 }
343             }
344 
345             return null;
346         },
347         /**
348          * Cancel all playing audio channels
349          * Get back the playing channels to available channel list.
350          *
351          * @return this
352          */
353         endSound : function() {
354             var i;
355             for( i=0; i<this.workingChannels.length; i++ ) {
356                 this.workingChannels[i].pause();
357                 this.channels.push( this.workingChannels[i] );
358             }
359 
360             for( i=0; i<this.loopingChannels.length; i++ ) {
361                 this.loopingChannels[i].pause();
362             }
363 
364             return this;
365         },
366         setSoundEffectsEnabled : function( enable ) {
367             this.fxEnabled= enable;
368             return this;
369         },
370         isSoundEffectsEnabled : function() {
371             return this.fxEnabled;
372         },
373         setMusicEnabled : function( enable ) {
374             this.musicEnabled= enable;
375             for( var i=0; i<this.loopingChannels.length; i++ ) {
376                 if ( enable ) {
377                     this.loopingChannels[i].play();
378                 } else {
379                     this.loopingChannels[i].pause();
380                 }
381             }
382             return this;
383         },
384         isMusicEnabled : function() {
385             return this.musicEnabled;
386         }
387     };
388 })();
