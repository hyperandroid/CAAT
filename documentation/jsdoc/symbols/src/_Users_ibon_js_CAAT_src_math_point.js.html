  5 CAAT.Module( {
  6 
  7     defines:        "CAAT.Math.Point",
  8     aliases:        ["CAAT.Point"],
  9     extendsWith:    function() {
 10         return {
 11 
 12             /**
 13              *
 14              * A point defined by two coordinates.
 15              *
 16              * @param xpos {number}
 17              * @param ypos {number}
 18              *
 19              * @constructor
 20              */
 21             x:  0,
 22                 y:  0,
 23             z:  0,
 24 
 25             __init : function(xpos, ypos, zpos) {
 26             this.x= xpos;
 27             this.y= ypos;
 28             this.z= zpos||0;
 29             return this;
 30         },
 31 
 32             /**
 33              * Sets this point coordinates.
 34              * @param x {number}
 35              * @param y {number}
 36              *
 37              * @return this
 38              */
 39             set : function(x,y,z) {
 40             this.x= x;
 41             this.y= y;
 42             this.z= z||0;
 43             return this;
 44         },
 45             /**
 46              * Create a new CAAT.Point equal to this one.
 47              * @return {CAAT.Point}
 48              */
 49             clone : function() {
 50                 var p= new CAAT.Math.Point(this.x, this.y, this.z );
 51                 return p;
 52             },
 53             /**
 54              * Translate this point to another position. The final point will be (point.x+x, point.y+y)
 55              * @param x {number}
 56              * @param y {number}
 57              *
 58              * @return this
 59              */
 60             translate : function(x,y,z) {
 61                 this.x+= x;
 62                 this.y+= y;
 63                 this.z+= z;
 64 
 65                 return this;
 66             },
 67             /**
 68              * Translate this point to another point.
 69              * @param aPoint {CAAT.Point}
 70              * @return this
 71              */
 72             translatePoint: function(aPoint) {
 73                 this.x += aPoint.x;
 74                 this.y += aPoint.y;
 75                 this.z += aPoint.z;
 76                 return this;
 77             },
 78             /**
 79              * Substract a point from this one.
 80              * @param aPoint {CAAT.Point}
 81              * @return this
 82              */
 83             subtract: function(aPoint) {
 84                 this.x -= aPoint.x;
 85                 this.y -= aPoint.y;
 86                 this.z -= aPoint.z;
 87                 return this;
 88             },
 89             /**
 90              * Multiply this point by a scalar.
 91              * @param factor {number}
 92              * @return this
 93              */
 94             multiply: function(factor) {
 95                 this.x *= factor;
 96                 this.y *= factor;
 97                 this.z *= factor;
 98                 return this;
 99             },
100             /**
101              * Rotate this point by an angle. The rotation is held by (0,0) coordinate as center.
102              * @param angle {number}
103              * @return this
104              */
105             rotate: function(angle) {
106                 var x = this.x, y = this.y;
107                 this.x = x * Math.cos(angle) - Math.sin(angle) * y;
108                 this.y = x * Math.sin(angle) + Math.cos(angle) * y;
109                 this.z = 0;
110                 return this;
111             },
112             /**
113              *
114              * @param angle {number}
115              * @return this
116              */
117             setAngle: function(angle) {
118                 var len = this.getLength();
119                 this.x = Math.cos(angle) * len;
120                 this.y = Math.sin(angle) * len;
121                 this.z = 0;
122                 return this;
123             },
124             /**
125              *
126              * @param length {number}
127              * @return this
128              */
129             setLength: function(length)	{
130                 var len = this.getLength();
131                 if (len)this.multiply(length / len);
132                 else this.x = this.y = this.z = length;
133                 return this;
134             },
135             /**
136              * Normalize this point, that is, both set coordinates proportionally to values raning 0..1
137              * @return this
138              */
139             normalize: function() {
140                 var len = this.getLength();
141                 this.x /= len;
142                 this.y /= len;
143                 this.z /= len;
144                 return this;
145             },
146             /**
147              * Return the angle from -Pi to Pi of this point.
148              * @return {number}
149              */
150             getAngle: function() {
151                 return Math.atan2(this.y, this.x);
152             },
153             /**
154              * Set this point coordinates proportinally to a maximum value.
155              * @param max {number}
156              * @return this
157              */
158             limit: function(max) {
159                 var aLenthSquared = this.getLengthSquared();
160                 if(aLenthSquared+0.01 > max*max)
161                 {
162                     var aLength = Math.sqrt(aLenthSquared);
163                     this.x= (this.x/aLength) * max;
164                     this.y= (this.y/aLength) * max;
165                     this.z= (this.z/aLength) * max;
166                 }
167                 return this;
168             },
169             /**
170              * Get this point's lenght.
171              * @return {number}
172              */
173             getLength: function() {
174                 var length = Math.sqrt(this.x*this.x + this.y*this.y + this.z*this.z);
175                 if ( length < 0.005 && length > -0.005) return 0.000001;
176                 return length;
177 
178             },
179             /**
180              * Get this point's squared length.
181              * @return {number}
182              */
183             getLengthSquared: function() {
184                 var lengthSquared = this.x*this.x + this.y*this.y + this.z*this.z;
185                 if ( lengthSquared < 0.005 && lengthSquared > -0.005) return 0;
186                 return lengthSquared;
187             },
188             /**
189              * Get the distance between two points.
190              * @param point {CAAT.Point}
191              * @return {number}
192              */
193             getDistance: function(point) {
194                 var deltaX = this.x - point.x;
195                 var deltaY = this.y - point.y;
196                 var deltaZ = this.z - point.z;
197                 return Math.sqrt( deltaX*deltaX + deltaY*deltaY + deltaZ*deltaZ );
198             },
199             /**
200              * Get the squared distance between two points.
201              * @param point {CAAT.Point}
202              * @return {number}
203              */
204             getDistanceSquared: function(point) {
205                 var deltaX = this.x - point.x;
206                 var deltaY = this.y - point.y;
207                 var deltaZ = this.z - point.z;
208                 return deltaX*deltaX + deltaY*deltaY + deltaZ*deltaZ;
209             },
210             /**
211              * Get a string representation.
212              * @return {string}
213              */
214             toString: function() {
215                 return "(CAAT.Math.Point)" +
216                     " x:" + String(Math.round(Math.floor(this.x*10))/10) +
217                     " y:" + String(Math.round(Math.floor(this.y*10))/10) +
218                     " z:" + String(Math.round(Math.floor(this.z*10))/10);
219             }
220         }
221     }
222 });
