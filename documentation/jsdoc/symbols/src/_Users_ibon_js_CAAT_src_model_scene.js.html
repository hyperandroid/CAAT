  9      * The only time when 2 scenes could be active will be during scene change.
 10      * An scene controls the way it enters/exits the scene graph. It is also the entry point for all
 11      * input related and timed related events to every actor on screen.
 12      *
 13      * @constructor
 14      * @extends CAAT.ActorContainer
 15      *
 16      */

 47         /**
 48          * Check and apply timers in frame time.
 49          * @param time {number} the current Scene time.
 50          */
 51         checkTimers : function(time) {
 52             var tl= this.timerList;
 53             var i=tl.length-1;
 54             while( i>=0 ) {
 55                 if ( !tl[i].remove ) {
 56                     tl[i].checkTask(time);
 57                 }
 58                 i--;
 59             }
 60         },
 61         /**
 62          * Make sure the timertask is contained in the timer task list by adding it to the list in case it
 63          * is not contained.
 64          * @param timertask {CAAT.TimerTask} a CAAT.TimerTask object.
 65          * @return this
 66          */
 67         ensureTimerTask : function( timertask ) {
 68             if ( !this.hasTimer(timertask) ) {
 69                 this.timerList.push(timertask);
 70             }
 71             return this;
 72         },
 73         /**
 74          * Check whether the timertask is in this scene's timer task list.
 75          * @param timertask {CAAT.TimerTask} a CAAT.TimerTask object.
 76          * @return {boolean} a boolean indicating whether the timertask is in this scene or not.
 77          */
 78         hasTimer : function( timertask ) {
 79             var tl= this.timerList;
 80             var i=tl.length-1;
 81             while( i>=0 ) {
 82                 if ( tl[i]===timertask ) {
 83                     return true;
 84                 }
 85                 i--;
 86             }
 87 
 88             return false;
 89         },
 90         /**
 91          * Creates a timer task. Timertask object live and are related to scene's time, so when an Scene
 92          * is taken out of the Director the timer task is paused, and resumed on Scene restoration.
 93          *
 94          * @param startTime {number} an integer indicating the scene time this task must start executing at.
 95          * @param duration {number} an integer indicating the timerTask duration.
 96          * @param callback_timeout {function} timer on timeout callback function.
 97          * @param callback_tick {function} timer on tick callback function.
 98          * @param callback_cancel {function} timer on cancel callback function.
 99          *
100          * @return {CAAT.TimerTask} a CAAT.TimerTask class instance.
101          */
102         createTimer : function( startTime, duration, callback_timeout, callback_tick, callback_cancel ) {
103 
104             var tt= new CAAT.TimerTask().create(
105                         startTime,
106                         duration,
107                         callback_timeout, 
108                         callback_tick,
109                         callback_cancel );
110 
111             tt.taskId= this.timerSequence++;
112             tt.sceneTime = this.time;
113             tt.scene= this;
114 
115             this.timerList.push( tt );
116 
117             return tt;
118         },

179         /**
180          * This method will setup Scene behaviours to switch an Scene via a translation.
181          * The anchor value can only be
182          *  <li>CAAT.Actor.prototype.ANCHOR_LEFT
183          *  <li>CAAT.Actor.prototype.ANCHOR_RIGHT
184          *  <li>CAAT.Actor.prototype.ANCHOR_TOP
185          *  <li>CAAT.Actor.prototype.ANCHOR_BOTTOM
186          * if any other value is specified, any of the previous ones will be applied.
187          *
188          * @param time integer indicating time in milliseconds for the Scene.
189          * @param alpha boolean indicating whether fading will be applied to the Scene.
190          * @param anchor integer indicating the Scene switch anchor.
191          * @param isIn boolean indicating whether the scene will be brought in.
192          * @param interpolator {CAAT.Interpolator} a CAAT.Interpolator to apply to the Scene transition.
193          */
194 		easeTranslation : function( time, alpha, anchor, isIn, interpolator ) {
195 
196             this.easeContainerBehaviour= new CAAT.ContainerBehavior();
197             this.easeIn= isIn;
198 
199             var pb= new CAAT.PathBehavior();
200             if ( interpolator ) {
201                 pb.setInterpolator( interpolator );
202             }
203 
204             pb.setFrameTime( 0, time );
205 
206             // BUGBUG anchors: 1..4
207             if ( anchor<1 ) {
208                 anchor=1;
209             } else if ( anchor>4 ) {
210                 anchor= 4;
211             }
212 
213 
214 			switch(anchor) {
215 			case CAAT.Actor.prototype.ANCHOR_TOP:
216                 if ( isIn ) {
217                     pb.setPath( new CAAT.Path().setLinear( 0, -this.height, 0, 0) );
218                 } else {
219                     pb.setPath( new CAAT.Path().setLinear( 0, 0, 0, -this.height) );
220                 }
221                 break;
222             case CAAT.Actor.prototype.ANCHOR_BOTTOM:
223                 if ( isIn ) {
224                     pb.setPath( new CAAT.Path().setLinear( 0, this.height, 0, 0) );
225                 } else {
226                     pb.setPath( new CAAT.Path().setLinear( 0, 0, 0, this.height) );
227                 }
228                 break;
229             case CAAT.Actor.prototype.ANCHOR_LEFT:
230                 if ( isIn ) {
231                     pb.setPath( new CAAT.Path().setLinear( -this.width, 0, 0, 0) );
232                 } else {
233                     pb.setPath( new CAAT.Path().setLinear( 0, 0, -this.width, 0) );
234                 }
235                 break;
236             case CAAT.Actor.prototype.ANCHOR_RIGHT:
237                 if ( isIn ) {
238                     pb.setPath( new CAAT.Path().setLinear( this.width, 0, 0, 0) );
239                 } else {
240                     pb.setPath( new CAAT.Path().setLinear( 0, 0, this.width, 0) );
241                 }
242                 break;
243             }
244 
245 			if (alpha) {
246 				this.createAlphaBehaviour(time,isIn);
247 			}
248 
249 			this.easeContainerBehaviour.addBehavior(pb);
250 
251 			this.easeContainerBehaviour.setFrameTime( this.time, time );
252 			this.easeContainerBehaviour.addListener(this);
253 
254 			this.emptyBehaviorList();
255 			CAAT.Scene.superclass.addBehavior.call( this, this.easeContainerBehaviour );
256 		},

283         /**
284          * Called from CAAT.Director to bring in ot take away an Scene.
285          * @param time integer indicating time in milliseconds for the Scene to be taken away.
286          * @param alpha boolean indicating whether fading will be applied to the Scene.
287          * @param anchor integer indicating the Scene switch anchor.
288          * @param interpolator {CAAT.Interpolator} a CAAT.Interpolator to apply to the Scene transition.
289          * @param starttime integer indicating in milliseconds from which scene time the behavior will be applied.
290          * @param isIn boolean indicating whether the Scene is being brought in.
291          */
292 		easeScale : function(starttime,time,alpha,anchor,isIn,interpolator) {
293 			this.easeContainerBehaviour= new CAAT.ContainerBehavior();
294 
295 			var x=0;
296 			var y=0;
297 			var x2=0;
298 			var y2=0;
299 
300 			switch(anchor) {
301 			case CAAT.Actor.prototype.ANCHOR_TOP_LEFT:
302 			case CAAT.Actor.prototype.ANCHOR_TOP_RIGHT:
303 			case CAAT.Actor.prototype.ANCHOR_BOTTOM_LEFT:
304 			case CAAT.Actor.prototype.ANCHOR_BOTTOM_RIGHT:
305 			case CAAT.Actor.prototype.ANCHOR_CENTER:
306 				x2=1;
307 				y2=1;
308 				break;
309 			case CAAT.Actor.prototype.ANCHOR_TOP:
310 			case CAAT.Actor.prototype.ANCHOR_BOTTOM:
311 				x=1;
312 				x2=1;
313 				y=0;
314 				y2=1;
315 				break;
316 			case CAAT.Actor.prototype.ANCHOR_LEFT:
317 			case CAAT.Actor.prototype.ANCHOR_RIGHT:
318 				y=1;
319 				y2=1;
320 				x=0;
321 				x2=1;
322 				break;
323 			default:
324 				alert('scale anchor ?? '+anchor);
325 			}
326 
327 			if ( !isIn ) {
328 				var tmp;
329 				tmp= x;
330 				x= x2;
331 				x2= tmp;
332 				
333 				tmp= y;
334 				y= y2;
335 				y2= tmp;
336 			}
337 			
338 			if (alpha) {
339 				this.createAlphaBehaviour(time,isIn);
340 			}
341 
342             var anchorPercent= this.getAnchorPercent(anchor);
343 			var sb= new CAAT.ScaleBehavior().
344 			        setFrameTime( starttime, time ).
345                     setValues(x,x2,y,y2, anchorPercent.x, anchorPercent.y);
346 
347             if ( interpolator ) {
348                 sb.setInterpolator(interpolator);
349             }
350 
351 			this.easeContainerBehaviour.addBehavior(sb);
352 			
353 			this.easeContainerBehaviour.setFrameTime( this.time, time );
354 			this.easeContainerBehaviour.addListener(this);
355 			
356 			this.emptyBehaviorList();
357 			CAAT.Scene.superclass.addBehavior.call( this, this.easeContainerBehaviour );
358 		},

390         /**
391          * Called from CAAT.Director to use Rotations for taking away or bringing Scenes in.
392          * @param time integer indicating time in milliseconds for the Scene to be taken away or brought in.
393          * @param alpha boolean indicating whether fading will be applied to the Scene.
394          * @param anchor integer indicating the Scene switch anchor.
395          * @param interpolator {CAAT.Interpolator} a CAAT.Interpolator to apply to the Scene transition.
396          * @param isIn boolean indicating whehter the Scene is brought in.
397          */
398 		easeRotation : function(time,alpha,anchor,isIn,interpolator) {
399 			this.easeContainerBehaviour= new CAAT.ContainerBehavior();
400 			
401 			var start=0;
402 			var end=0;
403 
404             if (anchor==CAAT.Actor.prototype.ANCHOR_CENTER ) {
405                 anchor= CAAT.Actor.prototype.ANCHOR_TOP;
406             }
407 
408 			switch(anchor) {
409 			case CAAT.Actor.prototype.ANCHOR_TOP:
410 			case CAAT.Actor.prototype.ANCHOR_BOTTOM:
411 			case CAAT.Actor.prototype.ANCHOR_LEFT:
412 			case CAAT.Actor.prototype.ANCHOR_RIGHT:
413 				start= Math.PI * (Math.random()<0.5 ? 1 : -1);
414 				break;
415 			case CAAT.Actor.prototype.ANCHOR_TOP_LEFT:
416 			case CAAT.Actor.prototype.ANCHOR_TOP_RIGHT:
417 			case CAAT.Actor.prototype.ANCHOR_BOTTOM_LEFT:
418 			case CAAT.Actor.prototype.ANCHOR_BOTTOM_RIGHT:
419 				start= Math.PI/2 * (Math.random()<0.5 ? 1 : -1);
420 				break;
421 			default:
422 				alert('rot anchor ?? '+anchor);
423 			}
424 
425 			if ( false===isIn ) {
426 				var tmp= start;
427 				start=end;
428 				end= tmp;
429 			}
430 
431 			if ( alpha ) {
432 				this.createAlphaBehaviour(time,isIn);
433 			}
434 
435             var anchorPercent= this.getAnchorPercent(anchor);
436 			var rb= new CAAT.RotateBehavior().
437 			        setFrameTime( 0, time ).
438                     setValues( start, end, anchorPercent.x, anchorPercent.y );
439 
440             if ( interpolator ) {
441                 rb.setInterpolator(interpolator);
442             }
443 			this.easeContainerBehaviour.addBehavior(rb);
444 			this.easeContainerBehaviour.setFrameTime( this.time, time );
445 			this.easeContainerBehaviour.addListener(this);
446 			
447 			this.emptyBehaviorList();
448 			CAAT.Scene.superclass.addBehavior.call( this, this.easeContainerBehaviour );
449 		},

494         /**
495          * Find a pointed actor at position point.
496          * This method tries lo find the correctly pointed actor in two different ways.
497          *  + first of all, if inputList is defined, it will look for an actor in it.
498          *  + if no inputList is defined, it will traverse the scene graph trying to find a pointed actor.
499          * @param point <CAAT.Point>
500          */
501         findActorAtPosition : function(point) {
502             var i,j;
503 
504             var p= new CAAT.Point();
505 
506             if ( this.inputList ) {
507                 var il= this.inputList;
508                 for( i=0; i<il.length; i++ ) {
509                     var ill= il[i];
510                     for( j=0; j<ill.length; j++ ) {
511                         p.set(point.x, point.y);
512                         var modelViewMatrixI= ill[j].worldModelViewMatrix.getInverse();
513                         modelViewMatrixI.transformCoord(p);
514                         if ( ill[j].contains(p.x, p.y) ) {
515                             return ill[j];
516                         }
517                     }
518                 }
519             }
520 
521             p.set(point.x, point.y);
522             return CAAT.Scene.superclass.findActorAtPosition.call(this,p);
523         },

525         /**
526          * Enable a number of input lists.
527          * These lists are set in case the developer doesn't want the to traverse the scene graph to find the pointed
528          * actor. The lists are a shortcut whete the developer can set what actors to look for input at first instance.
529          * The system will traverse the whole lists in order trying to find a pointed actor.
530          *
531          * Elements are added to each list either in head or tail.
532          *
533          * @param size <number> number of lists.
534          */
535         enableInputList : function( size ) {
536             this.inputList= [];
537             for( var i=0; i<size; i++ ) {
538                 this.inputList.push([]);
539             }
540 
541             return this;
542         },
543 
544         /**
545          * Add an actor to a given inputList.
546          * @param actor <CAAT.Actor> an actor instance
547          * @param index <number> the inputList index to add the actor to. This value will be clamped to the number of
548          * available lists.
549          * @param position <number> the position on the selected inputList to add the actor at. This value will be
550          * clamped to the number of available lists.
551          */
552         addActorToInputList : function( actor, index, position ) {
553             if ( index<0 ) index=0; else if ( index>=this.inputList.length ) index= this.inputList.length-1;
554             var il= this.inputList[index];
555 
556             if ( typeof position==="undefined" || position>=il.length ) {
557                 il.push( actor );
558             } else if (position<=0) {
559                 il.unshift( actor );
560             } else {
561                 il.splice( position, 0, actor );
562             }
563 
564             return this;
565         },
566 
567         /**
568          * Remove all elements from an input list.
569          * @param index <number> the inputList index to add the actor to. This value will be clamped to the number of
570          * available lists so take care when emptying a non existant inputList index since you could end up emptying
571          * an undesired input list.
572          */
573         emptyInputList : function( index ) {
574             if ( index<0 ) index=0; else if ( index>=this.inputList.length ) index= this.inputList.length-1;
575             this.inputList[index]= [];
576             return this;
577         },
578 
579         /**
580          * remove an actor from a given input list index.
581          * If no index is supplied, the actor will be removed from every input list.
582          * @param actor <CAAT.Actor>
583          * @param index <!number> an optional input list index. This value will be clamped to the number of
584          * available lists.
585          */
586         removeActorFromInputList : function( actor, index ) {
587             if ( typeof index==="undefined" ) {
588                 var i,j;
589                 for( i=0; i<this.inputList.length; i++ ) {
590                     var il= this.inputList[i];
591                     for( j=0; j<il.length; j++ ) {
592                         if ( il[j]==actor ) {
593                             il.splice( j,1 );
594                         }
595                     }
596                 }
597                 return this;
598             }
599 
600             if ( index<0 ) index=0; else if ( index>=this.inputList.length ) index= this.inputList.length-1;
601             var il= this.inputList[index];
602             for( j=0; j<il.length; j++ ) {
603                 if ( il[j]==actor ) {
604                     il.splice( j,1 );
605                 }
606             }
607 
608             return this;
609         }
