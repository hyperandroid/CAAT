1 /**
  2  * See LICENSE file.
  3  *
  4  * Behaviors are keyframing elements.
  5  * By using a BehaviorContainer, you can specify different actions on any animation Actor.
  6  * An undefined number of Behaviors can be defined for each Actor.
  7  *
  8  * There're the following Behaviors:
  9  *  + AlphaBehavior:   controls container/actor global alpha.
 10  *  + RotateBehavior:  takes control of rotation affine transform.
 11  *  + ScaleBehavior:   takes control of scaling on x/y axis affine transform.
 12  *  + PathBehavior:    takes control of translating an Actor/ActorContainer across a path [ie. pathSegment collection].
 13  *  + GenericBehavior: applies a behavior to any given target object's property, or notifies a callback.
 14  *
 15  *

 19     /**
 20      * Behavior base class.
 21      *
 22      * <p>
 23      * A behavior is defined by a frame time (behavior duration) and a behavior application function called interpolator.
 24      * In its default form, a behaviour is applied linearly, that is, the same amount of behavior is applied every same
 25      * time interval.
 26      * <p>
 27      * A concrete Behavior, a rotateBehavior in example, will change a concrete Actor's rotationAngle during the specified
 28      * period.
 29      * <p>
 30      * A behavior is guaranteed to notify (if any observer is registered) on behavior expiration.
 31      * <p>
 32      * A behavior can keep an unlimited observers. Observers are objects of the form:
 33      * <p>
 34      * <code>
 35      * {
 36      *      behaviorExpired : function( behavior, time, actor);
 37      *      behaviorApplied : function( behavior, time, normalizedTime, actor, value);
 38      * }
 39      * </code>
 40      * <p>
 41      * <strong>behaviorExpired</strong>: function( behavior, time, actor). This method will be called for any registered observer when
 42      * the scene time is greater than behavior's startTime+duration. This method will be called regardless of the time
 43      * granurality.
 44      * <p>
 45      * <strong>behaviorApplied</strong> : function( behavior, time, normalizedTime, actor, value). This method will be called once per
 46      * frame while the behavior is not expired and is in frame time (behavior startTime>=scene time). This method can be
 47      * called multiple times.
 48      * <p>
 49      * Every behavior is applied to a concrete Actor.
 50      * Every actor must at least define an start and end value. The behavior will set start-value at behaviorStartTime and
 51      * is guaranteed to apply end-value when scene time= behaviorStartTime+behaviorDuration.
 52      * <p>
 53      * You can set behaviors to apply forever that is cyclically. When a behavior is cycle=true, won't notify
 54      * behaviorExpired to its registered observers.
 55      * <p>
 56      * Other Behaviors simply must supply with the method <code>setForTime(time, actor)</code> overriden.
 57      *
 58      * @constructor
 59      */

390     /**
391      * <p>
392      * A ContainerBehavior is a holder to sum up different behaviors.
393      * <p>
394      * It imposes some constraints to contained Behaviors:
395      * <ul>
396      * <li>The time of every contained behavior will be zero based, so the frame time set for each behavior will
397      * be referred to the container's behaviorStartTime and not scene time as usual.
398      * <li>Cycling a ContainerBehavior means cycling every contained behavior.
399      * <li>The container will not impose any Interpolator, so calling the method <code>setInterpolator(CAAT.Interpolator)
400      * </code> will be useless.
401      * <li>The Behavior application time will be bounded to the Container's frame time. I.E. if we set a container duration
402      * to 10 seconds, setting a contained behavior's duration to 15 seconds will be useless since the container will stop
403      * applying the behavior after 10 seconds have elapsed.
404      * <li>Every ContainerBehavior adds itself as an observer for its contained Behaviors. The main reason is because
405      * ContainerBehaviors modify cycling properties of its contained Behaviors. When a contained
406      * Behavior is expired, if the Container has isCycle=true, will unexpire the contained Behavior, otherwise, it won't be
407      * applied in the next frame. It is left up to the developer to manage correctly the logic of other posible contained
408      * behaviors observers.
409      * </ul>
410      *
411      * <p>
412      * A ContainerBehavior can contain other ContainerBehaviors at will.
413      * <p>
414      * A ContainerBehavior will not apply any CAAT.Actor property change by itself, but will instrument its contained
415      * Behaviors to do so.
416      *
417      * @constructor
418      * @extends CAAT.Behavior
419      */
420     CAAT.ContainerBehavior= function() {
421 		CAAT.ContainerBehavior.superclass.constructor.call(this);
422 		this.behaviors= [];
423 		return this;

536         calculateKeyFrameData : function(referenceTime, prefix, prevValues )  {
537 
538             var i;
539             var bh;
540 
541             var retValue= {};
542             var time;
543             var cssRuleValue;
544             var cssProperty;
545             var property;
546 
547             for( i=0; i<this.behaviors.length; i++ ) {
548                 bh= this.behaviors[i];
549                 if ( /*!bh.expired*/ bh.status!==CAAT.Behavior.Status.EXPIRED && !(bh instanceof CAAT.GenericBehavior) ) {
550 
551                     // ajustar tiempos:
552                     //  time es tiempo normalizado a duracion de comportamiento contenedor.
553                     //      1.- desnormalizar
554                     time= referenceTime * this.behaviorDuration;
555 
556                     //      2.- calcular tiempo relativo de comportamiento respecto a contenedor
557                     if ( bh.behaviorStartTime<=time && bh.behaviorStartTime+bh.behaviorDuration>=time ) {
558                         //      3.- renormalizar tiempo reltivo a comportamiento.
559                         time= (time-bh.behaviorStartTime)/bh.behaviorDuration;
560

562                         cssRuleValue= bh.calculateKeyFrameData(time);
563                         cssProperty= bh.getPropertyName(prefix);
564 
565                         if ( typeof retValue[cssProperty] ==='undefined' ) {
566                             retValue[cssProperty]= "";
567                         }
568 
569                         //      5.- asignar a objeto, par de propiedad/valor css
570                         retValue[cssProperty]+= cssRuleValue+" ";
571                     }
572 
573                 }
574             }
575 
576 
577             var tr="";
578             var pv;
579             function xx(pr) {
580                 if ( retValue[pr] ) {
581                     tr+= retValue[pr];
582                 } else {
583                     if ( prevValues ) {
584                         pv= prevValues[pr];
585                         if ( pv ) {
586                             tr+= pv;
587                             retValue[pr]= pv;
588                         }
589                     }
590                 }
591 
592             }
593 
594             xx('translate');
595             xx('rotate');
596             xx('scale');
597 
598             var keyFrameRule= "";
599 
600             if ( tr ) {
601                 keyFrameRule='-'+prefix+'-transform: '+tr+';';
602             }
603 
604             tr="";

606             if( tr ) {

608             }
