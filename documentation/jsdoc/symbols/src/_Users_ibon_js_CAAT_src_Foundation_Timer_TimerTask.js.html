 17         /**
 18          * Create a TimerTask.
 19          * The taskId will be set by the scene.
 20          * @param startTime {number} an integer indicating TimerTask enable time.
 21          * @param duration {number} an integer indicating TimerTask duration.
 22          * @param callback_timeout {function( sceneTime {number}, timertaskTime{number}, timertask {CAAT.TimerTask} )} on timeout callback function.
 23          * @param callback_tick {function( sceneTime {number}, timertaskTime{number}, timertask {CAAT.TimerTask} )} on tick callback function.
 24          * @param callback_cancel {function( sceneTime {number}, timertaskTime{number}, timertask {CAAT.TimerTask} )} on cancel callback function.
 25          *
 26          * @return this
 27          */
 28         create: function( startTime, duration, callback_timeout, callback_tick, callback_cancel ) {
 29             this.startTime=         startTime;
 30             this.duration=          duration;
 31             this.callback_timeout=  callback_timeout;
 32             this.callback_tick=     callback_tick;
 33             this.callback_cancel=   callback_cancel;
 34             return this;
 35         },
 36         /**
 37          * Performs TimerTask operation. The task will check whether it is in frame time, and will
 38          * either notify callback_timeout or callback_tick.
 39          *
 40          * @param time {number} an integer indicating scene time.
 41          * @return this
 42          *
 43          * @protected
 44          *
 45          */
 46         checkTask : function(time) {
 47             var ttime= time;
 48             ttime-= this.startTime;
 49             if ( ttime>=this.duration ) {
 50                 this.remove= true;
 51                 if( this.callback_timeout ) {
 52                     this.callback_timeout( time, ttime, this );
 53                 }
 54             } else {
 55                 if ( this.callback_tick ) {
 56                     this.callback_tick( time, ttime, this );
 57                 }
 58             }
 59             return this;
 60         },
 61         remainingTime : function() {
 62             return this.duration - (this.scene.time-this.startTime);
 63         },
 64         /**
 65          * Reschedules this TimerTask by changing its startTime to current scene's time.
 66          * @param time {number} an integer indicating scene time.
 67          * @return this
 68          */
 69         reset : function( time ) {
 70             this.remove= false;
 71             this.startTime=  time;
 72             this.owner.ensureTimerTask(this);
 73             return this;
 74         },
 75         /**
 76          * Cancels this timer by removing it on scene's next frame. The function callback_cancel will
 77          * be called.
 78          * @return this
 79          */
 80         cancel : function() {
 81             this.remove= true;
 82             if ( null!=this.callback_cancel ) {
 83                 this.callback_cancel( this.scene.time, this.scene.time-this.startTime, this );
 84             }
 85             return this;
 86         },
 87         addTime : function( time ) {
 88             this.duration+= time;
 89             return this;
 90         }
 91     }
