143 
144             var nactors= container.getNumChildren();
145             if (nactors === 0) {
146                 return;
147             }
148 
149             var nrows = this.rows;
150             var ncols = this.columns;
151 
152             if (nrows > 0) {
153                 ncols = Math.floor( (nactors + nrows - 1) / nrows );
154             } else {
155                 nrows = Math.floor( (nactors + ncols - 1) / ncols );
156             }
157 
158             var totalGapsWidth = (ncols - 1) * this.hgap;
159             var widthWOInsets = container.width - (this.padding.left + this.padding.right);
160             var widthOnComponent = Math.floor( (widthWOInsets - totalGapsWidth) / ncols );
161             var extraWidthAvailable = Math.floor( (widthWOInsets - (widthOnComponent * ncols + totalGapsWidth)) / 2 );
162 
163             var totalGapsHeight = (nrows - 1) * this.vgap;
164             var heightWOInsets = container.height - (this.padding.top + this.padding.bottom);
165             var heightOnComponent = Math.floor( (heightWOInsets - totalGapsHeight) / nrows );
166             var extraHeightAvailable = Math.floor( (heightWOInsets - (heightOnComponent * nrows + totalGapsHeight)) / 2 );
167 
168             for (var c = 0, x = this.padding.left + extraWidthAvailable; c < ncols ; c++, x += widthOnComponent + this.hgap) {
169                 for (var r = 0, y = this.padding.top + extraHeightAvailable; r < nrows ; r++, y += heightOnComponent + this.vgap) {
170                     var i = r * ncols + c;
171                     if (i < nactors) {
172                         var child= container.getChildAt(i);
173                         if ( !this.animated ) {
174                             child.setBounds(x, y, widthOnComponent, heightOnComponent);
175                         } else {
176                             if ( child.width!==widthOnComponent || child.height!==heightOnComponent ) {
177                                 child.setSize(widthOnComponent, heightOnComponent);
178                                 if ( this.newChildren.indexOf( child ) !==-1 ) {
179                                     child.setPosition( x,y );
180                                     child.setScale(0.01,0.01);
181                                     child.scaleTo( 1,1, 500, 0,.5,.5, CAAT.UI.LayoutManager.newElementInterpolator );
182                                 } else {
183                                     child.moveTo( x, y, 500, 0, CAAT.UI.LayoutManager.moveElementInterpolator );
184                                 }
185                             }
186                         }
187                     }
188                 }
189             }
