108             /**
109              * This method will setup Scene behaviours to switch an Scene via a translation.
110              * The anchor value can only be
111              *  <li>CAAT.Actor.ANCHOR_LEFT
112              *  <li>CAAT.Actor.ANCHOR_RIGHT
113              *  <li>CAAT.Actor.ANCHOR_TOP
114              *  <li>CAAT.Actor.ANCHOR_BOTTOM
115              * if any other value is specified, any of the previous ones will be applied.
116              *
117              * @param time {number} time in milliseconds for the Scene.
118              * @param alpha {boolean} whether fading will be applied to the Scene.
119              * @param anchor {numnber} Scene switch anchor.
120              * @param isIn {boolean} whether the scene will be brought in.
121              * @param interpolator {CAAT.Behavior.Interpolator} how to apply to the Scene transition.
122              */
123             easeTranslation:function (time, alpha, anchor, isIn, interpolator) {
124 
125                 this.easeContainerBehaviour = new CAAT.Behavior.ContainerBehavior();
126                 this.easeIn = isIn;
127 
128                 var pb = new CAAT.Behavior.PathBehavior();
129                 if (interpolator) {
130                     pb.setInterpolator(interpolator);
131                 }
132 
133                 pb.setFrameTime(0, time);
134 
135                 // BUGBUG anchors: 1..4
136                 if (anchor < 1) {
137                     anchor = 1;
138                 } else if (anchor > 4) {
139                     anchor = 4;
140                 }
141 
142 
143                 switch (anchor) {
144                     case CAAT.Foundation.Actor.ANCHOR_TOP:
145                         if (isIn) {
146                             pb.setPath(new CAAT.PathUtil.Path().setLinear(0, -this.height + 1, 0, 0));
147                             this.setPosition(0,-this.height+1);
148                         } else {
149                             pb.setPath(new CAAT.PathUtil.Path().setLinear(0, 0, 0, -this.height + 1));
150                             this.setPosition(0,0);
151                         }
152                         break;
153                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM:
154                         if (isIn) {
155                             pb.setPath(new CAAT.PathUtil.Path().setLinear(0, this.height - 1, 0, 0));
156                             this.setPosition(0,this.height-1);
157                         } else {
158                             pb.setPath(new CAAT.PathUtil.Path().setLinear(0, 0, 0, this.height - 1));
159                             this.setPosition(0,0);
160                         }
161                         break;
162                     case CAAT.Foundation.Actor.ANCHOR_LEFT:
163                         if (isIn) {
164                             pb.setPath(new CAAT.PathUtil.Path().setLinear(-this.width + 1, 0, 0, 0));
165                             this.setPosition(-this.width+1,0);
166                         } else {
167                             pb.setPath(new CAAT.PathUtil.Path().setLinear(0, 0, -this.width + 1, 0));
168                             this.setPosition(0,0);
169                         }
170                         break;
171                     case CAAT.Foundation.Actor.ANCHOR_RIGHT:
172                         if (isIn) {
173                             pb.setPath(new CAAT.PathUtil.Path().setLinear(this.width - 1, 0, 0, 0));
174                             this.setPosition(this.width-1,0);
175                         } else {
176                             pb.setPath(new CAAT.PathUtil.Path().setLinear(0, 0, this.width - 1, 0));
177                             this.setPosition(0,0);
178                         }
179                         break;
180                 }
181 
182                 if (alpha) {
183                     this.createAlphaBehaviour(time, isIn);
184                 }
185 
186                 this.easeContainerBehaviour.addBehavior(pb);
187 
188                 this.easeContainerBehaviour.setFrameTime(this.time, time);
189                 this.easeContainerBehaviour.addListener(this);
190 
191                 this.emptyBehaviorList();
192                 CAAT.Foundation.Scene.superclass.addBehavior.call(this, this.easeContainerBehaviour);
193             },

220             /**
221              * Called from CAAT.Foundation.Director to bring in ot take away an Scene.
222              * @param time {number} time in milliseconds for the Scene to be brought in.
223              * @param alpha {boolean} whether fading will be applied to the Scene.
224              * @param anchor {number} Scene switch anchor.
225              * @param interpolator {CAAT.Behavior.Interpolator} how to apply to the Scene transition.
226              * @param starttime {number} scene time milliseconds from which the behavior will be applied.
227              * @param isIn boolean indicating whether the Scene is being brought in.
228              */
229             easeScale:function (starttime, time, alpha, anchor, isIn, interpolator) {
230                 this.easeContainerBehaviour = new CAAT.Behavior.ContainerBehavior();
231 
232                 var x = 0;
233                 var y = 0;
234                 var x2 = 0;
235                 var y2 = 0;
236 
237                 switch (anchor) {
238                     case CAAT.Foundation.Actor.ANCHOR_TOP_LEFT:
239                     case CAAT.Foundation.Actor.ANCHOR_TOP_RIGHT:
240                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM_LEFT:
241                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM_RIGHT:
242                     case CAAT.Foundation.Actor.ANCHOR_CENTER:
243                         x2 = 1;
244                         y2 = 1;
245                         break;
246                     case CAAT.Foundation.Actor.ANCHOR_TOP:
247                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM:
248                         x = 1;
249                         x2 = 1;
250                         y = 0;
251                         y2 = 1;
252                         break;
253                     case CAAT.Foundation.Actor.ANCHOR_LEFT:
254                     case CAAT.Foundation.Actor.ANCHOR_RIGHT:
255                         y = 1;
256                         y2 = 1;
257                         x = 0;
258                         x2 = 1;
259                         break;
260                     default:
261                         alert('scale anchor ?? ' + anchor);
262                 }
263 
264                 if (!isIn) {
265                     var tmp;
266                     tmp = x;
267                     x = x2;
268                     x2 = tmp;
269 
270                     tmp = y;
271                     y = y2;
272                     y2 = tmp;
273                 }
274 
275                 if (alpha) {
276                     this.createAlphaBehaviour(time, isIn);
277                 }
278 
279                 var anchorPercent = this.getAnchorPercent(anchor);
280                 var sb = new CAAT.Behavior.ScaleBehavior().
281                     setFrameTime(starttime, time).
282                     setValues(x, x2, y, y2, anchorPercent.x, anchorPercent.y);
283 
284                 if (interpolator) {
285                     sb.setInterpolator(interpolator);
286                 }
287 
288                 this.easeContainerBehaviour.addBehavior(sb);
289 
290                 this.easeContainerBehaviour.setFrameTime(this.time, time);
291                 this.easeContainerBehaviour.addListener(this);
292 
293                 this.emptyBehaviorList();
294                 CAAT.Foundation.Scene.superclass.addBehavior.call(this, this.easeContainerBehaviour);
295             },

327             /**
328              * Called from CAAT.Director to use Rotations for taking away or bringing Scenes in.
329              * @param time integer indicating time in milliseconds for the Scene to be taken away or brought in.
330              * @param alpha boolean indicating whether fading will be applied to the Scene.
331              * @param anchor integer indicating the Scene switch anchor.
332              * @param interpolator {CAAT.Interpolator} a CAAT.Interpolator to apply to the Scene transition.
333              * @param isIn boolean indicating whehter the Scene is brought in.
334              */
335             easeRotation:function (time, alpha, anchor, isIn, interpolator) {
336                 this.easeContainerBehaviour = new CAAT.Behavior.ContainerBehavior();
337 
338                 var start = 0;
339                 var end = 0;
340 
341                 if (anchor == CAAT.Foundation.Actor.ANCHOR_CENTER) {
342                     anchor = CAAT.Foundation.Actor.ANCHOR_TOP;
343                 }
344 
345                 switch (anchor) {
346                     case CAAT.Foundation.Actor.ANCHOR_TOP:
347                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM:
348                     case CAAT.Foundation.Actor.ANCHOR_LEFT:
349                     case CAAT.Foundation.Actor.ANCHOR_RIGHT:
350                         start = Math.PI * (Math.random() < 0.5 ? 1 : -1);
351                         break;
352                     case CAAT.Foundation.Actor.ANCHOR_TOP_LEFT:
353                     case CAAT.Foundation.Actor.ANCHOR_TOP_RIGHT:
354                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM_LEFT:
355                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM_RIGHT:
356                         start = Math.PI / 2 * (Math.random() < 0.5 ? 1 : -1);
357                         break;
358                     default:
359                         alert('rot anchor ?? ' + anchor);
360                 }
361 
362                 if (false === isIn) {
363                     var tmp = start;
364                     start = end;
365                     end = tmp;
366                 }
367 
368                 if (alpha) {
369                     this.createAlphaBehaviour(time, isIn);
370                 }
371 
372                 var anchorPercent = this.getAnchorPercent(anchor);
373                 var rb = new CAAT.Behavior.RotateBehavior().
374                     setFrameTime(0, time).
375                     setValues(start, end, anchorPercent.x, anchorPercent.y);
376 
377                 if (interpolator) {
378                     rb.setInterpolator(interpolator);
379                 }
380                 this.easeContainerBehaviour.addBehavior(rb);
381                 this.easeContainerBehaviour.setFrameTime(this.time, time);
382                 this.easeContainerBehaviour.addListener(this);
383 
384                 this.emptyBehaviorList();
385                 CAAT.Foundation.Scene.superclass.addBehavior.call(this, this.easeContainerBehaviour);
386             },

431             /**
432              * Find a pointed actor at position point.
433              * This method tries lo find the correctly pointed actor in two different ways.
434              *  + first of all, if inputList is defined, it will look for an actor in it.
435              *  + if no inputList is defined, it will traverse the scene graph trying to find a pointed actor.
436              * @param point <CAAT.Point>
437              */
438             findActorAtPosition:function (point) {
439                 var i, j;
440 
441                 var p = new CAAT.Math.Point();
442 
443                 if (this.inputList) {
444                     var il = this.inputList;
445                     for (i = 0; i < il.length; i++) {
446                         var ill = il[i];
447                         for (j = 0; j < ill.length; j++) {
448                             p.set(point.x, point.y);
449                             var modelViewMatrixI = ill[j].worldModelViewMatrix.getInverse();
450                             modelViewMatrixI.transformCoord(p);
451                             if (ill[j].contains(p.x, p.y)) {
452                                 return ill[j];
453                             }
454                         }
455                     }
456                 }
457 
458                 p.set(point.x, point.y);
459                 return CAAT.Foundation.Scene.superclass.findActorAtPosition.call(this, p);
460             },

462             /**
463              * Enable a number of input lists.
464              * These lists are set in case the developer doesn't want the to traverse the scene graph to find the pointed
465              * actor. The lists are a shortcut whete the developer can set what actors to look for input at first instance.
466              * The system will traverse the whole lists in order trying to find a pointed actor.
467              *
468              * Elements are added to each list either in head or tail.
469              *
470              * @param size <number> number of lists.
471              */

481             /**
482              * Add an actor to a given inputList.
483              * @param actor <CAAT.Actor> an actor instance
484              * @param index <number> the inputList index to add the actor to. This value will be clamped to the number of
485              * available lists.
486              * @param position <number> the position on the selected inputList to add the actor at. This value will be
487              * clamped to the number of available lists.
488              */
489             addActorToInputList:function (actor, index, position) {
490                 if (index < 0) index = 0; else if (index >= this.inputList.length) index = this.inputList.length - 1;
491                 var il = this.inputList[index];
492 
493                 if (typeof position === "undefined" || position >= il.length) {
494                     il.push(actor);
495                 } else if (position <= 0) {
496                     il.unshift(actor);
497                 } else {
498                     il.splice(position, 0, actor);
499                 }
500 
501                 return this;
502             },

508              * an undesired input list.
509              */

511                 if (index < 0) index = 0; else if (index >= this.inputList.length) index = this.inputList.length - 1;

524                 if (typeof index === "undefined") {
525                     var i, j;
526                     for (i = 0; i < this.inputList.length; i++) {
527                         var il = this.inputList[i];
528                         for (j = 0; j < il.length; j++) {
529                             if (il[j] == actor) {
530                                 il.splice(j, 1);
531                             }
532                         }
533                     }
534                     return this;
535                 }
536 
537                 if (index < 0) index = 0; else if (index >= this.inputList.length) index = this.inputList.length - 1;
538                 var il = this.inputList[index];
539                 for (j = 0; j < il.length; j++) {
540                     if (il[j] == actor) {
541                         il.splice(j, 1);
542                     }
543                 }
544 
545                 return this;
546             },
