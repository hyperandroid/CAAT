112             /**
113              * This method will setup Scene behaviours to switch an Scene via a translation.
114              * The anchor value can only be
115              *  <li>CAAT.Actor.ANCHOR_LEFT
116              *  <li>CAAT.Actor.ANCHOR_RIGHT
117              *  <li>CAAT.Actor.ANCHOR_TOP
118              *  <li>CAAT.Actor.ANCHOR_BOTTOM
119              * if any other value is specified, any of the previous ones will be applied.
120              *
121              * @param time {number} time in milliseconds for the Scene.
122              * @param alpha {boolean} whether fading will be applied to the Scene.
123              * @param anchor {numnber} Scene switch anchor.
124              * @param isIn {boolean} whether the scene will be brought in.
125              * @param interpolator {CAAT.Behavior.Interpolator} how to apply to the Scene transition.
126              */
127             easeTranslation:function (time, alpha, anchor, isIn, interpolator) {
128 
129                 this.easeContainerBehaviour = new CAAT.Behavior.ContainerBehavior();
130                 this.easeIn = isIn;
131 
132                 var pb = new CAAT.Behavior.PathBehavior();
133                 if (interpolator) {
134                     pb.setInterpolator(interpolator);
135                 }
136 
137                 pb.setFrameTime(0, time);
138 
139                 // BUGBUG anchors: 1..4
140                 if (anchor < 1) {
141                     anchor = 1;
142                 } else if (anchor > 4) {
143                     anchor = 4;
144                 }
145 
146 
147                 switch (anchor) {
148                     case CAAT.Foundation.Actor.ANCHOR_TOP:
149                         if (isIn) {
150                             pb.setPath(new CAAT.PathUtil.Path().setLinear(0, -this.height + 1, 0, 0));
151                             this.setPosition(0,-this.height+1);
152                         } else {
153                             pb.setPath(new CAAT.PathUtil.Path().setLinear(0, 0, 0, -this.height + 1));
154                             this.setPosition(0,0);
155                         }
156                         break;
157                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM:
158                         if (isIn) {
159                             pb.setPath(new CAAT.PathUtil.Path().setLinear(0, this.height - 1, 0, 0));
160                             this.setPosition(0,this.height-1);
161                         } else {
162                             pb.setPath(new CAAT.PathUtil.Path().setLinear(0, 0, 0, this.height - 1));
163                             this.setPosition(0,0);
164                         }
165                         break;
166                     case CAAT.Foundation.Actor.ANCHOR_LEFT:
167                         if (isIn) {
168                             pb.setPath(new CAAT.PathUtil.Path().setLinear(-this.width + 1, 0, 0, 0));
169                             this.setPosition(-this.width+1,0);
170                         } else {
171                             pb.setPath(new CAAT.PathUtil.Path().setLinear(0, 0, -this.width + 1, 0));
172                             this.setPosition(0,0);
173                         }
174                         break;
175                     case CAAT.Foundation.Actor.ANCHOR_RIGHT:
176                         if (isIn) {
177                             pb.setPath(new CAAT.PathUtil.Path().setLinear(this.width - 1, 0, 0, 0));
178                             this.setPosition(this.width-1,0);
179                         } else {
180                             pb.setPath(new CAAT.PathUtil.Path().setLinear(0, 0, this.width - 1, 0));
181                             this.setPosition(0,0);
182                         }
183                         break;
184                 }
185 
186                 if (alpha) {
187                     this.createAlphaBehaviour(time, isIn);
188                 }
189 
190                 this.easeContainerBehaviour.addBehavior(pb);
191 
192                 this.easeContainerBehaviour.setFrameTime(this.time, time);
193                 this.easeContainerBehaviour.addListener(this);
194 
195                 this.emptyBehaviorList();
196                 CAAT.Foundation.Scene.superclass.addBehavior.call(this, this.easeContainerBehaviour);
197             },

224             /**
225              * Called from CAAT.Foundation.Director to bring in ot take away an Scene.
226              * @param time {number} time in milliseconds for the Scene to be brought in.
227              * @param alpha {boolean} whether fading will be applied to the Scene.
228              * @param anchor {number} Scene switch anchor.
229              * @param interpolator {CAAT.Behavior.Interpolator} how to apply to the Scene transition.
230              * @param starttime {number} scene time milliseconds from which the behavior will be applied.
231              * @param isIn boolean indicating whether the Scene is being brought in.
232              */
233             easeScale:function (starttime, time, alpha, anchor, isIn, interpolator) {
234                 this.easeContainerBehaviour = new CAAT.Behavior.ContainerBehavior();
235 
236                 var x = 0;
237                 var y = 0;
238                 var x2 = 0;
239                 var y2 = 0;
240 
241                 switch (anchor) {
242                     case CAAT.Foundation.Actor.ANCHOR_TOP_LEFT:
243                     case CAAT.Foundation.Actor.ANCHOR_TOP_RIGHT:
244                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM_LEFT:
245                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM_RIGHT:
246                     case CAAT.Foundation.Actor.ANCHOR_CENTER:
247                         x2 = 1;
248                         y2 = 1;
249                         break;
250                     case CAAT.Foundation.Actor.ANCHOR_TOP:
251                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM:
252                         x = 1;
253                         x2 = 1;
254                         y = 0;
255                         y2 = 1;
256                         break;
257                     case CAAT.Foundation.Actor.ANCHOR_LEFT:
258                     case CAAT.Foundation.Actor.ANCHOR_RIGHT:
259                         y = 1;
260                         y2 = 1;
261                         x = 0;
262                         x2 = 1;
263                         break;
264                     default:
265                         alert('scale anchor ?? ' + anchor);
266                 }
267 
268                 if (!isIn) {
269                     var tmp;
270                     tmp = x;
271                     x = x2;
272                     x2 = tmp;
273 
274                     tmp = y;
275                     y = y2;
276                     y2 = tmp;
277                 }
278 
279                 if (alpha) {
280                     this.createAlphaBehaviour(time, isIn);
281                 }
282 
283                 var anchorPercent = this.getAnchorPercent(anchor);
284                 var sb = new CAAT.Behavior.ScaleBehavior().
285                     setFrameTime(starttime, time).
286                     setValues(x, x2, y, y2, anchorPercent.x, anchorPercent.y);
287 
288                 if (interpolator) {
289                     sb.setInterpolator(interpolator);
290                 }
291 
292                 this.easeContainerBehaviour.addBehavior(sb);
293 
294                 this.easeContainerBehaviour.setFrameTime(this.time, time);
295                 this.easeContainerBehaviour.addListener(this);
296 
297                 this.emptyBehaviorList();
298                 CAAT.Foundation.Scene.superclass.addBehavior.call(this, this.easeContainerBehaviour);

331             /**
332              * Called from CAAT.Director to use Rotations for taking away or bringing Scenes in.
333              * @param time integer indicating time in milliseconds for the Scene to be taken away or brought in.
334              * @param alpha boolean indicating whether fading will be applied to the Scene.
335              * @param anchor integer indicating the Scene switch anchor.
336              * @param interpolator {CAAT.Interpolator} a CAAT.Interpolator to apply to the Scene transition.
337              * @param isIn boolean indicating whehter the Scene is brought in.
338              */
339             easeRotation:function (time, alpha, anchor, isIn, interpolator) {
340                 this.easeContainerBehaviour = new CAAT.Behavior.ContainerBehavior();
341 
342                 var start = 0;
343                 var end = 0;
344 
345                 if (anchor == CAAT.Foundation.Actor.ANCHOR_CENTER) {
346                     anchor = CAAT.Foundation.Actor.ANCHOR_TOP;
347                 }
348 
349                 switch (anchor) {
350                     case CAAT.Foundation.Actor.ANCHOR_TOP:
351                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM:
352                     case CAAT.Foundation.Actor.ANCHOR_LEFT:
353                     case CAAT.Foundation.Actor.ANCHOR_RIGHT:
354                         start = Math.PI * (Math.random() < 0.5 ? 1 : -1);
355                         break;
356                     case CAAT.Foundation.Actor.ANCHOR_TOP_LEFT:
357                     case CAAT.Foundation.Actor.ANCHOR_TOP_RIGHT:
358                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM_LEFT:
359                     case CAAT.Foundation.Actor.ANCHOR_BOTTOM_RIGHT:
360                         start = Math.PI / 2 * (Math.random() < 0.5 ? 1 : -1);
361                         break;
362                     default:
363                         alert('rot anchor ?? ' + anchor);
364                 }
365 
366                 if (false === isIn) {
367                     var tmp = start;
368                     start = end;
369                     end = tmp;
370                 }
371 
372                 if (alpha) {
373                     this.createAlphaBehaviour(time, isIn);
374                 }
375 
376                 var anchorPercent = this.getAnchorPercent(anchor);
377                 var rb = new CAAT.Behavior.RotateBehavior().
378                     setFrameTime(0, time).
379                     setValues(start, end, anchorPercent.x, anchorPercent.y);
380 
381                 if (interpolator) {
382                     rb.setInterpolator(interpolator);
383                 }
384                 this.easeContainerBehaviour.addBehavior(rb);
385                 this.easeContainerBehaviour.setFrameTime(this.time, time);
386                 this.easeContainerBehaviour.addListener(this);
387 
388                 this.emptyBehaviorList();
389                 CAAT.Foundation.Scene.superclass.addBehavior.call(this, this.easeContainerBehaviour);
390             },

435             /**
436              * Find a pointed actor at position point.
437              * This method tries lo find the correctly pointed actor in two different ways.
438              *  + first of all, if inputList is defined, it will look for an actor in it.
439              *  + if no inputList is defined, it will traverse the scene graph trying to find a pointed actor.
440              * @param point <CAAT.Point>
441              */
442             findActorAtPosition:function (point) {
443                 var i, j;
444 
445                 var p = new CAAT.Math.Point();
446 
447                 if (this.inputList) {
448                     var il = this.inputList;
449                     for (i = 0; i < il.length; i++) {
450                         var ill = il[i];
451                         for (j = 0; j < ill.length; j++) {
452                             if ( ill[j].visible ) {
453                                 p.set(point.x, point.y);
454                                 var modelViewMatrixI = ill[j].worldModelViewMatrix.getInverse();
455                                 modelViewMatrixI.transformCoord(p);
456                                 if (ill[j].contains(p.x, p.y)) {
457                                     return ill[j];
458                                 }
459                             }
460                         }
461                     }
462                 }
463 
464                 p.set(point.x, point.y);
465                 return CAAT.Foundation.Scene.superclass.findActorAtPosition.call(this, p);
466             },

468             /**
469              * Enable a number of input lists.
470              * These lists are set in case the developer doesn't want the to traverse the scene graph to find the pointed
471              * actor. The lists are a shortcut whete the developer can set what actors to look for input at first instance.
472              * The system will traverse the whole lists in order trying to find a pointed actor.
473              *
474              * Elements are added to each list either in head or tail.
475              *
476              * @param size <number> number of lists.
477              */

487             /**
488              * Add an actor to a given inputList.
489              * @param actor <CAAT.Actor> an actor instance
490              * @param index <number> the inputList index to add the actor to. This value will be clamped to the number of
491              * available lists.
492              * @param position <number> the position on the selected inputList to add the actor at. This value will be
493              * clamped to the number of available lists.
494              */
495             addActorToInputList:function (actor, index, position) {
496                 if (index < 0) index = 0; else if (index >= this.inputList.length) index = this.inputList.length - 1;
497                 var il = this.inputList[index];
498 
499                 if (typeof position === "undefined" || position >= il.length) {
500                     il.push(actor);
501                 } else if (position <= 0) {
502                     il.unshift(actor);
503                 } else {
504                     il.splice(position, 0, actor);
505                 }
506 
507                 return this;
508             },

527              * available lists.
528              */

530                 if (typeof index === "undefined") {
531                     var i, j;
532                     for (i = 0; i < this.inputList.length; i++) {
533                         var il = this.inputList[i];
534                         for (j = 0; j < il.length; j++) {
535                             if (il[j] == actor) {
536                                 il.splice(j, 1);
537                             }
538                         }
539                     }
540                     return this;
541                 }
542 
543                 if (index < 0) index = 0; else if (index >= this.inputList.length) index = this.inputList.length - 1;
544                 var il = this.inputList[index];
545                 for (j = 0; j < il.length; j++) {
546                     if (il[j] == actor) {
547                         il.splice(j, 1);
548                     }
549                 }
550 
551                 return this;
552             },
