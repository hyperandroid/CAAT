  4  * Manages every Actor affine transformations.
  5  * Take into account that Canvas' renderingContext computes postive rotation clockwise, so hacks
  6  * to handle it properly are hardcoded.

  8  * Contained classes are CAAT.Matrix and CAAT.MatrixStack.

 14     /**
 15      * 2D affinetransform matrix represeantation.
 16      * It includes matrices for
 17      * <ul>
 18      *  <li>Rotation by any anchor point
 19      *  <li>translation
 20      *  <li>scale by any anchor point
 21      * </ul>
 22      *
 23      * <p>
 24      * Each matrix knows its own type to speed transformations up.
 25      */
 26 	CAAT.Matrix = function() {
 27 		this.matrix= [ [1,0,0],
 28 		               [0,1,0],
 29 		               [0,0,1] ];
 30 		return this;
 31 	};
 32 	
 33 	CAAT.Matrix.prototype= {
 34 		matrix:	null,
 35 		type : false,
 36 		
 37 		TYPE_ROTATE:	0,
 38 		TYPE_SCALE:		1,
 39 		TYPE_TRANSLATE:	2,
 40 
 41         /**
 42          * Transform a point by this matrix. The parameter point will be modified with the transformation values.
 43          * @param point {CAAT.Point}.
 44          * @return {CAAT.Point} the parameter point.
 45          */
 46 		transform : function(point) {
 47 			var x= point.x;
 48 			var y= point.y;
 49 			
 50 			if ( this.type==this.TYPE_ROTATE ) {
 51 				point.x= x*this.matrix[0][0] + y*this.matrix[1][0] + this.matrix[0][2];
 52 				point.y= x*this.matrix[0][1] + y*this.matrix[1][1] + this.matrix[1][2];
 53 				
 54 			} else if ( this.type==this.TYPE_SCALE ) {
 55 				point.x*= this.matrix[0][0];
 56 				point.y*= this.matrix[1][1];
 57 			} else if ( this.type==this.TYPE_TRANSLATE ) {
 58 				point.x+= this.matrix[0][2];
 59 				point.y+= this.matrix[1][2];				
 60 			}
 61 			else {
 62 				point.x= x*this.matrix[0][0] + y*this.matrix[0][1] + this.matrix[0][2];
 63 				point.y= x*this.matrix[1][0] + y*this.matrix[1][1] + this.matrix[1][2];
 64 			}
 65 			return point;
 66 		},
 67         /**
 68          * Create a new rotation matrix and set it up for the specified angle in radians.
 69          * @param angle {number}
 70          * @return {CAAT.Matrix} a matrix object.
 71          *
 72          * @static
 73          */
 74 		rotate : function(angle) {
 75 			var m= new CAAT.Matrix();
 76 			
 77 			m.matrix[0][0]= Math.cos(angle);
 78 			m.matrix[1][0]= -Math.sin(angle);
 79 			
 80 			m.matrix[0][1]= Math.sin(angle);
 81 			m.matrix[1][1]= Math.cos(angle);
 82 			
 83 			m.type= this.TYPE_ROTATE;
 84 			
 85 			return m;
 86 		},
 87         /**
 88          * Create a new rotation matrix by an arbitrary anchor point and set it up for the specified angle in radians.
 89          * @param angle {number}
 90          * @param tx {number} translation x coordinate.
 91          * @param ty {number} translation y coordinate.
 92          *
 93          * @return {CAAT.Matrix} a matrix object.
 94          *
 95          * @static
 96          */
 97 		_rotateAnchor : function(angle, tx, ty) {
 98 			var m= new CAAT.Matrix();
 99 			
100 			m.matrix[0][0]= Math.cos(angle);
101 			m.matrix[1][0]= -Math.sin(angle);
102 			
103 			m.matrix[0][1]= Math.sin(angle);
104 			m.matrix[1][1]= Math.cos(angle);
105 			
106 			m.matrix[0][2]= tx;
107 			m.matrix[1][2]= ty;
108 			
109 			m.type= this.TYPE_ROTATE;
110 			
111 			return m;
112 		},
113         /**
114          * Create a scale matrix.
115          * @param scalex {number} x scale magnitude.
116          * @param scaley {number} y scale magnitude.
117          *
118          * @return {CAAT.Matrix} a matrix object.
119          *
120          * @static
121          */
122 		scale : function(scalex, scaley) {
123 			var m= new CAAT.Matrix();
124 			
125 			m.matrix[0][0]= scalex;
126 			m.matrix[1][1]= scaley;
127 			
128 			m.type= this.TYPE_SCALE;
129 			
130 			return m;
131 		},
132         /**
133          * Create a translation matrix.
134          * @param x {number} x translation magnitude.
135          * @param y {number} y translation magnitude.
136          *
137          * @return {CAAT.Matrix} a matrix object.
138          * @static
139          *
140          */
141 		translate : function( x, y ) {
142 			var m= new CAAT.Matrix();
143 			
144 			m.matrix[0][2]= x;
145 			m.matrix[1][2]= y;
146 			
147 			m.type= this.TYPE_TRANSLATE;
148 			
149 			return m;
150 		},
151         /**
152          * Copy into this matrix the given matrix values.
153          * @param matrix {CAAT.Matrix}
154          * @return this
155          */
156 		copy : function( matrix ) {
157 			this.matrix[0][0]= matrix[0][0];
158 			this.matrix[0][1]= matrix[0][1];
159 			this.matrix[0][2]= matrix[0][2];
160 			this.matrix[1][0]= matrix[1][0];
161 			this.matrix[1][1]= matrix[1][1];
162 			this.matrix[1][2]= matrix[1][2];
163 			this.matrix[2][0]= matrix[2][0];
164 			this.matrix[2][1]= matrix[2][1];
165 			this.matrix[2][2]= matrix[2][2];
166 
167             return this;
168 		},
169         /**
170          * Set this matrix to the identity matrix.
171          * @return this
172          */
173 		identity : function() {
174 			this.matrix[0][0]= 1;
175 			this.matrix[0][1]= 0;
176 			this.matrix[0][2]= 0;
177 			
178 			this.matrix[1][0]= 0;
179 			this.matrix[1][1]= 1;
180 			this.matrix[1][2]= 0;
181 			
182 			this.matrix[2][0]= 0;
183 			this.matrix[2][1]= 0;
184 			this.matrix[2][2]= 1;
185 
186             return this;
187 		},
188         /**
189          * Multiply this matrix by a given matrix.
190          * @param m {CAAT.Matrix}
191          * @return this
192          */
193 		multiply : function( m ) {
194 			var m00= this.matrix[0][0]*m.matrix[0][0] + this.matrix[0][1]*m.matrix[1][0] + this.matrix[0][2]*m.matrix[2][0];
195 			var m01= this.matrix[0][0]*m.matrix[0][1] + this.matrix[0][1]*m.matrix[1][1] + this.matrix[0][2]*m.matrix[2][1];
196 			var m02= this.matrix[0][0]*m.matrix[0][2] + this.matrix[0][1]*m.matrix[1][2] + this.matrix[0][2]*m.matrix[2][2];
197 
198 			var m10= this.matrix[1][0]*m.matrix[0][0] + this.matrix[1][1]*m.matrix[1][0] + this.matrix[1][2]*m.matrix[2][0];
199 			var m11= this.matrix[1][0]*m.matrix[0][1] + this.matrix[1][1]*m.matrix[1][1] + this.matrix[1][2]*m.matrix[2][1];
200 			var m12= this.matrix[1][0]*m.matrix[0][2] + this.matrix[1][1]*m.matrix[1][2] + this.matrix[1][2]*m.matrix[2][2];
201 			
202 			var m20= this.matrix[2][0]*m.matrix[0][0] + this.matrix[2][1]*m.matrix[1][0] + this.matrix[2][2]*m.matrix[2][0];
203 			var m21= this.matrix[2][0]*m.matrix[0][1] + this.matrix[2][1]*m.matrix[1][1] + this.matrix[2][2]*m.matrix[2][1];
204 			var m22= this.matrix[2][0]*m.matrix[0][2] + this.matrix[2][1]*m.matrix[1][2] + this.matrix[2][2]*m.matrix[2][2];
205 
206 			this.matrix[0][0]= m00;
207 			this.matrix[0][1]= m01;
208 			this.matrix[0][2]= m02;
209 
210 			this.matrix[1][0]= m10;
211 			this.matrix[1][1]= m11;
212 			this.matrix[1][2]= m12;
213 
214 			this.matrix[2][0]= m20;
215 			this.matrix[2][1]= m21;
216 			this.matrix[2][2]= m22;
217 
218             return this;
219 		},
220         /**
221          * Premultiply this matrix by a given matrix.
222          * @param m {CAAT.Matrix}
223          * @return this
224          */
225 		premultiply : function(m) {
226 			var m00= m.matrix[0][0]*this.matrix[0][0] + m.matrix[0][1]*this.matrix[1][0] + m.matrix[0][2]*this.matrix[2][0];
227 			var m01= m.matrix[0][0]*this.matrix[0][1] + m.matrix[0][1]*this.matrix[1][1] + m.matrix[0][2]*this.matrix[2][1];
228 			var m02= m.matrix[0][0]*this.matrix[0][2] + m.matrix[0][1]*this.matrix[1][2] + m.matrix[0][2]*this.matrix[2][2];
229 
230 			var m10= m.matrix[1][0]*this.matrix[0][0] + m.matrix[1][1]*this.matrix[1][0] + m.matrix[1][2]*this.matrix[2][0];
231 			var m11= m.matrix[1][0]*this.matrix[0][1] + m.matrix[1][1]*this.matrix[1][1] + m.matrix[1][2]*this.matrix[2][1];
232 			var m12= m.matrix[1][0]*this.matrix[0][2] + m.matrix[1][1]*this.matrix[1][2] + m.matrix[1][2]*this.matrix[2][2];
233 			
234 			var m20= m.matrix[2][0]*this.matrix[0][0] + m.matrix[2][1]*this.matrix[1][0] + m.matrix[2][2]*this.matrix[2][0];
235 			var m21= m.matrix[2][0]*this.matrix[0][1] + m.matrix[2][1]*this.matrix[1][1] + m.matrix[2][2]*this.matrix[2][1];
236 			var m22= m.matrix[2][0]*this.matrix[0][2] + m.matrix[2][1]*this.matrix[1][2] + m.matrix[2][2]*this.matrix[2][2];
237 
238 			this.matrix[0][0]= m00;
239 			this.matrix[0][1]= m01;
240 			this.matrix[0][2]= m02;
241 
242 			this.matrix[1][0]= m10;
243 			this.matrix[1][1]= m11;
244 			this.matrix[1][2]= m12;
245 
246 			this.matrix[2][0]= m20;
247 			this.matrix[2][1]= m21;
248 			this.matrix[2][2]= m22;
249 
250             return this;
251 		},
252         /**
253          * Creates a new inverse matrix from this matrix.
254          * @return {CAAT.Matrix} an inverse matrix.
255          */
256 	    getInverse : function() {
257 			
258 			var tx=  this.matrix[0][2];
259 			var ty=  this.matrix[1][2];
260 			
261 			var m00= this.matrix[0][0];
262 			var m01= this.matrix[0][1];
263 			var m10= this.matrix[1][0];
264 			var m11= this.matrix[1][1];
265 
266 			var newMatrix= new CAAT.Matrix();
267 			if ( this.type==this.TYPE_ROTATE ) {
268 				newMatrix.matrix[0][0]= m00;
269 				newMatrix.matrix[0][1]= m10;
270 				newMatrix.matrix[1][0]= m01;
271 				newMatrix.matrix[1][1]= m11;
272 			} else if ( this.type==this.TYPE_SCALE ) {
273 				newMatrix.matrix[0][0]= 1/m00;
274 				newMatrix.matrix[1][1]= 1/m11;
275 			} else if ( this.type==this.TYPE_TRANSLATE ) {
276 				newMatrix.matrix[0][2]= -tx; //tx*newMatrix.matrix[0][0] + ty*newMatrix.matrix[0][1];
277 				newMatrix.matrix[1][2]= -ty; //tx*newMatrix.matrix[1][0] + ty*newMatrix.matrix[1][1];
278 			}
279 			
280 			newMatrix.type= this.type;
281 
282 			return newMatrix;
283 	    }

287 (function() {
288     /**
289      * Implementation of a matrix stack. Each CAAT.Actor instance contains a MatrixStack to hold of its affine
290      * transformations. The Canvas rendering context will be fed with this matrix stack values to keep a homogeneous
291      * transformation process.
292      *
293      * @constructor
294      */
295 	CAAT.MatrixStack= function() {
296 		this.matrix= new CAAT.Matrix();
297 		this.stack= [];
298 		this.saved= [];
299 		return this;
300 	};
301 
302 	CAAT.MatrixStack.prototype= {
303 		matrix:	null,
304 		stack: null,
305 		saved: null,
306 		
307         /**
308          * Add a matrix to the transformation stack.
309          * @return this
310          */
311 		pushMatrix : function(matrix) {
312 			this.stack.push(matrix);
313             return this;
314 		},
315         /**
316          * Remove the last matrix from this stack.
317          * @return {CAAT.Matrix} the poped matrix.
318          */
319 		popMatrix : function()	{
320 			return this.stack.pop();
321 		},
322         /**
323          * Create a restoration point of pushed matrices.
324          * @return this
325          */
326 		save : function() {
327 			this.saved.push(this.stack.length);
328             return this
329 		},
330         /**
331          * Restore from the last restoration point set.
332          * @return this
333          */
334 		restore : function() {
335 			var pos= this.saved.pop();
336 			while( this.stack.length!=pos ) {
337 				this.popMatrix();
338 			}
339             return this;
340 		},
341         /**
342          * Set the canvas rendering context up with the transformations held in this matrix stack.
343          * @param ctx a canvas rendering context.
344          * @return this
345          */
346 		transform : function(ctx) {
347 
348 			for( var i=0; i<this.stack.length; i++ ) {
349 				var m= this.stack[i].matrix;
350 				ctx.transform( m[0][0], m[0][1], m[1][0], m[1][1], m[0][2], m[1][2] );
351 			}
352             return this;
353 		},
354         /**
355          * Set the canvas rendering context up with the transformation held in this matrix stack but in inverse order.
356          * @param ctx a canvas rendering context
357          * @return this
358          */
359 		transformInverse : function(ctx) {
360 			
361 			for( var i=this.stack.length-1; i>=0; i-- ) {
362 				var im= this.stack[i].getInverse();
363 				ctx.transform( 	 
364 						im.matrix[0][0], im.matrix[0][1], 
365 						im.matrix[1][0], im.matrix[1][1],
366 						im.matrix[0][2], im.matrix[1][2] );
367 			}
368 
369             return this;
370 		},
371         /**
372          * Set a canvas rendering context up for the transformations an actor has set.
373          * @param canvas a canvas rendering context
374          * @param actor {CAAT.Actor} an scene graph actor.
375          * @return this
376          */
377 		prepareGraphics : function(canvas, actor) {
378 
379             this.stack= [];
380 
381             if ( actor.dirty ) {
382                 this.pushMatrix( CAAT.Matrix.prototype.translate( actor.x, actor.y) );
383 
384                 if ( actor.rotationAngle!=0 ) {
385                     this.pushMatrix( CAAT.Matrix.prototype.translate( actor.rotationX, actor.rotationY) );
386                     this.pushMatrix( CAAT.Matrix.prototype.rotate( actor.rotationAngle ) );
387                     this.pushMatrix( CAAT.Matrix.prototype.translate( -actor.rotationX, -actor.rotationY) );
388                 }
389 
390                 if ( actor.scaleX!=1 || actor.scaleY!=1 ) {
391                     if ( actor.scaleX!=0 && actor.scaleY!=0 ) {
392                         this.pushMatrix( CAAT.Matrix.prototype.translate( actor.scaleTX-actor.width , actor.scaleTY-actor.height ) );
393                         this.pushMatrix( CAAT.Matrix.prototype.scale( actor.scaleX, actor.scaleY ) );
394                         this.pushMatrix( CAAT.Matrix.prototype.translate(
395                                 -(actor.scaleTX - actor.width / actor.scaleX),
396                                 -(actor.scaleTY - actor.height / actor.scaleY) ) ) ;
397                     } else {
398                         this.pushMatrix( CAAT.Matrix.prototype.scale( 0.01,0.01 ) );
399                     }
400                 }
401             }
402 
403             if ( this.oldX!=actor.x || this.oldY!=actor.y ) {
404                 this.stack[0].matrix[0][2]= actor.x;
405                 this.stack[0].matrix[1][2]= actor.y;
406             }
407 
408             this.oldX= actor.x;
409             this.oldY= actor.y;
410 
411 			this.transform(canvas);
412 
413             return this;
414 		},
415         /**
416          * Transform a coordinate by this matrix stack.
417          * @param point {CAAT.Point} a coordinate to transform. The point value will be overwritten by the transformation
418          * result.
419          *
420          * @return {CAAT.Point}
421          */
422 		transformCoord : function(point) {
423 
424 			//for( var i=0; i<this.stack.length; i++ ) {
425 			for( var i=this.stack.length-1; i>=0; i-- ) {
426 				var matrix= this.stack[i];
427 				matrix.transform(point);
428 			}
429 			
430 			return point;
431 		},
432         /**
433          * Transform a coordinate by this matrix stack but not applying translation.
434          * @param point {CAAT.Point} a coordinate to transform. The point value will be overwritten by the transformation
435          * result.
436          *
437          * @return {CAAT.Point}
438          */
439 		transformCoordNoTranslate : function(point) {
440 
441 			//for( var i=0; i<this.stack.length; i++ ) {
442 			for( var i=this.stack.length-1; i>=1; i-- ) {
443 				var matrix= this.stack[i];
444 				matrix.transform(point);
445 			}
446 
447 			return point;
448 		},
449         /**
450          * Transform a coordinate by this matrix stack but in inverse order.
451          * @param point {CAAT.Point} a coordinate to transform. The point value will be overwritten by the transformation
452          * result.
453          *
454          * @return {CAAT.Point}
455          */
456         inverseTransformCoord : function(point) {
457 
458 			for( var i=0; i<this.stack.length; i++ ) {
459 				var matrix= this.stack[i].getInverse();
460 				matrix.transform(point);
461 			}
462 			
463 			return point;
464 		},
465         /**
466          * Return the concatenation (multiplication) matrix of all the matrices contained in this stack.
467          * @return {CAAT.Matrix} a new matrix.
468          */
469         getMatrix : function() {
470             var matrix= new CAAT.Matrix();
471 
472 			for( var i=0; i<this.stack.length; i++ ) {
473 				var matrixStack= this.stack[i];
474                 matrix.multiply( matrixStack );
475             }
476 
477             return matrix;
478         }
479 	};
