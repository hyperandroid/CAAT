1 /**
  2  * @author  Hyperandroid  ||  http://hyperandroid.com/
  3  *
  4  * Rectangle Class.
  5  * Needed to compute Curve bounding box.
  6  * Needed to compute Actor affected area on change.
  7  *
  8  **/
  9 
 10 
 11 (function() {
 12     /**
 13      * A Rectangle implementation, which defines an area positioned somewhere.
 14      *
 15      * @constructor
 16      */
 17 	CAAT.Rectangle= function() {
 18 		return this;
 19 	};
 20 	
 21 	CAAT.Rectangle.prototype= {
 22 		x:		0,
 23 		y:		0,
 24 		x1:		0,
 25 		y1:		0,
 26 		width:	0,
 27 		height:	0,
 28 
 29         /**
 30          * Set this rectangle's location.
 31          * @param x {number}
 32          * @param y {number}
 33          */
 34         setLocation: function( x,y ) {
 35             this.x= x;
 36             this.y= y;
 37             this.x1= this.x+this.width;
 38             this.y1= this.y+this.height;
 39             return this;
 40         },
 41         /**
 42          * Set this rectangle's dimension.
 43          * @param w {number}
 44          * @param h {number}
 45          */
 46         setDimension : function( w,h ) {
 47             this.width= w;
 48             this.height= h;
 49             this.x1= this.x+this.width;
 50             this.y1= this.y+this.height;
 51             return this;
 52         },
 53         /**
 54          * Return whether the coordinate is inside this rectangle.
 55          * @param px {number}
 56          * @param py {number}
 57          *
 58          * @return {boolean}
 59          */
 60 		contains : function(px,py) {
 61 			return px>=0 && px<this.width && py>=0 && py<this.height; 
 62 		},
 63         /**
 64          * Return whether this rectangle is empty, that is, has zero dimension.
 65          * @return {boolean}
 66          */
 67 		isEmpty : function() {
 68 			return this.width===0 && this.height===0;
 69 		},
 70         /**
 71          * Set this rectangle as the union of this rectangle and the given point.
 72          * @param px {number}
 73          * @param py {number}
 74          */
 75 		union : function(px,py) {
 76 			
 77 			if ( this.isEmpty() && this.x==0 && this.y==0 ) {
 78 				this.x= px;
 79 				this.y= py;
 80 				return;
 81 			}
 82 			
 83 			this.x1= this.x+this.width;
 84 			this.y1= this.y+this.height;
 85 			
 86 			if ( py<this.y ) {
 87 				this.y= py;
 88 			}
 89 			if ( px<this.x ) {
 90 				this.x= px;
 91 			}
 92 			if ( py>this.y1 ) {
 93 				this.y1= py;
 94 			}
 95 			if ( px>this.x1 ){
 96 				this.x1= px;
 97 			}
 98 			
 99 			this.width= this.x1-this.x;
100 			this.height= this.y1-this.y;
101 		}
102 	};
103 })();
