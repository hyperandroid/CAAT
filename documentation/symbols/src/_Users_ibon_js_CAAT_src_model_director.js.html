  8     /**
  9      * Director is the animator scene graph manager.
 10      * <p>
 11      * The director elements is an ActorContainer itself with the main responsibility of managing
 12      * different Scenes.
 13      * <p>
 14      * It is responsible for:
 15      * <ul>
 16      * <li>scene changes.
 17      * <li>route input to the appropriate scene graph actor.
 18      * <li>be the central point for resource caching.
 19      * <li>manage the timeline.
 20      * <li>manage frame rate.
 21      * <li>etc.
 22      * </ul>
 23      *
 24      * <p>
 25      * One document can contain different CAAT.Director instances which will be kept together in CAAT
 26      * function.
 27      *
 28      * @constructor
 29      * @extends CAAT.ActorContainer
 30      */

 35         this.audioManager=  new CAAT.AudioManager().initialize(8);
 36         this.scenes=        [];
 37 
 38         return this;
 39 	};

 60         /**
 61          * This method performs Director initialization. Must be called once.
 62          * If the canvas parameter is not set, it will create a Canvas itself,
 63          * and the developer must explicitly add the canvas to the desired DOM position.
 64          * This method will also set the Canvas dimension to the specified values

105         /**
106          * Creates an initializes a Scene object.

114         /**
115          * This is the entry point for the animation system of the Director.
116          * The director is fed with the elapsed time value to maintain a virtual timeline.
117          * This virtual timeline will provide each Scene with its own virtual timeline, and will only
118          * feed time when the Scene is the current Scene, or is being switched.
119          *
120          * @param time {number} integer indicating the elapsed time between two consecutive frames of the
121          * Director.
122          */
123         render : function(time) {
124 			this.time+= time;
125 			this.crc.globalAlpha=1;
126             this.crc.globalCompositeOperation='source-over';
127 
128             if ( this.clear ) {
129                 this.crc.clearRect(0,0,this.width,this.height);
130             }
131 
132 			this.crc.setTransform(1,0,0,1,0,0);
133 
134             /**
135              * calculate animable elements and their bbox.
136              */
137             var i;
138             var tt;
139 
140 			for( i=0; i<this.childrenList.length; i++ ) {
141 				if (this.childrenList[i].isInAnimationFrame(this.time)) {
142                     tt= this.childrenList[i].time - this.childrenList[i].start_time;
143 					this.childrenList[i].animate(this, tt);
144 				}
145 			}
146 
147             /**
148              * draw actors on scene.
149              */
150 			for( i=0; i<this.childrenList.length; i++ ) {
151 				if (this.childrenList[i].isInAnimationFrame(this.time)) {
152 					this.crc.save();
153                     tt= this.childrenList[i].time - this.childrenList[i].start_time;
154 					this.childrenList[i].paintActor(this, tt);
155                     this.childrenList[i].time+= time;
156 					this.crc.restore();
157 
158                     if ( this.debug ) {
159                         this.childrenList[i].drawScreenBoundingBox(this, tt);
160                     }
161 				}
162 			}
163 
164             this.endAnimate(this,time);
165 
166 		},
167         /**
168          * This method draws an Scene to an offscreen canvas. This offscreen canvas is also a child of
169          * another Scene (transitionScene). So instead of drawing two scenes while transitioning from one to another,
170          * first of all an scene is drawn to offscreen, and that image is translated.
171          * <p>
172          * Until the creation of this method, both scenes where drawn while transitioning with its performance
173          * penalty since drawing two scenes could be twice as expensive than drawing only one.
174          * <p>
175          * Though a high performance increase, we should keep an eye on memory consumption.
176          *
177          * @param ctx a <code>canvas.getContext('2d')</code> instnce.
178          * @param scene {CAAT.Scene} the scene to draw offscreen.
179          */
180         renderToContext : function( ctx, scene ) {
181             ctx.globalAlpha=1;
182             ctx.globalCompositeOperation='source-over';
183 
184             ctx.clearRect(0,0,this.width,this.height);
185 
186             ctx.setTransform(1,0,0,1,0,0);
187 
188             var octx= this.ctx;
189             var ocrc= this.crc;
190 
191             this.ctx= this.crc= ctx;
192 
193             /**
194              * draw actors on scene.
195              */
196             if (scene.isInAnimationFrame(this.time)) {
197                 ctx.save();
198                 scene.paintActor(this, scene.time - scene.start_time);
199                 ctx.restore();
200             }
201 
202             this.ctx= octx;
203             this.crc= ocrc;
204         },

233         /**
234          * This method offers full control over the process of switching between any given two Scenes.
235          * To apply this method, you must specify the type of transition to apply for each Scene and
236          * the anchor to keep the Scene pinned at.
237          * <p>
238          * The type of transition will be one of the following values defined in CAAT.Scene.prototype:
239          * <ul>
240          *  <li>EASE_ROTATION
241          *  <li>EASE_SCALE
242          *  <li>EASE_TRANSLATION
243          * </ul>
244          *
245          * <p>
246          * The anchor will be any of these values defined in CAAT.Actor.prototype:
247          * <ul>
248          *  <li>ANCHOR_CENTER
249 		 *  <li>ANCHOR_TOP
250 		 *  <li>ANCHOR_BOTTOM
251 		 *  <li>ANCHOR_LEFT
252          *  <li>ANCHOR_RIGHT
253 		 *  <li>ANCHOR_TOP_LEFT
254 		 *  <li>ANCHOR_TOP_RIGHT
255 		 *  <li>ANCHOR_BOTTOM_LEFT
256 		 *  <li>ANCHOR_BOTTOM_RIGHT
257          * </ul>
258          *
259          * <p>
260          * In example, for an entering scene performing a EASE_SCALE transition, the anchor is the
261          * point by which the scene will scaled.
262          *
263          * @param inSceneIndex integer indicating the Scene index to bring in to the Director.
264          * @param typein integer indicating the type of transition to apply to the bringing in Scene.
265          * @param anchorin integer indicating the anchor of the bringing in Scene.
266          * @param outSceneIndex integer indicating the Scene index to take away from the Director.
267          * @param typeout integer indicating the type of transition to apply to the taking away in Scene.
268          * @param anchorout integer indicating the anchor of the taking away Scene.
269          * @param time inteter indicating the time to perform the process of switchihg between Scene object
270          * in milliseconds.
271          * @param alpha boolean boolean indicating whether alpha transparency fading will be applied to
272          * the scenes.
273          * @param interpolatorIn CAAT.Interpolator object to apply to entering scene.
274          * @param interpolatorOut CAAT.Interpolator object to apply to exiting scene.
275          */
276 		easeInOut : function( inSceneIndex, typein, anchorin, outSceneIndex, typeout, anchorout, time, alpha, interpolatorIn, interpolatorOut ) {
277 
278             if ( inSceneIndex==this.getCurrentSceneIndex() ) {
279                 return;
280             }
281 
282 
283 
284 			var ssin=this.scenes[ inSceneIndex ];
285 			var sout=this.scenes[ outSceneIndex ];
286 
287             this.renderToContext( this.transitionScene.ctx, sout );
288 
289             sout=this.transitionScene;
290 
291             ssin.setExpired(false);
292             sout.setExpired(false);
293 
294             ssin.mouseEnabled= false;
295             sout.mouseEnabled= false;
296 
297 			ssin.resetTransform();
298 			sout.resetTransform();
299 
300             ssin.setLocation(0,0);
301             sout.setLocation(0,0);
302 
303             ssin.alpha = 1;
304             sout.alpha = 1;
305 
306 			if (typein==CAAT.Scene.prototype.EASE_ROTATION) {
307 				ssin.easeRotationIn(time, alpha, anchorin, interpolatorIn );
308 			} else if (typein==CAAT.Scene.prototype.EASE_SCALE) {
309 				ssin.easeScaleIn(0,time, alpha, anchorin, interpolatorIn );
310 			} else {
311                 ssin.easeTranslationIn(time,alpha,anchorin,interpolatorIn );
312             }
313 
314 			if (typeout==CAAT.Scene.prototype.EASE_ROTATION) {
315 				sout.easeRotationOut(time, alpha, anchorout, interpolatorOut );
316 			} else if (typeout==CAAT.Scene.prototype.EASE_SCALE) {
317 				sout.easeScaleOut(0,time, alpha, anchorout, interpolatorOut );
318 			} else {
319                 sout.easeTranslationOut(time,alpha,anchorout, interpolatorOut);
320             }
321 
322 			this.childrenList= [];
323 
324 			this.addChild(sout);
325 			this.addChild(ssin);
326 		},
327         /**
328          * This method will switch between two given Scene indexes (ie, take away scene number 2,
329          * and bring in scene number 5).
330          * <p>
331          * It will randomly choose for each Scene the type of transition to apply and the anchor
332          * point of each transition type.
333          * <p>
334          * It will also set for different kind of transitions the following interpolators:
335          * <ul>
336          * <li>EASE_ROTATION    -> ExponentialInOutInterpolator, exponent 4.
337          * <li>EASE_SCALE       -> ElasticOutInterpolator, 1.1 and .4
338          * <li>EASE_TRANSLATION -> BounceOutInterpolator
339          * </ul>
340          *
341          * <p>
342          * These are the default values, and could not be changed by now.
343          * This method in final instance delegates the process to easeInOutMethod.
344          *
345          * @see easeInOutMethod.
346          *
347          * @param inIndex integer indicating the entering scene index.
348          * @param outIndex integer indicating the exiting scene index.
349          * @param time integer indicating the time to take for the process of Scene in/out in milliseconds.
350          * @param alpha boolean indicating whether alpha transparency fading should be applied to transitions.
351          */
352 		easeInOutRandom : function(inIndex,outIndex,time,alpha) {
353 
354             var pin=Math.random();
355             var pout=Math.random();
356 
357             var typeIn;
358             var interpolatorIn;
359 
360             if (pin<.33) {
361                 typeIn= CAAT.Scene.prototype.EASE_ROTATION;
362                 interpolatorIn= new CAAT.Interpolator().createExponentialInOutInterpolator(4);
363             } else if (pin<.66) {
364                 typeIn= CAAT.Scene.prototype.EASE_SCALE;
365                 interpolatorIn= new CAAT.Interpolator().createElasticOutInterpolator(1.1, .4);
366             } else {
367                 typeIn= CAAT.Scene.prototype.EASE_TRANSLATE;
368                 interpolatorIn= new CAAT.Interpolator().createBounceOutInterpolator();
369             }
370 
371             var typeOut;
372             var interpolatorOut;
373 
374             if (pout<.33) {
375                 typeOut= CAAT.Scene.prototype.EASE_ROTATION;
376                 interpolatorOut= new CAAT.Interpolator().createExponentialInOutInterpolator(4);
377             } else if (pout<.66) {
378                 typeOut= CAAT.Scene.prototype.EASE_SCALE;
379                 interpolatorOut= new CAAT.Interpolator().createExponentialOutInterpolator(4);
380             } else {
381                 typeOut= CAAT.Scene.prototype.EASE_TRANSLATE;
382                 interpolatorOut= new CAAT.Interpolator().createBounceOutInterpolator();
383             }
384 
385             this.easeInOut(
386                     inIndex,
387                     typeIn,
388                     (Math.random()*8.99)>>0,
389 
390                     outIndex,
391                     typeOut,
392                     (Math.random()*8.99)>>0,
393 
394                     time,
395                     alpha,
396 
397                     interpolatorIn,
398                     interpolatorOut);
399 
400 		},
401         /**
402          * This method changes Director's current Scene to the scene index indicated by
403          * inSceneIndex parameter. The Scene running in the director won't be eased out.
404          *
405          * @see {CAAT.Interpolator}
406          * @see {CAAT.Actor}
407          * @see {CAAT.Scene}
408          *
409          * @param inSceneIndex integer indicating the new Scene to set as current.
410          * @param type integer indicating the type of transition to apply to bring the new current
411          * Scene to the Director. The values will be one of: CAAT.Scene.prototype.EASE_ROTATION,
412          * CAAT.Scene.prototype.EASE_SCALE, CAAT.Scene.prototype.EASE_TRANSLATION.
413          * @param time integer indicating how much time in milliseconds the Scene entrance will take.
414          * @param alpha boolean indicating whether alpha transparency fading will be applied to the
415          * entereing Scene.
416          * @param anchor integer indicating the anchor to fix for Scene transition. It will be any of
417          * CAAT.Actor.prototype.ANCHOR_* values.
418          * @param interpolator an CAAT.Interpolator object indicating the interpolation function to
419          * apply.
420          */
421 		easeIn : function( inSceneIndex, type, time, alpha, anchor, interpolator ) {
422 			var sin= this.scenes[ inSceneIndex ];
423 			if (type==CAAT.Scene.prototype.EASE_ROTATION) {
424 				sin.easeRotationIn(time, alpha, anchor, interpolator );
425 			} else if (type==CAAT.Scene.prototype.EASE_SCALE) {
426 				sin.easeScaleIn(0,time, alpha, anchor, interpolator );
427 			} else {
428                 sin.easeTranslationIn( time, alpha, anchor, interpolator );
429             }
430 			this.childrenList= [];
431 			this.addChild(sin);
432 
433 			sin.resetTransform();
434             sin.setLocation(0,0);
435             sin.alpha = 1;
436             sin.mouseEnabled= false;
437             sin.setExpired(false);
438 		},

479         /**

481          * @see switchToScene.
482          *

493 			}

504          * @see switchToScene.
505          *
506          * @param time {number} integer indicating the time the Scene transition will take.
507          * @param alpha {boolean} a boolean indicating whether Scene transition should be fading.
508          * @param transition {boolean} a boolean indicating whether the scene change must smoothly animated.

522 				this.easeInOutRandom( currentSceneIndex+1, currentSceneIndex, time, alpha );
523 			}

673          * context are available). A value of 60 is a high frame rate and should not be exceeded.

675          * @param fps {number} integer value indicating the target frames per second to run
676          * the animation at.
677          */

683             var floop= function loop() {
684                 var t= new Date().getTime(),
685 					delta = t - me.timeline;
686 
687                 me.render( delta );
688                 me.timeline= t;
689 
690 				if(callback) {
691 					callback(me, delta);
692 				}
693             };
694 
695             floop();
696             setInterval( floop, fps);
697
